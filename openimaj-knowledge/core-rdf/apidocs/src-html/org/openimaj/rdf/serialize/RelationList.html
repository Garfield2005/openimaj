030package org.openimaj.rdf.serialize;
031
032import java.lang.annotation.Retention;
033import java.lang.annotation.RetentionPolicy;
034
035/**
036 *      An annotation that marks a member of a class that must be an instance of
037 *      Collection<IndependentPair<URI,Object>>
038 *      to be treated specially during serialisation. When it is encountered,
039 *      the relations stored in the list will be added to the graph without validation.
040 *      The first object in the pair should be the predicate URI and the second object
041 *      will be serialised by the {@link RDFSerializer} to its own subgraph.
042 *      <p>
043 *      For example:
044 *      <code><pre>
045 *              class MyObject
046 *              {
047 *                      @RelationList
048 *                      public ArrayList<IndependentPair<URI,Object>> unvalidatedRelations;
049 *
050 *                      MyObject()
051 *                      {
052 *                              unvalidatedRelations = new ArrayList<>();
053 *                              unvalidatedRelations.add( new IndependentPair<>(
054 *                                      new URIImpl( "http://example.com/hasThingy" ), "Thingy" ) );
055 *                              unvalidatedRelations.add( new IndependentPair<>(
056 *                                      new URIImpl( "http://example.com/hasBlah" ), "Blah" ) );
057 *                      }
058 *              }
059 *      </pre></code>
060 *      ...would be serialised as follows:
061 *      <code><pre>
062 *              http://example.com/MyObject
063 *                      http://example.com/hasThingy "Thingy";
064 *                      http://example.com/hasBlah "Blah".
065 *      </pre></code>
066 *      <p>
067 *      Note that these are not unserialised again when deserialising the object
068 *      from RDF (there is no way for the deserialiser to know that these triples
069 *      came from this object).
070 *
071 *      @author David Dupplaw (dpd@ecs.soton.ac.uk)
072 *  @created 16 Nov 2012
073 *      @version $Author$, $Revision$, $Date$
074 */
075@Retention(RetentionPolicy.RUNTIME)
076public @interface RelationList
077{
078
079}
