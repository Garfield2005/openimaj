33  package org.openimaj.docs.tutorial.fund.audio;

35  import java.net.MalformedURLException;
36  import java.net.URL;
37  
38  import org.openimaj.audio.analysis.FourierTransform;
39  import org.openimaj.audio.filters.EQFilter;
40  import org.openimaj.audio.filters.EQFilter.EQType;
41  import org.openimaj.video.xuggle.XuggleAudio;
42  import org.openimaj.vis.general.BarVisualisation;
43  
44  /**
45   * Example that demonstrates chaining of audio processors as well as visualising
46   * the output of an FFT operation.
47   *
48   * @author David Dupplaw (dpd@ecs.soton.ac.uk)
49   * @created 18 Jun 2013
50   * @version $Author$, $Revision$, $Date$
51   */
52  public class FFTEQ
53  {
54  	/**
55  	 * Main method
56  	 *
57  	 * @param args Command-line args (unused)
58  	 * @throws MalformedURLException
59  	 */
60  	public static void main( final String[] args ) throws MalformedURLException
61  	{
62  		// Open a URL to the sine wave sweep. If you have downloaded
63  		// this file you should use a new File(<filename>) here.
64  		final XuggleAudio xa = new XuggleAudio(
65  			new URL( "http://www.audiocheck.net/download.php?" +
66  					"filename=Audio/audiocheck.net_sweep20-20klin.wav" ) );
67  
68  		// Create an EQ processor to pre-process the audio before doing the FFT
69  		// It will be a low-pass filter with a cut-off at 5KHz
70  		final EQFilter eq = new EQFilter( xa, EQType.LPF, 5000 );
71  
72  		// Create the Fourier transform processor chained to the audio decoder
73  		final FourierTransform fft = new FourierTransform( eq );
74  
75  		// Create a visualisation to show our FFT and open the window now
76  		// To avoid the visualisation scaling up the values once the peak
77  		// fades out, we need to disable auto scaling.
78  		final BarVisualisation bv = new BarVisualisation( 400, 200 );
79  		bv.setMaxValue( 1E12 );
80  		bv.setAutoScale( false );
81  		bv.showWindow( "FFTs" );
82  
83  		// Loop through the sample chunks from the audio capture thread
84  		// sending each one through the feature extractor and displaying
85  		// the results in the visualisation.
86  		while( fft.nextSampleChunk() != null )
87  		{
88  			final float[][] ffts = fft.getMagnitudes();
89  			bv.setData( ffts[0] );
90  		}
91  	}
92  }
