33  import java.io.IOException;
34  import java.util.ArrayList;
35  import java.util.HashMap;
36  import java.util.Map.Entry;

38  import org.openimaj.io.FileUtils;
39  import org.openimaj.text.nlp.language.LanguageDetector;
40  import org.openimaj.text.nlp.language.LanguageDetector.WeightedLocale;
41  import org.openimaj.twitter.collection.MemoryTwitterStatusList;
42  import org.openimaj.twitter.collection.TwitterStatusList;
43  import org.openimaj.util.pair.IndependentPair;

45  /**
46   * Perform an experiment, a set of tweets with a known language compared to the {@link LanguageDetector}'s response
47   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
48   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk), 
49   *
50   */
51  public class TrueLanguageExperiment {
52  	String[] inputFiles = new String[]{
53  		"dev-out-r",
54  		"trn-out-r",
55  		"tst-out-r"
56  	};
57  	private ArrayList<TrueLanguageTwitterStatus> inputStatusLists;
58  	private HashMap<String, IndependentPair<Float, Long>> scores;
59  	
60  	/**
61  	 * Load the tweets needed for this experiment
62  	 * @throws IOException
63  	 */
64  	public void prepareExperiment() throws IOException{
65  		this.inputStatusLists = new ArrayList<TrueLanguageTwitterStatus>();
66  		
67  		for (String input : this.inputFiles) {
68  			File inputFile = FileUtils.copyStreamToFile(TrueLanguageExperiment.class.getResourceAsStream(input), File.createTempFile(input, ".txt"));
69  			TwitterStatusList<TrueLanguageTwitterStatus> list = MemoryTwitterStatusList.read(inputFile,TrueLanguageTwitterStatus.class);
70  			this.inputStatusLists.addAll(list);
71  			inputFile.delete();
72  		}
73  	}
74  	
75  	/**
76  	 * do the experiment
77  	 * @throws IOException
78  	 */
79  	public void doExperiment() throws IOException{
80  		this.scores = new HashMap<String,IndependentPair<Float,Long>>();
81  		IndependentPair<Float, Long> totalScore;
82  		this.scores.put("total",  totalScore = IndependentPair.pair(0f, 0l));
83  		LanguageDetector languageDetector = new LanguageDetector();
84  		for (TrueLanguageTwitterStatus status : inputStatusLists) {
85  			WeightedLocale detected = languageDetector.classify(status.text);
86  			IndependentPair<Float, Long> currentScore = this.scores.get(detected.language);
87  			if(currentScore == null){
88  				currentScore = IndependentPair.pair(0.0f, 0l);
89  				this.scores.put(detected.language, currentScore);
90  			}
91  			if(detected.language.equals(status.lang_true)){
92  				currentScore.setFirstObject(currentScore.firstObject() + 1);
93  				totalScore.setFirstObject(totalScore.firstObject() + 1);
94  			}
95  			currentScore.setSecondObject(currentScore.secondObject() + 1);
96  			totalScore.setSecondObject(totalScore.secondObject() + 1);
97  		}
98  		for (Entry<String, IndependentPair<Float, Long>> entry : this.scores.entrySet()) {
99  			System.out.println(entry.getKey() + ":" + entry.getValue().firstObject() / entry.getValue().secondObject());
100 		}
101 		System.out.println("Total score:" + totalScore.firstObject() / totalScore.secondObject());
102 	}
103 	
104 	/**
105 	 * @param args
106 	 * @throws IOException
107 	 */
108 	public static void main(String[] args) throws IOException {
109 		TrueLanguageExperiment exp = new TrueLanguageExperiment();
110 		exp.prepareExperiment();
111 		exp.doExperiment();
112 	}
