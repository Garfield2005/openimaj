72  	/**
73  	 * see if reading the tweet dates works
74  	 * 
75  	 * @throws ParseException
76  	 * @throws IOException
77  	 */
78  	@Test
79  	public void testDates() throws ParseException, IOException {
80  		final File twitterfile = fileFromeStream(USMFStatus.class
81  				.getResourceAsStream("/org/openimaj/twitter/json_tweets.txt"));
82  		final FileTwitterStatusList<USMFStatus> status = FileTwitterStatusList.readUSMF(twitterfile, "UTF-8",
83  				GeneralJSONTwitter.class);
84  		for (final USMFStatus twitterStatus : status) {
85  			if (twitterStatus.isInvalid())
86  				continue;
87  			final DateTime d = twitterStatus.createdAt();
88  			assertEquals(d.getYear(), 2010);
89  			assertEquals(d.getMonthOfYear(), 10);
90  		}
91  	}

256 	protected File fileFromeStream(InputStream stream) throws IOException {
257 		final File f = folder.newFile("broken_raw" + stream.hashCode() + ".txt");
258 		final PrintWriter writer = new PrintWriter(new BufferedOutputStream(new FileOutputStream(f)));
259 		final BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
260 		String line = null;
261 		while ((line = reader.readLine()) != null) {
262 			writer.println(line);
263 		}
264 		writer.flush();
265 		writer.close();
266 		return f;
267 	}
