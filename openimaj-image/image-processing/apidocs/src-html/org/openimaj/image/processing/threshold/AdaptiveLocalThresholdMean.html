030package org.openimaj.image.processing.threshold;
031
032import org.openimaj.image.FImage;
033import org.openimaj.image.processing.convolution.AverageBoxFilter;
034
035/**
036 * Adaptive local thresholding using the mean of the patch and an offset.
037 * 
038 * @see <a
039 *      href="http://homepages.inf.ed.ac.uk/rbf/HIPR2/adpthrsh.htm">http://homepages.inf.ed.ac.uk/rbf/HIPR2/adpthrsh.htm</a>
040 * 
041 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
042 * 
043 */
044public class AdaptiveLocalThresholdMean extends AbstractLocalThreshold {
045        float offset;
046
047        /**
048         * Construct the thresholding operator with the given patch size (assumed
049         * square)
050         * 
051         * @param size
052         *            size of the local image patch
053         */
054        public AdaptiveLocalThresholdMean(int size) {
055                super(size);
056        }
057
058        /**
059         * Construct the thresholding operator with the given patch size
060         * 
061         * @param size_x
062         *            width of patch
063         * @param size_y
064         *            height of patch
065         */
066        public AdaptiveLocalThresholdMean(int size_x, int size_y) {
067                super(size_x, size_y);
068        }
069
070        /**
071         * Construct the thresholding operator with the given patch size (assumed
072         * square) and offset
073         * 
074         * @param size
075         *            size of the local image patch
076         * @param offset
077         *            offset from the patch mean at which the threshold occurs
078         */
079        public AdaptiveLocalThresholdMean(int size, float offset) {
080                this(size, size, offset);
081        }
082
083        /**
084         * Construct the thresholding operator with the given patch size and offset
085         * 
086         * @param size_x
087         *            width of patch
088         * @param size_y
089         *            height of patch
090         * @param offset
091         *            offset from the patch mean at which the threshold occurs
092         */
093        public AdaptiveLocalThresholdMean(int size_x, int size_y, float offset) {
094                super(size_x, size_y);
095                this.offset = offset;
096        }
097
098        @Override
099        public void processImage(FImage image) {
100                final FImage tmp = image.process(new AverageBoxFilter(sizeX, sizeY));
101
102                final float[][] tpix = tmp.pixels;
103                final float[][] ipix = image.pixels;
104                for (int y = 0; y < image.height; y++)
105                        for (int x = 0; x < image.width; x++)
106                                tpix[y][x] = ipix[y][x] < (tpix[y][x] - offset) ? 0f : 1f;
107
108                image.internalAssign(tmp);
109        }
110}
