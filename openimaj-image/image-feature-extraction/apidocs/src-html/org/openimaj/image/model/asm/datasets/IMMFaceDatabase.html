032import java.io.File;
033import java.io.IOException;
034import java.net.URL;
035
036import org.apache.commons.io.FileUtils;
037import org.openimaj.citation.annotation.Reference;
038import org.openimaj.citation.annotation.ReferenceType;
039import org.openimaj.data.DataUtils;
040import org.openimaj.experiment.annotations.DatasetDescription;
041import org.openimaj.image.Image;
042import org.openimaj.io.InputStreamObjectReader;
043
044/**
045 * The IMM Face Database (a set of labelled faces with connected points).
046 * 
047 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
048 */
049@DatasetDescription(
050                name = "The IMM Face Database",
051                description = "A dataset consisting of 240 annotated monocular " +
052                                "images of 40 different human faces. Points of correspondence are placed " +
053                                "on each image so the dataset can be readily used for building statistical " +
054                                "models of shape.",
055                creator = "Michael M. Nordstrom, Mads Larsen, Janusz Sierakowski, and Mikkel B. Stegmann",
056                url = "http://www2.imm.dtu.dk/~aam/datasets/datasets.html",
057                downloadUrls = {
058                                "http://datasets.openimaj.org/imm_face_db.zip"
059                })
060@Reference(
061                type = ReferenceType.Article,
062                author = { "M. B. Stegmann", "B. K. Ersb{\\o}ll", "R. Larsen" },
063                title = "{FAME} -- A Flexible Appearance Modelling Environment",
064                year = "2003",
065                journal = "IEEE Trans. on Medical Imaging",
066                pages = { "1319", "1331" },
067                number = "10",
068                publisher = "IEEE",
069                volume = "22")
070public class IMMFaceDatabase {
071        private static final String DATA_ZIP = "imm_face_db.zip";
072        private static final String DATA_DOWNLOAD_URL = "http://datasets.openimaj.org/imm_face_db.zip";
073
074        /**
075         * Get a dataset of the IMM images and points. If the dataset hasn't been
076         * downloaded, it will be fetched automatically and stored in the OpenIMAJ
077         * data directory. The images in the dataset are grouped by their class.
078         * 
079         * @see DataUtils#getDataDirectory()
080         * 
081         * @param reader
082         *            the reader for images (can be <code>null</code> if you only
083         *            care about the points)
084         * @return the dataset
085         * @throws IOException
086         *             if an error occurs
087         */
088        public static <IMAGE extends Image<?, IMAGE>> ShapeModelDataset<IMAGE> load(InputStreamObjectReader<IMAGE> reader)
089                        throws IOException
090        {
091                return ShapeModelDatasets.loadASFDataset(downloadAndGetPath(), reader);
092        }
093
094        private static String downloadAndGetPath() throws IOException {
095                final File dataset = DataUtils.getDataLocation(DATA_ZIP);
096
097                if (!(dataset.exists())) {
098                        dataset.getParentFile().mkdirs();
099                        FileUtils.copyURLToFile(new URL(DATA_DOWNLOAD_URL), dataset);
100                }
101
102                return "zip:file:" + dataset.toString();
103        }
104
105}
