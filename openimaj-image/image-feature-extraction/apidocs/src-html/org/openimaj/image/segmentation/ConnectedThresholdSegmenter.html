033package org.openimaj.image.segmentation;
034
035import java.util.List;
036
037import org.openimaj.image.FImage;
038import org.openimaj.image.connectedcomponent.ConnectedComponentLabeler;
039import org.openimaj.image.connectedcomponent.ConnectedComponentLabeler.Algorithm;
040import org.openimaj.image.pixel.ConnectedComponent;
041import org.openimaj.image.pixel.ConnectedComponent.ConnectMode;
042import org.openimaj.image.processor.Processor;
043
044/**
045 * Simple wrapper to make thresholding algorithms into {@link Segmenter}s by
046 * applying the thresholding operation and then applying connected component
047 * labeling. This class will produce components for both the foreground and
048 * background elements of thresholded input image.
049 * 
050 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
051 * 
052 */
053public class ConnectedThresholdSegmenter extends ThresholdSegmenter {
054        private ConnectMode mode;
055        private Algorithm algorithm;
056
057        /**
058         * Construct with the given thresholding algorithm implementation and
059         * connection mode.
060         * 
061         * @param thresholder
062         *            the thresholding algorithm
063         * @param mode
064         *            the connection mode
065         */
066        public ConnectedThresholdSegmenter(Processor<FImage> thresholder, ConnectMode mode) {
067                this(thresholder, ConnectedComponentLabeler.Algorithm.TWO_PASS, mode);
068        }
069
070        /**
071         * Construct with the given thresholding algorithm implementation.
072         * 
073         * @param thresholder
074         *            the thresholding algorithm
075         * @param algorithm
076         *            the connected component labeling algorithm to use
077         * @param mode
078         *            the connection mode
079         */
080        public ConnectedThresholdSegmenter(Processor<FImage> thresholder, Algorithm algorithm, ConnectMode mode) {
081                super(thresholder);
082                this.mode = mode;
083                this.algorithm = algorithm;
084        }
085
086        @Override
087        public List<ConnectedComponent> segment(FImage image) {
088                final FImage timg = image.process(thresholder);
089
090                return algorithm.findComponents(timg, 0, mode);
091        }
092}
