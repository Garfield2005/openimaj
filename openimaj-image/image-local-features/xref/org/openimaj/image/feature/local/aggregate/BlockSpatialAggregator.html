30  package org.openimaj.image.feature.local.aggregate;
31  
32  import java.util.ArrayList;
33  import java.util.List;
34  
35  import org.openimaj.feature.ArrayFeatureVector;
36  import org.openimaj.feature.FeatureVector;
37  import org.openimaj.feature.local.LocalFeature;
38  import org.openimaj.feature.local.SpatialLocation;
39  import org.openimaj.math.geometry.shape.Rectangle;
40  import org.openimaj.util.concatenate.Concatenatable;
41  
42  /**
43   * A {@link BlockSpatialAggregator} performs spatial pooling of local features
44   * by grouping the local features into non-overlapping, fixed-size spatial
45   * blocks, and applying a {@link VectorAggregator} (i.e. a
46   * {@link BagOfVisualWords}) to the features within each block before combining
47   * the aggregated results into a single vector (by passing through the blocks in
48   * a left-right, top-bottom fashion).
49   * 
50   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
51   * 
52   * @param <T>
53   *            Primitive array type of the backing array of each local feature
54   * @param <AGGREGATE>
55   *            Type of the aggregate {@link FeatureVector} produced
56   */
57  public class BlockSpatialAggregator<T, AGGREGATE extends FeatureVector & Concatenatable<AGGREGATE, AGGREGATE>>
58  		implements
59  		SpatialVectorAggregator<ArrayFeatureVector<T>, SpatialLocation, Rectangle>
60  {
61  	protected VectorAggregator<ArrayFeatureVector<T>, AGGREGATE> innerAggregator;
62  	protected int blocksX;
63  	protected int blocksY;
64  
65  	/**
66  	 * Construct with the given aggregator and number of blocks in the X and Y
67  	 * dimensions.
68  	 * 
69  	 * @param innerAggregator
70  	 *            the aggregator
71  	 * @param blocksX
72  	 *            the number of blocks in X
73  	 * @param blocksY
74  	 *            the number of blocks in Y
75  	 */
76  	public BlockSpatialAggregator(VectorAggregator<ArrayFeatureVector<T>, AGGREGATE> innerAggregator, int blocksX,
77  			int blocksY)
78  	{
79  		this.innerAggregator = innerAggregator;
80  		this.blocksX = blocksX;
81  		this.blocksY = blocksY;
82  	}
83  
84  	@Override
85  	public AGGREGATE aggregate(
86  			List<? extends LocalFeature<? extends SpatialLocation, ? extends ArrayFeatureVector<T>>> features,
87  			Rectangle bounds)
88  	{
89  		final List<List<LocalFeature<? extends SpatialLocation, ? extends ArrayFeatureVector<T>>>> spatialFeatures =
90  				new ArrayList<List<LocalFeature<? extends SpatialLocation, ? extends ArrayFeatureVector<T>>>>(blocksX
91  						* blocksY);
92  
93  		for (int i = 0; i < blocksX * blocksY; i++)
94  			spatialFeatures
95  					.add(new ArrayList<LocalFeature<? extends SpatialLocation, ? extends ArrayFeatureVector<T>>>());
96  
97  		final float stepX = (bounds.width - bounds.x) / blocksX;
98  		final float stepY = (bounds.height - bounds.y) / blocksY;
99  
100 		for (final LocalFeature<? extends SpatialLocation, ? extends ArrayFeatureVector<T>> f : features) {
101 			final SpatialLocation l = f.getLocation();
102 
103 			final int xbin = (int) Math.floor((l.x - bounds.x) / stepX);
104 			final int ybin = (int) Math.floor((l.y - bounds.y) / stepY);
105 			final int idx = xbin + blocksX * ybin;
106 
107 			spatialFeatures.get(idx).add(f);
108 		}
109 
110 		final AGGREGATE first = innerAggregator.aggregate(spatialFeatures.get(0));
111 
112 		final List<AGGREGATE> others = new ArrayList<AGGREGATE>(spatialFeatures.size() - 1);
113 		for (int i = 1; i < spatialFeatures.size(); i++) {
114 			others.add(innerAggregator.aggregate(spatialFeatures.get(i)));
115 		}
116 
117 		return first.concatenate(others);
118 	}
