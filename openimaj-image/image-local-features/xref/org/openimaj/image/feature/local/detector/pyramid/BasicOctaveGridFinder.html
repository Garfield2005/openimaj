30  package org.openimaj.image.feature.local.detector.pyramid;

32  import org.openimaj.image.FImage;
33  import org.openimaj.image.Image;
34  import org.openimaj.image.analysis.pyramid.Octave;
35  import org.openimaj.image.analysis.pyramid.gaussian.GaussianOctave;
36  import org.openimaj.image.analysis.pyramid.gaussian.GaussianPyramidOptions;
37  import org.openimaj.image.processor.SinglebandImageProcessor;
38  
39  /**
40   * An {@link AbstractOctaveInterestPointFinder} that detects points on a regular
41   * grid.
42   * 
43   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
44   * 
45   * @param <OCTAVE>
46   *            Type of underlying {@link Octave}
47   * @param <IMAGE>
48   *            Type of underlying {@link Image}
49   */
50  public class BasicOctaveGridFinder<OCTAVE extends Octave<?, ?, IMAGE>, IMAGE extends Image<?, IMAGE> & SinglebandImageProcessor.Processable<Float, FImage, IMAGE>>
51  		extends AbstractOctaveInterestPointFinder<OCTAVE, IMAGE>
52  {
53  	int skipX = 50;
54  	int skipY = 50;
55  	int borderX = 5;
56  	int borderY = 5;
57  	int startScaleIndex = 1;
58  	int scaleSkip = 1;
59  	int stopScaleIndex = 2;
60  
61  	@SuppressWarnings("unchecked")
62  	@Override
63  	public void process(OCTAVE octave) {
64  		this.octave = octave;
65  		int scales = 0;
66  
67  		if (octave instanceof GaussianOctave) {
68  			scales = ((GaussianPyramidOptions<IMAGE>) octave.options).getScales();
69  		}
70  
71  		final IMAGE[] images = octave.images;
72  		final int height = images[0].getHeight();
73  		final int width = images[0].getWidth();
74  
75  		// search through the scale-space images, leaving a border
76  		for (currentScaleIndex = startScaleIndex; currentScaleIndex < stopScaleIndex; currentScaleIndex += scaleSkip) {
77  			for (int y = borderY; y < height - borderY; y += skipY) {
78  				for (int x = borderX; x < width - borderX; x += skipX) {
79  					float octaveScale = currentScaleIndex;
80  
81  					if (octave instanceof GaussianOctave) {
82  						octaveScale = ((GaussianPyramidOptions<IMAGE>) octave.options).getInitialSigma()
83  								* (float) Math.pow(2.0, currentScaleIndex / scales);
84  					}
85  
86  					listener.foundInterestPoint(this, x, y, octaveScale);
87  				}
88  			}
89  		}
90  	}
91  }
