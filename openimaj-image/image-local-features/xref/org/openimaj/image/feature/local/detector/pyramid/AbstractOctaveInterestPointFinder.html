37  /**
38   * Abstract base class for objects capable of detecting interest points
39   * within an octave.
40   * 
41   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
42   * 
43   * @param <OCTAVE> Type of underlying {@link Octave}
44   * @param <IMAGE> Type of underlying {@link Image}
45   */
46  public abstract class AbstractOctaveInterestPointFinder<
47  		OCTAVE extends Octave<?,?,IMAGE>, 
48  		IMAGE extends Image<?,IMAGE> & SinglebandImageProcessor.Processable<Float,FImage,IMAGE>> 
49  	implements 
50  		OctaveInterestPointFinder<OCTAVE, IMAGE> 
51  {
52  	/**
53  	 * The current octave.  
54  	 */
55  	protected OCTAVE octave;
56  	
57  	/**
58  	 * The index of the scale currently being processed within the octave.
59  	 * This should be changed as the finder progresses through the scales.
60  	 */
61  	protected int currentScaleIndex;
62  	
63  	/**
64  	 * The listener object that gets informed when interest points are detected.
65  	 */
66  	protected OctaveInterestPointListener<OCTAVE, IMAGE> listener;
67  	
68  	/* (non-Javadoc)
69  	 * @see dogsiftdevel.pyramid.OctaveInterestPointFinder#getOctave()
70  	 */
71  	@Override
72  	public OCTAVE getOctave() {
73  		return octave;
74  	}
75  
76  	/* (non-Javadoc)
77  	 * @see dogsiftdevel.pyramid.OctaveInterestPointFinder#getCurrentScaleIndex()
78  	 */
79  	@Override
80  	public int getCurrentScaleIndex() {
81  		return currentScaleIndex;
82  	}
83  
84  	/* (non-Javadoc)
85  	 * @see dogsiftdevel.pyramid.OctaveInterestPointFinder#setInterestPointListener(dogsiftdevel.pyramid.OctaveInterestPointListener)
86  	 */
87  	@Override
88  	public void setOctaveInterestPointListener(OctaveInterestPointListener<OCTAVE, IMAGE> listener) {
89  		this.listener = listener;
90  	}
91  
92  	/* (non-Javadoc)
93  	 * @see dogsiftdevel.pyramid.OctaveInterestPointFinder#getInterestPointListener()
94  	 */
95  	@Override
96  	public OctaveInterestPointListener<OCTAVE, IMAGE> getOctaveInterestPointListener() {
97  		return listener;
98  	}
