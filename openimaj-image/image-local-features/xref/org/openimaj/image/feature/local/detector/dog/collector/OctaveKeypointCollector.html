43  /**
44   * Concrete implementation of an {@link AbstractOctaveLocalFeatureCollector}
45   * that collects {@link Keypoint}s with the feature vector provided by the 
46   * given feature extractor. {@link Keypoint}s contain the x, y and scale 
47   * coordinates of the interest point along with its dominant orientation
48   * 
49   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
50   *
51   * @param <IMAGE> Type of underlying {@link Image}
52   */
53  public class OctaveKeypointCollector<
54  		IMAGE extends Image<?,IMAGE> & SinglebandImageProcessor.Processable<Float,FImage,IMAGE>>
55  	extends 
56  		AbstractOctaveLocalFeatureCollector<
57  			GaussianOctave<IMAGE>, 
58  			ScaleSpaceFeatureExtractor<OrientedFeatureVector, IMAGE>, 
59  			Keypoint, 
60  			IMAGE
61  		> 
62  {
63  	protected ScaleSpaceImageExtractorProperties<IMAGE> extractionProperties = new ScaleSpaceImageExtractorProperties<IMAGE>();
64  	
65  	/**
66  	 * Construct with the given feature extractor.
67  	 * @param featureExtractor the feature extractor.
68  	 */
69  	public OctaveKeypointCollector(ScaleSpaceFeatureExtractor<OrientedFeatureVector, IMAGE> featureExtractor) {
70  		super(featureExtractor);
71  	}
72  		
73  	@Override
74  	public void foundInterestPoint(OctaveInterestPointFinder<GaussianOctave<IMAGE>, IMAGE> finder, float x, float y, float octaveScale) {
75  		int currentScaleIndex = finder.getCurrentScaleIndex();
76  		extractionProperties.image = finder.getOctave().images[currentScaleIndex];
77  		extractionProperties.scale = octaveScale;
78  		extractionProperties.x = x;
79  		extractionProperties.y = y;
80  		
81  		float octSize = finder.getOctave().octaveSize;
82  		
83  		addFeature(octSize * x, octSize * y, octSize * octaveScale);
84  	}
85  	
86  	protected void addFeature(float imx, float imy, float imscale) {
87  		OrientedFeatureVector[] fvs = featureExtractor.extractFeature(extractionProperties);
88  		
89  		for (OrientedFeatureVector fv : fvs) {
90  			features.add(new Keypoint(imx, imy, fv.orientation, imscale, fv.values));
91  		}
92  	}
93  }
