42  /**
43   * <p>
44   * Class capable of extracting local descriptors from a circular region
45   * in an image defined by its scale and centre. The actual feature 
46   * extracted is determined by the {@link GradientFeatureProvider} that
47   * is provided by the {@link GradientFeatureProviderFactory} set during
48   * construction.
49   * </p>
50   * <p>
51   * The GradientFeatureExtractor first calculates the dominant orientation
52   * of the image patch described by the {@link ScaleSpaceImageExtractorProperties}
53   * and then iterates over the pixels in an oriented square, centered on the
54   * interest point, passing the gradient and magnitude values of the respective
55   * pixel to the {@link GradientFeatureProvider}.
56   * </p>
57   * <p>
58   * The size of the sampling square, relative to scale is set by a single parameter,
59   * magnification. For some types of feature provider, this number
60   * might need to be set based on the internal settings of the provider. For example,
61   * with a {@link SIFTFeatureProvider} this will probably be set to a constant multiplied
62   * by the number of spatial bins of the feature. For SIFT, this constant is typically 
63   * around 3, so with a standard 4-spatial binned SIFT provider, the magnification
64   * factor of the extractor should be about 12.
65   * </p>
66   * 
67   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
68   *
69   */
70  public class GradientFeatureExtractor implements ScaleSpaceFeatureExtractor<OrientedFeatureVector, FImage> {
71  	AbstractDominantOrientationExtractor dominantOrientationExtractor;
72  	
73  	GradientFeatureProviderFactory factory;
74  	
75  	private GradientScaleSpaceImageExtractorProperties<FImage> currentGradientProperties = new GradientScaleSpaceImageExtractorProperties<FImage>();
76  	
77  	/**
78  	 * The magnification factor determining the size of the sampling
79  	 * region relative to the scale of the interest point.
80  	 */
81  	protected float magnification = 12;
82  		
83  	/**
84  	 * Construct with the given orientation extractor and gradient feature provider.
85  	 * The default magnification factor of 12 is used.
86  	 * 
87  	 * @param dominantOrientationExtractor the orientation extractor
88  	 * @param factory the gradient feature provider
89  	 */
90  	public GradientFeatureExtractor(AbstractDominantOrientationExtractor dominantOrientationExtractor, GradientFeatureProviderFactory factory) {
91  		this.dominantOrientationExtractor = dominantOrientationExtractor;
92  		this.factory = factory;
93  	}
94  	
95  	/**
96  	 * Construct with the given orientation extractor, gradient feature provider
97  	 * and magnification factor determining the size of the sampling
98  	 * region relative to the scale of the interest point.
99  	 * 
100 	 * @param dominantOrientationExtractor the orientation extractor
101 	 * @param factory the gradient feature provider
102 	 * @param magnification the magnification factor.
103 	 */
104 	public GradientFeatureExtractor(AbstractDominantOrientationExtractor dominantOrientationExtractor, GradientFeatureProviderFactory factory, float magnification) {
105 		this(dominantOrientationExtractor, factory);
106 		this.magnification = magnification;
107 	}

109 	@Override
110 	public OrientedFeatureVector[] extractFeature(ScaleSpaceImageExtractorProperties<FImage> properties) {
111 		GradientScaleSpaceImageExtractorProperties<FImage> gprops = getCurrentGradientProps(properties);
112 
113 		float [] dominantOrientations = dominantOrientationExtractor.extractFeatureRaw(gprops);
114 
115 		OrientedFeatureVector[] ret = new OrientedFeatureVector[dominantOrientations.length];
116 
117 		for (int i=0; i<dominantOrientations.length; i++) {
118 			ret[i] = createFeature(dominantOrientations[i]);
119 		}
120 
121 		return ret;
122 	}
123 
124 	/**
125 	 * Get the GradientScaleSpaceImageExtractorProperties for the given properties.
126 	 * The returned properties are the same as the input properties, but with the
127 	 * gradient images added. 
128 	 * 
129 	 * For efficiency, this method always returns the same cached GradientScaleSpaceImageExtractorProperties,
130 	 * and internally updates this as necessary. The gradient images are only recalculated
131 	 * when the input image from the input properties is different to the cached one.
132 	 * 
133 	 * @param properties input properties
134 	 * @return cached GradientScaleSpaceImageExtractorProperties 
135 	 */
136 	public GradientScaleSpaceImageExtractorProperties<FImage> getCurrentGradientProps(ScaleSpaceImageExtractorProperties<FImage> properties) {
137 		if (properties.image != currentGradientProperties.image) {
138 			currentGradientProperties.image = properties.image;
139 
140 			//only if the size of the image has changed do we need to reset the gradient and orientation images. 
141 			if (currentGradientProperties.orientation == null || 
142 					currentGradientProperties.orientation.height != currentGradientProperties.image.height || 
143 					currentGradientProperties.orientation.width != currentGradientProperties.image.width) { 
144 				currentGradientProperties.orientation = new FImage(currentGradientProperties.image.width, currentGradientProperties.image.height);
145 				currentGradientProperties.magnitude = new FImage(currentGradientProperties.image.width, currentGradientProperties.image.height);				
146 			}
147 
148 			FImageGradients.gradientMagnitudesAndOrientations(currentGradientProperties.image, currentGradientProperties.magnitude, currentGradientProperties.orientation);
149 		}
150 		
151 		currentGradientProperties.x = properties.x;
152 		currentGradientProperties.y = properties.y;
153 		currentGradientProperties.scale = properties.scale;
154 		
155 		return currentGradientProperties;
156 	}

158 	/*
159 	 * Iterate over the pixels in a sampling patch around the given feature coordinates
160 	 * and pass the information to a feature provider that will extract the relevant
161 	 * feature vector.
162 	 */
163 	protected OrientedFeatureVector createFeature(final float orientation) {
164 		final float fx = currentGradientProperties.x;
165 		final float fy = currentGradientProperties.y;
166 		final float scale = currentGradientProperties.scale; 
167 		
168 		//create a new feature provider and initialise it with the dominant orientation
169 		GradientFeatureProvider sfe = factory.newProvider();
170 		sfe.setPatchOrientation(orientation);
171 		
172 		//the integer coordinates of the patch
173 		final int ix = Math.round(fx);
174 		final int iy = Math.round(fy);
175 
176 		final float sin = (float) Math.sin(orientation);
177 		final float cos = (float) Math.cos(orientation);
178 
179 		//get the amount of extra sampling outside the unit square requested by the feature
180 		final float oversampling = sfe.getOversamplingAmount();
181 		
182 		//this is the size of the unit bounding box of the patch in the image in pixels
183 		final float boundingBoxSize = magnification * scale;
184 		
185 		//the amount of extra sampling per side in pixels
186 		final float extraSampling = oversampling * boundingBoxSize;
187 		
188 		//the actual sampling area is bigger than the boundingBoxSize by an extraSampling on each side
189 		final float samplingBoxSize = extraSampling + boundingBoxSize + extraSampling;
190 		
191 		//In the image, the box (with sides parallel to the image frame) that contains the
192 		//sampling box is:
193 		final float orientedSamplingBoxSize = Math.abs(sin * samplingBoxSize) + Math.abs(cos * samplingBoxSize);
194 		
195 		//now half the size and round to an int so we can iterate
196 		final int orientedSamplingBoxHalfSize = Math.round(orientedSamplingBoxSize / 2.0f);
197 
198 		//get the images and their size
199 		final FImage mag = currentGradientProperties.magnitude;
200 		final FImage ori = currentGradientProperties.orientation;
201 		final int width = mag.width;
202 		final int height = mag.height;
203 		
204 		//now pass over all the pixels in the image that *might* contribute to the sampling area
205 		for (int y = -orientedSamplingBoxHalfSize; y <= orientedSamplingBoxHalfSize; y++) {
206 			for (int x = -orientedSamplingBoxHalfSize; x <= orientedSamplingBoxHalfSize; x++) {
207 				int px = x + ix;
208 				int py = y + iy;
209 				
210 				//check if the pixel is in the image bounds; if not ignore it
211 				if (px >= 0 && px < width && py >= 0 && py < height) {
212 					//calculate the actual position of the sample in the patch coordinate system
213 					float sx = 0.5f + ((-sin * y + cos * x) - (fx - ix)) / boundingBoxSize;
214 					float sy = 0.5f + ((cos * y + sin * x) - (fy - iy)) / boundingBoxSize;
215 					
216 					//if the pixel is in the bounds of the sampling area then add it
217 					if (sx > -oversampling && sx < 1 + oversampling && sy > -oversampling && sy < 1 + oversampling) {
218 						sfe.addSample(sx, sy, mag.pixels[py][px], ori.pixels[py][px]);
219 					}
220 				}
221 			}
222 		}
223 
224 		return sfe.getFeatureVector();
225 	}
