40  /**
41   * Matcher rejects matches with no local support
42   * 
43   * @author Jonathon Hare
44   * @param <T> 
45   *
46   */
47  public class VotingKeypointMatcher<T extends Keypoint> extends FastBasicKeypointMatcher<T> implements LocalFeatureMatcher<T> {
48  	int neighbours;
49  	List<Pair<T>> consistentMatches = new ArrayList<Pair<T>>();
50  	protected int minVote;
51  	protected float singularityDistance;
52  	
53  	/**
54  	 * @param threshold threshold for determining matching keypoints
55  	 */
56  	public VotingKeypointMatcher(int threshold) {
57  		this(threshold, 15, 1, 200.0f); //default to 15 as in VideoGoogle paper
58  	}
59  	
60  	/**
61  	 * @param threshold threshold for determining matching keypoints
62  	 * @param neighbours number of neighbours within which to check for local support
63  	 * @param minVote
64  	 * @param singularityDistance
65  	 */
66  	public VotingKeypointMatcher(int threshold, int neighbours, int minVote, float singularityDistance) {
67  		super(threshold);
68  		this.neighbours = neighbours;
69  		this.minVote = minVote;
70  		this.singularityDistance = singularityDistance;
71  	}
72  	
73  	/**
74  	 * @return a list of consistent matching keypoints according
75  	 * to the estimated model parameters.
76  	 */
77  	@Override
78  	public List<Pair<T>> getMatches() {
79  		return consistentMatches;
80  	}
81  
82  	/**
83  	 * @return a list of all matches irrespective of whether they fit the model
84  	 */
85  	public List<Pair<T>> getAllMatches() {
86  		return matches;
87  	}
88  
89  	@Override
90  	public boolean findMatches(List<T> keys1) {
91  		super.findMatches(keys1);
92  		
93  		consistentMatches = new ArrayList<Pair<T>>();
94  		
95  		//filter dups
96  		//matches = ConsistentKeypointMatcher.filterColinear(matches, 1);
97  		
98  		//filter spurious matches by voting
99  		for (Pair<T> match : matches) {
100 			int vote = vote(match);
101 			if (vote > minVote)
102 				consistentMatches.add(match);
103 		}
104 		
105 		if (consistentMatches.size() == 0) 
106 			return false;
107 		
108 		//reject mappings to very close points
109 		if (checkSingularity()) {
110 			consistentMatches.clear();
111 			return false;
112 		}
113 		
114 		return true;
115 	}
116 	
117 	protected float [] getCentroid() {
118 		float mx = consistentMatches.get(0).secondObject().x;
119 		float my = consistentMatches.get(0).secondObject().y;
120 		for (int i=1; i<consistentMatches.size(); i++) {
121 			mx += consistentMatches.get(i).secondObject().x;
122 			my += consistentMatches.get(i).secondObject().y;
123 		}
124 		return new float[] {mx/consistentMatches.size(), my/consistentMatches.size()};
125 	}
126 	
127 	protected boolean checkSingularity() {
128 		float [] centroid = getCentroid();
129 		Keypoint k = new Keypoint();
130 		k.y = centroid[1];
131 		k.x = centroid[0];
132 		
133 		for (Pair<T> p : consistentMatches) {
134 			if (euclideanSqr(p.secondObject(), k) > singularityDistance) return false;
135 		}
136 		return true;
137 	}
138 	
139 	protected int vote(Pair<T> match) {
140 		List<T> nn = findModelNeighbours(match.secondObject());
141 		int vote = 0;
142 		
143 		for (Pair<T> m : matches) {
144 			for (Keypoint k : nn) {
145 				if (m.secondObject() == k) {
146 					vote++;
147 					break;
148 				}
149 			}
150 		}
151 		return vote;
152 	}
153 	
154 	protected float euclideanSqr(Keypoint k1, Keypoint k2) {
155 		return ((k1.x - k2.x) * (k1.x - k2.x)) + 
156 				((k1.y - k2.y)*(k1.y - k2.y));		
157 	}
158 	
159 	protected List<T> findModelNeighbours(final T kp) {
160 		class KpDist<Q extends Keypoint> implements Comparable<KpDist<Q>> {
161 			float distance;
162 			T keypoint;
163 			
164 			KpDist(T keypoint) {
165 				this.keypoint = keypoint;
166 				
167 				distance = euclideanSqr(keypoint, kp);
168 			}
169 
170 			@Override
171 			public int compareTo(KpDist<Q> o) {
172 				if (distance > o.distance) return 1;
173 				if (distance < o.distance) return -1;
174 				return 0;
175 			}
176 		}
177 
178 		List<KpDist<T>> list = new ArrayList<KpDist<T>>(); 
179 		for (T k : modelKeypoints) {
180 			list.add(new KpDist<T>(k));
181 		}
182 		Collections.sort(list);
183 		
184 		List<T> keys = new ArrayList<T>();
185 		for (int i=0; i<Math.min(neighbours, list.size()); i++)
186 			keys.add(list.get(i).keypoint);
187 		
188 		return keys;
189 	}
