030package org.openimaj.image.feature.local.descriptor.gradient;
031
032import org.openimaj.feature.OrientedFeatureVector;
033import org.openimaj.image.feature.local.descriptor.LocalFeatureProvider;
034import org.openimaj.math.util.Interpolation;
035
036/**
037 * Interface for classes capable of building local descriptors from the gradient
038 * (magnitude and orientation) information in an image patch. The model for the
039 * construction of various concrete feature extractors that implement this
040 * interface is that some external code will provide individual pixel samples
041 * (consisting of both magnitude and orientation values) and their relative
042 * positions within a unit sampling patch (or slightly beyond if interpolation
043 * is used; see {@link Interpolation#bilerp }). <br/>
044 * <br/>
045 * The general contract for methods using this interface is as follows: <br/>
046 * <br/>
047 * <code>
048 *      f = new GradientBasedLocalFeatureExtractorImpl() <br/>
049 *      f.setPatchOrientation(orientation) <br/>
050 *      for (pixels in patch) <br/>
051 *      &nbsp;&nbsp;&nbsp;&nbsp;f.addSample(normalised-pixel-position, pixel-magnitude, pixel-orientation) <br/>
052 * </code>
053 *
054 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
055 *
056 */
057public interface GradientFeatureProvider extends LocalFeatureProvider<OrientedFeatureVector> {
058        /**
059         * Set the primary orientation of the sample patch in the image being
060         * processed. This might be used by concrete implementations to provide
061         * rotation invariance.
062         *
063         * This method should only be called once by calling code, and the call
064         * should be before any calls to
065         * {@link #addSample(float, float, float, float)} are made.
066         *
067         * @param patchOrientation
068         *            the actual orientation of the square in the image
069         */
070        public abstract void setPatchOrientation(float patchOrientation);
071
072        /**
073         * Add a sample to the feature. The x and y coordinates are given in terms
074         * of a unit index square (i.e. 0<=x<=1 and 0<=y<=1).
075         *
076         * @param x
077         *            x-coordinate within the unit indexing square
078         * @param y
079         *            y-coordinate within the unit indexing square
080         * @param gradmag
081         *            the gradient magnitude at the given coordinate
082         * @param gradori
083         *            the gradient direction at the given coordinate
084         */
085        public abstract void addSample(float x, float y, float gradmag, float gradori);
086
087        @Override
088        public OrientedFeatureVector getFeatureVector();
089}
