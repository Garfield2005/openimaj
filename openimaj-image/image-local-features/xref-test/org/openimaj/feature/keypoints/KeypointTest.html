95  	private void displayCatKeys() throws IOException{
96  		FImage cat = ImageUtilities.readF(this.getClass().getResourceAsStream("/org/openimaj/image/data/cat.jpg"));
97  		cat = ResizeProcessor.doubleSize(cat);
98  		LocalFeatureList<Keypoint> catKeys = engine.findFeatures(cat);
99  		double minScale = 2.5;
100 		double minCol = cat.width/2.0;
101 		double minRow = cat.height/2.0;
102 		double maxCol = cat.width/2.0 + 40;
103 		double maxRow = cat.height/2.0 + 40;
104 		Keypoint selectedK = null;
105 		for(Keypoint k : catKeys){
106 			selectedK = k;
107 			if(k.scale>minScale && k.x > minCol &&  k.y > minRow && k.y < maxRow && k.x < maxCol ){
108 				catKeys = new MemoryLocalFeatureList<Keypoint>();
109 //				k.ori = (float) (k.ori - Math.PI/2.0);
110 				catKeys.add(k);
111 				break;
112 			}
113 		}
114 		
115 		FImage catInv = cat.clone().inverse();
116 		LocalFeatureList<Keypoint> catInvKeys = engine.findFeatures(catInv);
117 		for(Keypoint k : catInvKeys){
118 			selectedK = k;
119 			if(k.scale>minScale && k.x > minCol &&  k.y > minRow && k.y < maxRow && k.x < maxCol ){
120 				catInvKeys = new MemoryLocalFeatureList<Keypoint>();
121 //				k.ori = (float) (k.ori - Math.PI/2.0);
122 				catInvKeys.add(k);
123 				break;
124 			}
125 		}
126 		
127 		KeypointVisualizer<Float[],MBFImage> kpv = new KeypointVisualizer<Float[],MBFImage>(new MBFImage(cat,cat,cat), catKeys);
128 		MBFImage left = kpv.drawPatches(new Float[]{1.0f,0.0f,0.0f}, new Float[]{0.0f,1.0f,0.0f});
129 		left = left.extractCenter((int)selectedK.x, (int)selectedK.y, (int)(selectedK.scale * 3 * 3 * 2 +10), (int)(selectedK.scale * 3 * 3  * 2 +10));
130 		
131 		KeypointVisualizer<Float[],MBFImage> kpvRight = new KeypointVisualizer<Float[],MBFImage>(new MBFImage(catInv,catInv,catInv), catInvKeys);
132 		MBFImage right = kpvRight.drawPatches(new Float[]{1.0f,0.0f,0.0f}, new Float[]{0.0f,1.0f,0.0f});
133 		right = right.extractCenter((int)selectedK.x, (int)selectedK.y, (int)(selectedK.scale * 3 * 3 * 2 +10), (int)(selectedK.scale * 3 * 3  * 2 +10));
134 		
135 		MBFImage combined = new MBFImage(left.getWidth() + right.getWidth(), Math.max(left.getHeight(), right.getHeight()), 3);
136 		MBFImageRenderer renderer = combined.createRenderer();
137 		renderer.drawImage(left, 0, 0);
138 		renderer.drawImage(right, left.getWidth(), 0);
139 		
140 //		DisplayUtilities.display(combined);

148 	/**
149 	 * Save and load a keypoint, check to see if it is consistent
150 	 * 
151 	 * @throws IOException
152 	 */
153 	@Test public void testIO() throws IOException {
154 		File ascii = null;
155 		File binary = null;
156 		
157 		
158 		try {
159 			ascii = folder.newFile("kpttest.ascii");
160 			binary = folder.newFile("kpttest.bin");
161 			
162 			IOUtils.writeASCII(ascii, k1);
163 			IOUtils.writeBinary(binary, k1);
164 			
165 			//test ascii read
166 			List<Keypoint> asciiKeys = MemoryLocalFeatureList.read(ascii, Keypoint.class);
167 			List<Keypoint> asciiKeys2 = MemoryLocalFeatureList.read(ascii, Keypoint.class);
168 			
169 			assertEquals(k1.size(), asciiKeys.size());
170 			assertEquals(k1.size(), asciiKeys2.size());
171 			for (int i=0; i<k1.size(); i++) {
172 				assertEquals(k1.get(i).x, asciiKeys.get(i).x, FLOAT_EPS);
173 				assertEquals(k1.get(i).x, asciiKeys2.get(i).x, FLOAT_EPS);
174 				
175 				assertEquals(k1.get(i).y, asciiKeys.get(i).y, FLOAT_EPS);
176 				assertEquals(k1.get(i).y, asciiKeys2.get(i).y, FLOAT_EPS);
177 				
178 				assertEquals(k1.get(i).scale, asciiKeys.get(i).scale, FLOAT_EPS);
179 				assertEquals(k1.get(i).scale, asciiKeys2.get(i).scale, FLOAT_EPS);
180 				
181 				assertEquals(k1.get(i).ori, asciiKeys.get(i).ori, FLOAT_EPS);
182 				assertEquals(k1.get(i).ori, asciiKeys2.get(i).ori, FLOAT_EPS);
183 				
184 				assertArrayEquals(k1.get(i).ivec, asciiKeys.get(i).ivec);
185 				assertArrayEquals(k1.get(i).ivec, asciiKeys2.get(i).ivec);
186 				
187 				for (int v : k1.get(i).ivec) {
188 					assert(v+128 >= 0);
189 				}
190 			}
191 			
192 			//test binary read
193 			List<Keypoint> binaryKeys = MemoryLocalFeatureList.read(binary, Keypoint.class);
194 			
195 			assertEquals(k1.size(), binaryKeys.size());
196 			for (int i=0; i<k1.size(); i++) {
197 				assertEquals(k1.get(i).x, binaryKeys.get(i).x, FLOAT_EPS);
198 				
199 				assertEquals(k1.get(i).y, binaryKeys.get(i).y, FLOAT_EPS);
200 				
201 				assertEquals(k1.get(i).scale, binaryKeys.get(i).scale, FLOAT_EPS);
202 				
203 				assertEquals(k1.get(i).ori, binaryKeys.get(i).ori, FLOAT_EPS);
204 				
205 				assertArrayEquals(k1.get(i).ivec, binaryKeys.get(i).ivec);
206 			}
207 		} finally {
208 			if (ascii != null) ascii.delete();
209 			if (binary != null) binary.delete();
210 		}
211 	}
