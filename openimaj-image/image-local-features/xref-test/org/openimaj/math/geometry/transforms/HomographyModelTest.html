82  		randomPoints = new ArrayList<Point2d>();
83  
84  		final int[] randomX = RandomData.getRandomIntArray(1, nPoints, squareX + 1, squareX + squareWidth - 1, 0)[0];
85  		final int[] randomY = RandomData.getRandomIntArray(1, nPoints, squareX + 1, squareX + squareWidth - 1, 0)[0];

87  			randomPoints.add(new Point2dImpl(randomX[i], randomY[i]));
88  		}

91  	/**
92  	 * Transform the square by a random transform and see if you can find the
93  	 * square again by using a RANSAC model and a homography model. Apply some
94  	 * error the points once transformed to make things a bit more interesting
95  	 */
96  	@Test
97  	public void testRandomSquareTransform()
98  	{
99  
100 		Matrix trans = Matrix.identity(3, 3);
101 		trans = TransformUtilities.centeredRotationMatrix(Math.PI / 3.0, 2 * (squareX + squareWidth / 2),
102 				2 * (squareY + squareHeight / 2)).times(trans);
103 		trans = TransformUtilities.translateMatrix(200, 200).times(trans);
104 		// trans = TransformUtilities.rotationMatrix(Math.PI/3.0);
105 
106 		final ArrayList<Point2d> transformedPoints = new ArrayList<Point2d>();
107 		final List<IndependentPair<Point2d, Point2d>> pairs = new ArrayList<IndependentPair<Point2d, Point2d>>();
108 		final Random r = new Random();
109 		final int error = 3;
110 		final float stoppingCondition = 0.8f;
111 		for (final Point2d randomPoint : randomPoints) {
112 			final Point2dImpl pointTrans = ((Point2dImpl) randomPoint).transform(trans);
113 
114 			pointTrans.setX(pointTrans.getX() + (r.nextFloat()) * error * (r.nextBoolean() ? -1 : 1));
115 			pointTrans.setY(pointTrans.getY() + (r.nextFloat()) * error * (r.nextBoolean() ? -1 : 1));
116 
117 			transformedPoints.add(pointTrans);
118 			pairs.add(new IndependentPair<Point2d, Point2d>(randomPoint, pointTrans));
119 		}
120 
121 		int i = 0;
122 		while (i++ < 10) {
123 			pallet = new MBFImage(new FImage[] { new FImage(500, 500), new FImage(500, 500), new FImage(500, 500) });
124 			final MBFImageRenderer renderer = pallet.createRenderer();
125 
126 			final HomographyModel model = new HomographyModel();
127 			final SingleImageTransferResidual2d<HomographyModel> errorModel = new SingleImageTransferResidual2d<HomographyModel>();
128 			final float errorThresh = ((float) Math.sqrt(2 * error * error) * 2) + 1;
129 
130 			model.estimate(pairs);
131 			renderer.drawPolygon(this.square.asPolygon().transform(model.getTransform()), 1, RGBColour.ORANGE);
132 			final RANSAC<Point2d, Point2d, HomographyModel> fitterNormal = new RANSAC<Point2d, Point2d, HomographyModel>(
133 					model, errorModel, errorThresh,
134 					1500, new RANSAC.PercentageInliersStoppingCondition(stoppingCondition), false);
135 			Matrix fitterNormalTransform = null;
136 			renderer.drawPolygon(this.square.asPolygon(), 1, RGBColour.RED);
137 			renderer.drawPolygon(this.square.asPolygon().transform(trans), 1, RGBColour.RED);
138 			renderer.drawPoints(randomPoints, RGBColour.GREEN, 3);
139 			renderer.drawPoints(transformedPoints, RGBColour.GREEN, 3);
140 			if (fitterNormal.fitData(pairs))
141 			{
142 				fitterNormalTransform = model.getTransform().copy();
143 				renderer.drawPolygon(this.square.asPolygon().transform(fitterNormalTransform), 1, RGBColour.YELLOW);
144 
145 				final List<? extends IndependentPair<Point2d, Point2d>> inlierPairs = fitterNormal.getInliers();
146 				System.out.println("Number of best inliers for recalculation: " + inlierPairs.size());
147 				model.estimate(inlierPairs);
148 				final Matrix fitterRefitTransform = model.getTransform().copy();
149 				renderer.drawPolygon(this.square.asPolygon().transform(fitterRefitTransform), 1, RGBColour.CYAN);
150 				assertTrue(inlierPairs.size() >= nPoints * stoppingCondition);
151 
152 			}
153 			else {
154 				fitterNormalTransform = model.getTransform().copy();
155 				final List<? extends IndependentPair<Point2d, Point2d>> mData = fitterNormal.getModelConstructionData();
156 				for (final IndependentPair<Point2d, Point2d> independentPair : mData) {
157 					// Point2d a = independentPair.firstObject();
158 					final Point2d b = independentPair.secondObject();
159 					renderer.drawPoint(b, RGBColour.PINK, 4);
160 				}
161 				renderer.drawPolygon(this.square.asPolygon().transform(fitterNormalTransform), 1, RGBColour.GREEN);
162 			}
163 		}
164 
165 	}
