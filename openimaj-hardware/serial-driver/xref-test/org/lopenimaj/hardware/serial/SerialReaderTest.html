27  	/**
28  	 * @throws UnsupportedEncodingException 
29  	 */
30  	@Test
31  	public void testSerialReader() throws UnsupportedEncodingException
32  	{
33  		// This is the data stream from our simulated serial port.
34  		byte[] bytes = new byte[] {43, 49, 10, 43, 50, 10, 43, 51, 10, 43, 52};
35  		InputStream is = new ByteArrayInputStream( bytes );
36  		
37  		// This is the parser we'll use
38  		SerialDataParser parser = new SerialDataParser()
39  		{
40  			private String leftOvers;
41  			
42  			@Override
43  			public String[] parse( String data )
44  			{
45  				Pattern p = Pattern.compile( "(.*)\n" );
46  				Matcher m = p.matcher( data );
47  				int count = 0;
48  				ArrayList<String> splits = new ArrayList<String>();
49  				while( m.find() )
50  				{
51  					splits.add( m.group(1) );
52  					count += m.group().length();
53  				}
54  				
55  				leftOvers = data.substring( count );
56  				String[] dataSplit = splits.toArray( new String[splits.size()] );
57  				
58  				return dataSplit;
59  			}
60  			
61  			@Override
62  			public String getLeftOverString()
63  			{
64  				return leftOvers;
65  			}
66  		};
67  		
68  		// Construct a serial reader that reads from the string above
69  		SerialReader sr = new SerialReader( is, parser  );
70  		sr.setMaxBufferSize( 7 );
71  
72  		// We'll print out the data the parser is parsing.
73  		final ArrayList<String> dataItems = new ArrayList<String>();
74  		sr.addSerialDataListener( new SerialDataListener()
75  		{
76  			private int count = 0;
77  			
78  			@Override
79  			public void dataReceived( String data )
80  			{
81  				System.out.println( count+++" : '"+data+"'" );
82  				dataItems.add( data );
83  			}
84  		} );
85  		
86  		// The event isn't actually used. The serial reader reads from
87  		// the InputStream that's passed into the constructor.
88  		
89  		// First n bytes
90  		sr.serialEvent( null );
91  		
92  		// Next n bytes
93  		sr.serialEvent( null );
94  
95  		// Next n bytes
96  		sr.serialEvent( null );
97  
98  		// Next n bytes
99  		sr.serialEvent( null );
100 
101 		// We should have a list containing {'+1','+2','+3','+4'}
102 		Assert.assertEquals( 4, dataItems.size() );
103 		Assert.assertEquals( "+1", dataItems.get( 0 ) );
104 		Assert.assertEquals( "+2", dataItems.get( 1 ) );
105 		Assert.assertEquals( "+3", dataItems.get( 2 ) );
106 		Assert.assertEquals( "+4", dataItems.get( 3 ) );
107 	}
