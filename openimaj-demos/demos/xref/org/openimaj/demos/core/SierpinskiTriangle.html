42  /**
43   * Draw a Sierpinski Triangle into an FImage using two
44   * different techniques. 
45   * 
46   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
47   */
48  @Demo(
49  	author = "Jonathon Hare", 
50  	description = "Demonstrates some of the core drawing tools within " +
51  			"OpenIMAJ by drawing a Sierpinski triangle using two different " +
52  			"techniques.", 
53  	keywords = { "sierpinski", "triangle", "render", "point", "drawing" }, 
54  	title = "Sierpinski Triangle",
55  	icon = "/org/openimaj/demos/icons/core/sierpinski-icon.png"
56  )
57  public class SierpinskiTriangle {
58  	/**
59  	 * Draw a Sierpinski Triangle by plotting random points
60  	 * @return image with triangle
61  	 */
62  	public static FImage randomPointTriangle() {
63  		FImage image = new FImage(500, 500);
64  		FImageRenderer renderer = image.createRenderer();
65  		
66  		Point2d [] vertices = {
67  			new Point2dImpl(0, 500),
68  			new Point2dImpl(250, 0),
69  			new Point2dImpl(500, 500),
70  		};
71  		
72  		Point2d p = new Point2dImpl(75, 450);
73  		
74  		Random random = new Random();
75  		
76  		for (int i=0; i<5000; i++) {
77  			int j = random.nextInt(3);
78  			
79  			p.setX((p.getX() + vertices[j].getX()) / 2);
80  			p.setY((p.getY() + vertices[j].getY()) / 2);
81  			
82  			renderer.drawPoint(p, 1.0f, 1);
83  		}
84  		
85  		return image;
86  	}
87  
88  	protected static void divideTriangle(Point2d a, Point2d b, Point2d c, int k, FImageRenderer renderer) {
89  		if (k>0) {
90  			Point2d ab = new Point2dImpl((a.getX() + b.getX()) / 2, (a.getY() + b.getY()) / 2);
91  			Point2d ac = new Point2dImpl((a.getX() + c.getX()) / 2, (a.getY() + c.getY()) / 2);
92  			Point2d bc = new Point2dImpl((b.getX() + c.getX()) / 2, (b.getY() + c.getY()) / 2);
93  			
94  			divideTriangle(a, ab, ac, k-1, renderer);
95  			divideTriangle(c, ac, bc, k-1, renderer);
96  			divideTriangle(b, bc, ab, k-1, renderer);
97  		} else {
98  			renderer.drawShapeFilled(new Triangle(a, b, c), 1.0f);
99  		}
100 	}
101 	
102 	/**
103 	 * Draw a Sierpinski Triangle by recursively drawing sub-triangles
104 	 * @return image with triangle
105 	 */
106 	public static FImage polygonTriangle() {
107 		FImage image = new FImage(500, 500);
108 		FImageRenderer renderer = image.createRenderer();
109 		
110 		Point2d [] v = new Point2d[] {
111 				new Point2dImpl(0, 500),
112 				new Point2dImpl(500, 500),
113 				new Point2dImpl(250, 0),
114 		};
115 		
116 		divideTriangle(v[0], v[1], v[2], 4, renderer);
117 		
118 		return image;
119 	}
120 	
121 	/**
122 	 * 	Default main
123 	 *  @param args Command-line arguments
124 	 */
125 	public static void main(String [] args) {
126 		DisplayUtilities.display(randomPointTriangle());
127 		DisplayUtilities.display(polygonTriangle());
128 	}
