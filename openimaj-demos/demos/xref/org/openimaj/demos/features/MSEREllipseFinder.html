35  import org.openimaj.demos.Demo;

42  import org.openimaj.image.colour.Transforms;
43  import org.openimaj.image.feature.local.detector.mser.MSERFeatureGenerator;
44  import org.openimaj.image.renderer.MBFImageRenderer;
45  import org.openimaj.math.geometry.shape.Ellipse;

47  /**
48   * Demo showing ellipse fitted to MSERs
49   *
50   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
51   */
52  @Demo(
53  		author = "Sina Samangooei",
54  		description = "Shows ellipse fitting for regions detected using an MSER feature extractor",
55  		keywords = {
56  				"mser", "ellipse", "feat" },
57  				title = "MSER Ellipse Finder")
58  public class MSEREllipseFinder {
59  	/**
60  	 * Construct demo
61  	 */
62  	public MSEREllipseFinder() {
63  		final MBFImage image = new MBFImage(400, 400, ColourSpace.RGB);
64  		final MBFImageRenderer renderer = image.createRenderer();
65  
66  		image.fill(RGBColour.WHITE);
67  		final List<Ellipse> ellipses = new ArrayList<Ellipse>();
68  		ellipses.add(new Ellipse(200, 100, 100, 80, Math.PI / 4));
69  		ellipses.add(new Ellipse(200, 300, 50, 30, -Math.PI / 4));
70  		ellipses.add(new Ellipse(100, 300, 30, 50, -Math.PI / 3));
71  
72  		for (final Ellipse ellipse : ellipses) {
73  			renderer.drawShapeFilled(ellipse, RGBColour.BLACK);
74  		}
75  
76  		@SuppressWarnings("unchecked")
77  		final MSERFeatureGenerator mser = new MSERFeatureGenerator(MomentFeature.class);
78  		final List<Component> features = mser.generateMSERs(Transforms
79  				.calculateIntensityNTSC(image));
80  		for (final Component c : features) {
81  			final MomentFeature feature = c.getFeature(MomentFeature.class);
82  			renderer.drawShape(feature.getEllipse(2), RGBColour.RED);
83  			renderer.drawShape(feature.getEllipse(2)
84  					.calculateOrientedBoundingBox(), RGBColour.GREEN);
85  		}
86  		DisplayUtilities.display(image);
87  	}
88  
89  	/**
90  	 * The main method
91  	 *
92  	 * @param args
93  	 *            ignored
94  	 */
95  	public static void main(String args[]) {
96  		new MSEREllipseFinder();
97  	}
98  }
