36  import Jama.Matrix;
37  import Jama.QRDecomposition;

39  public class IncrementalSVD{
40  	private static final double DEFAULT_DAMPENING = 1.f;
41  	private int updateK;
42  	private Matrix Rworkspace;
43  	public IncrementalSVD(int updateK) {
44  		this.updateK = updateK;
45  		this.Rworkspace = new Matrix(updateK*2, updateK*2);
46  	}
47  
48  	Matrix U;
49  	double[] S;
50  	Matrix Sdiag;
51  	private double defaultDamp = DEFAULT_DAMPENING ;
52  	/**
53  	 * @param m new data to update the SVD with
54  	 */
55  	public void update(Matrix m){
56  		update(m,this.defaultDamp);
57  	}
58  	/**
59  	 * @param m new data to update the SVD with
60  	 * @param dampening each current eigen vector is weighted by this amount
61  	 */
62  	public void update(Matrix m, double dampening){
63  		Matrix d = Matrix.identity(this.updateK, this.updateK);
64  		update(m,d.timesEquals(dampening));
65  	}
66  	/**
67  	 * @param m
68  	 * @param dampening multiplied by the current eigen values (see the paper)
69  	 */
70  	public void update(Matrix m, Matrix dampening){
71  		ThinSingularValueDecomposition inputSVD = new ThinSingularValueDecomposition(m, updateK);
72  		Matrix U2 = inputSVD.U;
73  		double[] S2 = inputSVD.S;
74  		if(S2.length < updateK){
75  			double[] rep = new double[updateK];
76  			for (int i = 0; i < S2.length; i++) {
77  				rep[i] = S2[i];
78  			}
79  			S2 = rep;
80  		}
81  		Matrix S2diag = MatrixUtils.diag(S2);
82  		if(U == null){
83  			U = U2;
84  			S = S2;
85  			Sdiag = S2diag;
86  			return;
87  		}
88  
89  		Matrix Z = U.transpose().times(U2);
90  		Matrix U1U2Delta = U2.minus(U.times(Z));
91  		QRDecomposition qr = new QRDecomposition(U1U2Delta);
92  		Matrix R = qr.getR();
93  		Matrix Uprime = qr.getQ();
94  		this.Rworkspace.setMatrix(0,this.updateK-1, 0,this.updateK-1, dampening.times(Sdiag)); // top left
95  		this.Rworkspace.setMatrix(0,this.updateK-1, this.updateK,this.updateK*2 -1, Z.times(S2diag)); // top right
96  		this.Rworkspace.setMatrix(this.updateK,this.updateK*2 -1, this.updateK,this.updateK*2 -1, R.times(S2diag)); // bottom right
97  		// this.Rworkspace bottom left = 0
98  
99  		ThinSingularValueDecomposition rWorkspaceSVD = new ThinSingularValueDecomposition(this.Rworkspace, updateK);
100 		S = rWorkspaceSVD.S;
101 		Sdiag = MatrixUtils.diag(S);
102 		Matrix UR = rWorkspaceSVD.U;
103 
104 		Matrix topUR = UR.getMatrix(0,this.updateK-1,0,this.updateK-1);
105 		Matrix bottomUR = UR.getMatrix(this.updateK,this.updateK*2-1,0,this.updateK-1);
106 		U = U.times(topUR);
107 		U.plus(Uprime.times(bottomUR));
108 	}

119 		System.out.println("This is A1: ");
120 		A1.print(8, 8);
121 		System.out.println("This is A2: ");
122 		A2.print(8, 8);
123 		System.out.println("This is A3: ");
124 		A3.print(8, 8);
125 
126 		incSVD.update(A1);
127 		System.out.println("From IncSVD (A1): ");
128 		incSVD.U.print(5, 5);
129 		incSVD.Sdiag.print(5, 5);
130 
131 		System.out.println("From Thinsvd: ");
132 		ThinSingularValueDecomposition A1SVD = new ThinSingularValueDecomposition(A1, incSVD.updateK);
133 		A1SVD.U.print(5, 5);
134 		MatrixUtils.diag(A1SVD.S).print(5, 5);
135 
136 		incSVD.update(A2);
137 		System.out.println("From IncSVD (A1,A2): ");

141 		ThinSingularValueDecomposition A1A2SVD = new ThinSingularValueDecomposition(MatrixUtils.hstack(A1,A2), incSVD.updateK);

143 		MatrixUtils.diag(A1A2SVD.S).print(5, 5);

150 		ThinSingularValueDecomposition ASVD = new ThinSingularValueDecomposition(A, incSVD.updateK);
