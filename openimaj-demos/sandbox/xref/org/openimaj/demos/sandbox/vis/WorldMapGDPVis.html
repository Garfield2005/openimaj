33  package org.openimaj.demos.sandbox.vis;
34

36  import java.net.MalformedURLException;
37  import java.net.URL;
38  import java.util.HashMap;
39  
40  import org.openimaj.image.colour.ColourMap;
41  import org.openimaj.image.colour.RGBColour;
42  import org.openimaj.math.geometry.point.Point2d;
43  import org.openimaj.video.VideoDisplay;
44  import org.openimaj.vis.VideoVisualisation;
45  import org.openimaj.vis.general.LabelledPointVisualisation;
46  import org.openimaj.vis.general.LabelledPointVisualisation.LabelledDot;

52  /**
53   *	A demonstration of the World visualisation. It downloads a CSV file from the web,
54   *	parses it and selects some data from the table. It then steps through all the years
55   *	in the particular dataset and colours the countries based on the values of the data.
56   *	<p>
57   *	You can supply an indicator name on the command line (when you run it, it will print out
58   *	all available indicators). If you get the name wrong you'll get an NPE probably.
59   *	<p>
60   *	It demonstrates the {@link WorldVis} interface as well as using the {@link VideoVisualisation}
61   *	API to convert a {@link VisualisationImpl} into a {@link Video} which is displayed using a
62   *	standard {@link VideoDisplay}.
63   *
64   *	@author David Dupplaw (dpd@ecs.soton.ac.uk)
65   *  @created 12 Jun 2013
66   */
67  public class WorldMapGDPVis
68  {
69  	private static final HashMap<String,HashMap<Integer,HashMap<String,Double>>> indicators =
70  			new HashMap<String, HashMap<Integer, HashMap<String,Double>>>();
71  
72  	/**
73  	 * 	Main Method
74  	 *	@param args Command-line args. You can provide an indicator name here to override the default
75  	 * 	@throws IOException If the CSV cannot be downloaded
76  	 * 	@throws MalformedURLException Should not be thrown
77  	 */
78  	public static void main( final String[] args ) throws MalformedURLException, IOException
79  	{
80  		// Create a parser for a CSV file
81  		// The file gives values by country for a large set of indicators over a set of years.
82  		// Years are in columns: the first column is the country, the second column the indicator type.
83  		// We are using the ExcelCSVParser to parse this data for us.
84  		final LabeledCSVParser l = new LabeledCSVParser(
85  			new ExcelCSVParser(
86  				new URL(
87  					"http://users.ecs.soton.ac.uk/dpd/projects/openimaj/GDPgrowth-countries.csv"
88  				).openStream()
89  			) );
90  
91  		// Get each line in turn and make a data structure
92  		while( l.getLine() != null )
93  		{
94  			// Get the country and indicator (data) type.
95  			final String c = l.getValueByLabel( "Country" );
96  			final String t = l.getValueByLabel( "IndicatorName" );
97  
98  			// The data is between years 1971 and 2011, so we'll get each of the years
99  			// for every country and indicator and store it away in a big hashmap.
100 			for( int year = 1971; year <= 2011; year++ )
101 			{
102 				final String s = l.getValueByLabel( ""+year );
103 				if( !s.isEmpty() )
104 				{
105 					final Double v = Double.parseDouble( s );
106 
107 					HashMap<Integer,HashMap<String, Double>> yi = WorldMapGDPVis.indicators.get( t );
108 					if( yi == null )
109 						WorldMapGDPVis.indicators.put( t, yi = new HashMap<Integer,HashMap<String,Double>>() );
110 
111 					HashMap<String, Double> ci = yi.get( year );
112 					if( ci == null )
113 						yi.put( year, ci = new HashMap<String,Double>() );
114 
115 					ci.put( c, v );
116 				}
117 			}
118 		}
119 
120 		// List the indicators so we can choose one.
121 		System.out.println( "Possible indicators: "+WorldMapGDPVis.indicators.keySet() );
122 
123 		// The labelled point vis is used to plot country names.
124 		final LabelledPointVisualisation dpv = new LabelledPointVisualisation();
125 
126 		// The world map vis to show the world
127 		final WorldMap<LabelledDot> wm = new WorldMap<LabelledDot>( 1280, 720, dpv );
128 
129 		// Set the colours (and we'll remove the long/lat axes)
130 		wm.setDefaultCountryLandColour( RGBColour.BLACK );
131 		wm.getAxesRenderer().setDrawXAxis( false );
132 		wm.getAxesRenderer().setDrawYAxis( false );
133 
134 		// This is the indicator we'll use
135 		String showIndicator = "Agriculture, hunting, forestry, fishing (ISIC A-B)";
136 		if( args.length > 0 )
137 			showIndicator = args[0];
138 
139 		// Create a video from the visualisation.
140 		final VideoVisualisation vv = new VideoVisualisation( wm );
141 		VideoDisplay.createVideoDisplay( vv );
142 
143 		// Loop through all the years and update the visualisation to show
144 		// the values for each year. We do this once a second.
145 		for( int year = 1971; year < 2011; year++ )
146 		{
147 			wm.clearData();
148 			WorldMapGDPVis.updateYear( wm, showIndicator, year );
149 			try
150 			{
151 				Thread.sleep( 1000 );
152 			}
153 			catch( final InterruptedException e )
154 			{
155 				e.printStackTrace();
156 			}
157 		}
158 	}
159 
160 	/**
161 	 * 	Gets the data for a specific year and updates the visualisation.
162 	 *
163 	 *	@param wm The visualisation to update.
164 	 *	@param showIndicator Which indicator to show
165 	 *	@param year Which year to show for
166 	 */
167 	private static void updateYear( final WorldMap<LabelledDot> wm,
168 			final String showIndicator, final int year )
169 	{
170 		// Get the dataset for the given indicator and year.
171 		final HashMap<String, Double> set = WorldMapGDPVis.indicators.get( showIndicator ).get( year );
172 
173 		// Calculate the max and mean. Only values above the mean will be shown
174 		double max = 0;
175 		double mean = 0;
176 		for( final double d : set.values() )
177 		{
178 			max = Math.max( max, d );
179 			mean += d;
180 		}
181 		mean /= set.values().size();
182 
183 		// Loop through all the countries
184 		for( final String c : set.keySet() )
185 		{
186 			// Get the value
187 			double v = set.get( c );
188 
189 			// Clip the data (just in case)
190 			if( v < 0 ) v = 0;
191 
192 			// Get the country from the world vis (some won't work just to naming differences)
193 			final String cc = wm.getCountryCodeByName( c );
194 
195 			// If it didn't work, we will ignore it.
196 			if( cc != null )
197 			{
198 				// Get the position of the country (for plotting it's name label)
199 				final Point2d cl = wm.getCountryLocation(cc);
200 
201 				// Highlight the country with a colour based on its value
202 				wm.addHighlightCountry( cc, ColourMap.Autumn.apply( (float)(v/max) ) );
203 
204 				// Only if it's above the mean will we plot its name (just to keep the vis tidier)
205 				if( v > mean )
206 					wm.addPoint( cl.getX(), cl.getY(), new LabelledDot( c+": "+v, 1, RGBColour.WHITE ) );
207 			}
208 			else	System.out.println( "country "+c+" unknown");
209 		}
210 
211 	}
