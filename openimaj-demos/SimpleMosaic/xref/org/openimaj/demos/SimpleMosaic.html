52  /**
53   * Demonstration of building an image mosiac
54   * 
55   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
56   * 
57   */
58  public class SimpleMosaic {
59  	/**
60  	 * Build a mosaic
61  	 * 
62  	 * @param args
63  	 *            ignored
64  	 * @throws IOException
65  	 */
66  	public static void main(String args[]) throws IOException {
67  		final ResizeProcessor rp = new ResizeProcessor(800, 600);
68  		final DoGSIFTEngine engine = new DoGSIFTEngine();
69  
70  		final MBFImage imageMiddle = ImageUtilities.readMBF(new File("data/trento-view-1.jpg"));
71  		imageMiddle.processInplace(rp);
72  		final FImage workingImageMiddle = Transforms.calculateIntensityNTSC(imageMiddle);
73  		final LocalFeatureList<Keypoint> middleKP = engine.findFeatures(workingImageMiddle);
74  
75  		final ConsistentLocalFeatureMatcher2d<Keypoint> matcher =
76  				new ConsistentLocalFeatureMatcher2d<Keypoint>(new FastBasicKeypointMatcher<Keypoint>(8));
77  		final HomographyModel model = new HomographyModel();
78  		final RANSAC<Point2d, Point2d, HomographyModel> modelFitting =
79  				new RANSAC<Point2d, Point2d, HomographyModel>(model,
80  						new SingleImageTransferResidual2d<HomographyModel>(), 8.0, 1600,
81  						new RANSAC.BestFitStoppingCondition(), true);
82  		matcher.setFittingModel(modelFitting);
83  		matcher.setModelFeatures(middleKP);
84  		final ProjectionProcessor<Float[], MBFImage> ptp = new ProjectionProcessor<Float[], MBFImage>();
85  		imageMiddle.accumulateWith(ptp);
86  
87  		final MBFImage imageRight = ImageUtilities.readMBF(new File("data/trento-view-0.jpg"));
88  		imageRight.processInplace(rp);
89  		final FImage workingImageRight = Transforms.calculateIntensityNTSC(imageRight);
90  		final LocalFeatureList<Keypoint> rightKP = engine.findFeatures(workingImageRight);
91  		matcher.findMatches(rightKP);
92  		ptp.setMatrix(model.getTransform());
93  		imageRight.accumulateWith(ptp);
94  
95  		final MBFImage imageLeft = ImageUtilities.readMBF(new File("data/trento-view-2.jpg"));
96  		imageLeft.processInplace(rp);
97  		final FImage workingImageLeft = Transforms.calculateIntensityNTSC(imageLeft);
98  		final LocalFeatureList<Keypoint> leftKP = engine.findFeatures(workingImageLeft);
99  		matcher.findMatches(leftKP);
100 		ptp.setMatrix(model.getTransform());
101 		imageLeft.accumulateWith(ptp);
102 
103 		final MBFImage projected = ptp.performBlendedProjection(
104 				(-imageMiddle.getWidth()),
105 				imageMiddle.getWidth() + imageMiddle.getWidth(),
106 				-imageMiddle.getHeight() / 2, 3 * imageMiddle.getHeight() / 2, (Float[]) null);
107 		DisplayUtilities.display(projected);
108 		ImageUtilities.write(projected, "png", new File("/Users/jsh2/Desktop/mosaic.png"));
109 	}
