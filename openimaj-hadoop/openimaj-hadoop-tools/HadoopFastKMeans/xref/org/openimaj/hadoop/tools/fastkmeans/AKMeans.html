83  	/**
84  	 * the map for approximate kmeans. Uses the {@link ByteKMeans} under the
85  	 * hood. For each feature assign the feature to a centroid and emit with
86  	 * centroid as key.
87  	 * 
88  	 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
89  	 * 
90  	 */
91  	public static class Map extends Mapper<Text, BytesWritable, IntWritable, BytesWritable> {
92  		private static Path centroidsPath = null;
93  		private static int k = -1;
94  		private static HardAssigner<byte[], float[], IntFloatPair> assigner = null;
95  		private static double randomFallbackChance;
96  		private static boolean exact;

99  		protected void setup(Mapper<Text, BytesWritable, IntWritable, BytesWritable>.Context context) throws IOException,
100 				InterruptedException
101 		{
102 			loadCluster(context);
103 		}

105 		protected static synchronized void loadCluster(
106 				Mapper<Text, BytesWritable, IntWritable, BytesWritable>.Context context) throws IOException
107 		{
108 			final Path newPath = new Path(context.getConfiguration().getStrings(CENTROIDS_PATH)[0]);
109 			final boolean current = centroidsPath != null && centroidsPath.toString().equals(newPath.toString());
110 			if (!current) {
111 				k = Integer.parseInt(context.getConfiguration().getStrings(CENTROIDS_K)[0]);
112 				exact = Boolean.parseBoolean(context.getConfiguration().getStrings(CENTROIDS_EXACT)[0]);
113 				System.out.println("This is exact mode: " + exact);
114 				randomFallbackChance = 0.01;
115 				if (context.getConfiguration().getStrings(CENTROIDS_FALLBACK_CHANCE) != null) {
116 					randomFallbackChance = Double.parseDouble(context.getConfiguration().getStrings(
117 							CENTROIDS_FALLBACK_CHANCE)[0]);
118 				}
119 
120 				centroidsPath = newPath;
121 				System.out.println("Loading centroids from: " + centroidsPath);
122 				final URI uri = centroidsPath.toUri();
123 				final FileSystem fs = HadoopFastKMeansOptions.getFileSystem(uri);
124 				final InputStream is = fs.open(centroidsPath);
125 				final ByteCentroidsResult centroids = IOUtils.read(is, ByteCentroidsResult.class);
126 
127 				if (exact)
128 					assigner = new ExactByteAssigner(centroids);
129 				else
130 					assigner = new KDTreeByteEuclideanAssigner(centroids);
131 			} else {
132 				// System.out.println("No need to reload tree");
133 			}
134 		}

136 		@Override
137 		public void map(Text key, BytesWritable value, Context context) throws IOException, InterruptedException {
138 			final byte[] values = value.getBytes();
139 			final byte[] points = new byte[value.getLength()];
140 			System.arraycopy(values, 0, points, 0, points.length);
141 
142 			final int cluster = assigner.assign(points);
143 
144 			context.write(new IntWritable(cluster), new BytesWritable(points));
145 
146 			if (new Random().nextDouble() < randomFallbackChance) {
147 				context.write(new IntWritable(k + 1), new BytesWritable(points));
148 			}
149 		}

152 	private static int accumulateFromFeature(int[] sum, byte[] assigned) throws IOException {
153 		if (assigned.length != sum.length)
154 			throw new IOException("Inconsistency in sum and feature length");
155 		for (int i = 0; i < sum.length; i++) {
156 			sum[i] += assigned[i];
157 		}
158 		return 1;
159 	}
160 
161 	private static int accumulateFromSum(int[] sum, byte[] assigned) throws IOException {
162 		final int flen = (assigned.length / 4) - 1;
163 		final DataInputStream dis = new DataInputStream(new ByteArrayInputStream(assigned));
164 		if (flen != sum.length)
165 			throw new IOException("Inconsistency in sum and feature length");
166 		final int totalAssigned = dis.readInt();
167 		for (int i = 0; i < sum.length; i++) {
168 			sum[i] += dis.readInt();
169 		}
170 		return totalAssigned;
171 	}

173 	/**
174 	 * for efficiency, combine centroids early, emitting sums and k for
175 	 * centroids combined
176 	 * 
177 	 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
178 	 * 
179 	 */
180 	public static class Combine extends Reducer<IntWritable, BytesWritable, IntWritable, BytesWritable> {
181 		private int k;
182 
183 		@Override
184 		public void setup(Context context) throws IOException, InterruptedException {
185 			k = Integer.parseInt(context.getConfiguration().getStrings(CENTROIDS_K)[0]);
186 		}
187 
188 		@Override
189 		public void reduce(IntWritable key, Iterable<BytesWritable> values, Context context) throws IOException,
190 				InterruptedException
191 		{
192 			final int[] sum = new int[128];
193 			int totalAssigned = 0;
194 			for (final BytesWritable val : values) {
195 				// Copy the important part of the array
196 				final byte[] assigned = new byte[val.getLength()];
197 				System.arraycopy(val.getBytes(), 0, assigned, 0, assigned.length);
198 				// Skip over all the random runoff emittions
199 				if (key.get() > k) {
200 					context.write(key, new BytesWritable(assigned));
201 					continue;
202 				}
203 				int added = 0;
204 				// Accumulate either with feature or with an existing sum of
205 				// features
206 				if (assigned.length == 128)
207 					added += accumulateFromFeature(sum, assigned);
208 				else
209 					added += accumulateFromSum(sum, assigned);
210 				totalAssigned += added;
211 			}
212 			if (key.get() > k)
213 				return;
214 			// Write accumulation and current count
215 			final ByteArrayOutputStream bos = new ByteArrayOutputStream();
216 			final DataOutputStream dos = new DataOutputStream(bos);
217 			dos.writeInt(totalAssigned);
218 			for (final int i : sum) {
219 				dos.writeInt(i);
220 			}
221 			context.write(key, new BytesWritable(bos.toByteArray()));
222 		}
223 	}

225 	/**
226 	 * The AKmeans reducer. average the combined features assigned to each
227 	 * centroid, emit new centroids. may (if not assigned) result in some
228 	 * centroids with no value.
229 	 * 
230 	 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
231 	 * 
232 	 */
233 	public static class Reduce extends Reducer<IntWritable, BytesWritable, IntWritable, BytesWritable> {
234 		private int k;
235 
236 		@Override
237 		public void setup(Context context) throws IOException, InterruptedException {
238 			k = Integer.parseInt(context.getConfiguration().getStrings(CENTROIDS_K)[0]);
239 		}
240 
241 		@Override
242 		public void reduce(IntWritable key, Iterable<BytesWritable> values, Context context) throws IOException,
243 				InterruptedException
244 		{
245 			final int[] sum = new int[128];
246 			final byte[] out = new byte[128];
247 
248 			int totalAssigned = 0;
249 			for (final BytesWritable val : values) {
250 				final byte[] assigned = new byte[val.getLength()];
251 
252 				System.arraycopy(val.getBytes(), 0, assigned, 0, assigned.length);
253 
254 				if (key.get() > k) {
255 					context.write(key, new BytesWritable(assigned));
256 					continue;
257 				}
258 
259 				int added = 0;
260 				if (assigned.length == 128)
261 					added += accumulateFromFeature(sum, assigned);
262 				else
263 					added += accumulateFromSum(sum, assigned);
264 
265 				totalAssigned += added;
266 			}
267 
268 			if (key.get() > k)
269 				return;
270 
271 			for (int i = 0; i < sum.length; i++) {
272 				out[i] = (byte) ((sum[i] / totalAssigned));
273 			}
274 
275 			context.write(key, new BytesWritable(out));
276 		}
277 	}

279 	static class SelectTopKDump extends KeyValueDump<IntWritable, BytesWritable> {
280 		int index = 0;
281 		int randomGens = 0;
282 		byte[][] centroids;
283 
284 		SelectTopKDump(int k) {
285 			centroids = new byte[k][];
286 		}
287 
288 		@Override
289 		public void dumpValue(IntWritable key, BytesWritable val) {
290 			if (index >= centroids.length)
291 				return;
292 			final byte[] bytes = new byte[val.getLength()];
293 			System.arraycopy(val.getBytes(), 0, bytes, 0, bytes.length);
294 			centroids[index] = bytes;
295 			index++;
296 			if (key.get() == centroids.length + 1) {
297 				randomGens++;
298 			}
299 		}
300 
301 	}

303 	/**
304 	 * Given the location of a binary dump of centroids on the HDFS, load the
305 	 * binary dump and construct a proper {@link ByteKMeans} instance
306 	 * 
307 	 * @param centroids
308 	 * @param selected
309 	 * @param options
310 	 * @return {@link ByteKMeans} for the centoirds on the HDFS
311 	 * @throws Exception
312 	 */
313 	public static ByteCentroidsResult
314 			completeCentroids(String centroids, String selected, HadoopFastKMeansOptions options)
315 					throws Exception
316 	{
317 		System.out.println("Attempting to complete");
318 		final Path centroidsPath = new Path(centroids);
319 		SequenceFileUtility<IntWritable, BytesWritable> utility = new IntBytesSequenceMemoryUtility(
320 				centroidsPath.toUri(), true);
321 		final SelectTopKDump dump = new SelectTopKDump(options.k);
322 		utility.exportData(NamingStrategy.KEY, new ExtractionState(), 0, dump);
323 
324 		byte[][] newcentroids;
325 		newcentroids = dump.centroids;
326 
327 		// We need to pick k - (dump.index - dump.randomGens) new items
328 		System.out.println("Expecting " + options.k + " got " + dump.index + " of which " + dump.randomGens
329 				+ " were random");
330 		if (dump.index < options.k) {
331 			final int randomNeeded = options.k - (dump.index - dump.randomGens);
332 
333 			final SequenceFileByteFeatureSelector sfbs = new SequenceFileByteFeatureSelector(selected, options.output
334 					+ "/randomswap", options);
335 			final String initialCentroids = sfbs.getRandomFeatures(randomNeeded);
336 			final Path newcentroidsPath = new Path(initialCentroids);
337 			utility = new IntBytesSequenceMemoryUtility(newcentroidsPath.toUri(), true);
338 			final SelectTopKDump neededdump = new SelectTopKDump(randomNeeded);
339 			utility.exportData(NamingStrategy.KEY, new ExtractionState(), 0, neededdump);
340 			newcentroids = neededdump.centroids;
341 		}
342 
343 		final ByteCentroidsResult newFastKMeansCluster = new ByteCentroidsResult();
344 		newFastKMeansCluster.centroids = newcentroids;
345 
346 		return newFastKMeansCluster;
347 	}
