45  /**
46   * Command-line options for the downloader tool
47   * 
48   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
49   */
50  public class HadoopDownloaderOptions {
51  	private String[] args;
52  
53  	@Option(name = "--input", aliases = "-i", required = true, usage = "Input file or URL.", metaVar = "STRING")
54  	private String input;
55  
56  	@Option(name = "--output", aliases = "-o", required = true, usage = "Output file or URL.", metaVar = "STRING")
57  	private String output;
58  
59  	@Option(
60  			name = "--remove",
61  			aliases = "-rm",
62  			required = false,
63  			usage = "Remove the existing output location if it exists.",
64  			metaVar = "BOOLEAN")
65  	private boolean replace = false;
66  
67  	@Option(
68  			name = "--num-reducers",
69  			aliases = "-nr",
70  			required = false,
71  			usage = "Number of reducers. Controls the number of sequencefile parts created.")
72  	private int nreducers = 0;
73  
74  	@Option(
75  			name = "--num-threads",
76  			aliases = "-nt",
77  			required = false,
78  			usage = "Number of mapper threads. If > 1, then a multithreaded mapper will be used.")
79  	private int nThreads = 1;
80  
81  	@Option(
82  			name = "--input-mode",
83  			aliases = "-m",
84  			required = false,
85  			usage = "How should the URLs be processed to be downloaded.",
86  			handler = ProxyOptionHandler.class)
87  	private InputMode inputMode = InputMode.PLAIN;
88  	private Parser inputModeOp;
89  
90  	@Option(
91  			name = "--sleep",
92  			aliases = "-s",
93  			required = false,
94  			usage = "Time in milliseconds to sleep after downloading a file.",
95  			metaVar = "LONG")
96  	private long sleep = 0;
97  
98  	@Option(name = "--follow-redirects", aliases = "-f", usage = "Follow URL redirections", required = false)
99  	private boolean followRedirects = false;
100 
101 	@Option(name = "--log-failures", aliases = "-l", usage = "Log failed records to a file", required = false)
102 	private boolean writeFailures = false;
103 
104 	/**
105 	 * Construct with the given arguments
106 	 * 
107 	 * @param args
108 	 *            the arguments
109 	 */
110 	public HadoopDownloaderOptions(String[] args) {
111 		this.args = args;
112 	}
113 
114 	/**
115 	 * Prepare the options
116 	 * 
117 	 * @param initial
118 	 *            true if initial setup is being performed; false if inside the
119 	 *            mapper
120 	 */
121 	public void prepare(boolean initial) {
122 		final CmdLineParser parser = new CmdLineParser(this);
123 		try {
124 			parser.parseArgument(args);
125 			this.validate(initial);
126 		} catch (final CmdLineException e) {
127 			System.err.println(e.getMessage());
128 			System.err.println("Usage: hadoop -jar HadoopImageDownloader [options...] [files...]");
129 			parser.printUsage(System.err);
130 
131 			System.exit(1);
132 		}
133 	}
134 
135 	private void validate(boolean initial) {
136 		if (replace && initial) {
137 			try {
138 				final URI outuri = SequenceFileUtility.convertToURI(output);
139 
140 				final FileSystem fs = SequenceFileUtility.getFileSystem(outuri, new Configuration());
141 
142 				fs.delete(new Path(outuri.toString()), true);
143 			} catch (final IOException e) {
144 
145 			}
146 		}
147 	}
148

154 	 */

156 		return SequenceFileUtility.getFilePaths(input, "part");
157 	}
158 
159 	/**
160 	 * @return the output file location
161 	 */
162 	public Path getOutputPath() {
163 		return new Path(SequenceFileUtility.convertToURI(output).toString());
164 	}
165 
166 	/**
167 	 * @return the number of reducers
168 	 */
169 	public int getNumberOfReducers() {
170 		return this.nreducers;
171 	}
172 
173 	/**
174 	 * @return the {@link Parser} corresponding to the selected mode.
175 	 */
176 	public Parser getInputParser() {
177 		return inputModeOp;
178 	}
179 
180 	/**
181 	 * @return the time in milliseconds to sleep after downloading a file
182 	 */

184 		return sleep;
185 	}
186 
187 	/**
188 	 * @return true if redirects should be followed; false otherwise
189 	 */
190 	public boolean followRedirects() {
191 		return followRedirects;
192 	}
193 
194 	/**
195 	 * @return true if failed records should be logged to a file
196 	 */
197 	public boolean writeFailures() {
198 		return writeFailures;
199 	}
200 
201 	/**
202 	 * Get the number of threads to use in the mapper. If >1 a
203 	 * {@link MultithreadedMapper} will be used.
204 	 * 
205 	 * @return number of threads to use
206 	 */
207 	public int getNumberOfThreads() {
208 		return nThreads;
209 	}
