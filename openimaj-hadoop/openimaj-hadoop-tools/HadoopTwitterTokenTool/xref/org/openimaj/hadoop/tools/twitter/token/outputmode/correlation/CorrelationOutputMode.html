53  public class CorrelationOutputMode extends TwitterTokenOutputMode {
54  
55  	@Option(name = "--max-p-value", aliases = "-maxp", required = false, usage = "The maximum P-Value")
56  	double maxp = -1;
57  
58  	@Override
59  	public void write(HadoopTwitterTokenToolOptions opts, TwitterTokenMode completedMode) throws Exception {
60  		// Get time period
61  		final Path[] paths = HadoopToolsUtil.getInputPaths(completedMode.finalOutput(opts),
62  				CountTweetsInTimeperiod.TIMECOUNT_DIR);
63  		final IndependentPair<Long, Long> startend = readBegginingEndTime(paths, opts);
64  		// Get yahoo finance data for this time period
65  		final YahooFinanceData finance = new YahooFinanceData("AAPL", new DateTime(startend.firstObject()), new DateTime(
66  				startend.secondObject()));
67  		final Map<String, double[]> timeperiodFinance = finance.results();
68  		final String financeOut = outputPath + "/financedata";
69  		final Path p = HadoopToolsUtil.getOutputPath(financeOut);
70  		final FileSystem fs = HadoopToolsUtil.getFileSystem(p);
71  		final FSDataOutputStream os = fs.create(p);
72  		IOUtils.writeASCII(os, finance);
73  		// Correlate words with this time period's finance data
74  		final MultiStagedJob stages = new MultiStagedJob(
75  				HadoopToolsUtil.getInputPaths(completedMode.finalOutput(opts), CountWordsAcrossTimeperiod.WORDCOUNT_DIR),
76  				HadoopToolsUtil.getOutputPath(outputPath),
77  				opts.getArgs()
78  				);
79  		stages.queueStage(new CorrelateWordTimeSeries(financeOut, startend));
80  		stages.queueStage(new CorrelateWordSort(maxp));
81  		stages.runAll();
82  	}
83  
84  	private IndependentPair<Long, Long> readBegginingEndTime(Path[] paths, HadoopTwitterTokenToolOptions opts)
85  			throws Exception
86  	{
87  		final MultiStagedJob stages = new MultiStagedJob(
88  				paths,
89  				HadoopToolsUtil.getOutputPath(outputPath),
90  				opts.getArgs()
91  				);
92  		stages.queueStage(new TimeIndex().stage());
93  		stages.runAll();
94  		final LinkedHashMap<Long, IndependentPair<Long, Long>> tindex = TimeIndex.readTimeCountLines(outputPath);
95  		final Iterator<Long> ks = tindex.keySet().iterator();
96  		final long first = ks.next();
97  		long last = first;
98  		for (; ks.hasNext();) {
99  			last = ks.next();
100 		}
101 		return IndependentPair.pair(first, last);
102 	}
