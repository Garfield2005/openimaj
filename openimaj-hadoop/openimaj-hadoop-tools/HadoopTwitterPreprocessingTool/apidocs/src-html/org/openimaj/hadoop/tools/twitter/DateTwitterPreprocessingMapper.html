030package org.openimaj.hadoop.tools.twitter;
031
032import java.io.IOException;
033import java.io.PrintWriter;
034import java.io.StringWriter;
035import java.util.List;
036
037import org.apache.hadoop.io.LongWritable;
038import org.apache.hadoop.io.Text;
039import org.apache.hadoop.mapreduce.Mapper;
040import org.joda.time.DateTime;
041import org.kohsuke.args4j.CmdLineException;
042import org.openimaj.tools.twitter.modes.preprocessing.TwitterPreprocessingMode;
043import org.openimaj.tools.twitter.options.AbstractTwitterPreprocessingToolOptions;
044import org.openimaj.twitter.GeneralJSONTwitter;
045import org.openimaj.twitter.USMFStatus;
046
047/**
048 * This mapper loads arguments for the
049 * {@link AbstractTwitterPreprocessingToolOptions} from the
050 * {@link HadoopTwitterPreprocessingTool#ARGS_KEY} variable (once per in memory
051 * mapper) and uses these to preprocess tweets.
052 * 
053 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
054 *
055 */
056public class DateTwitterPreprocessingMapper extends Mapper<LongWritable, Text, LongWritable, Text> {
057        private static HadoopTwitterPreprocessingToolOptions options = null;
058        private static List<TwitterPreprocessingMode<?>> modes = null;
059
060        protected static synchronized void loadOptions(Mapper<LongWritable, Text, LongWritable, Text>.Context context)
061                        throws IOException
062        {
063                if (options == null) {
064                        try {
065                                options = new HadoopTwitterPreprocessingToolOptions(context.getConfiguration().getStrings(
066                                                HadoopTwitterPreprocessingTool.ARGS_KEY));
067                                options.prepare();
068                                modes = options.preprocessingMode();
069                        } catch (final CmdLineException e) {
070                                throw new IOException(e);
071                        } catch (final Exception e) {
072                                throw new IOException(e);
073                        }
074                }
075        }
076
077        @Override
078        protected void setup(Mapper<LongWritable, Text, LongWritable, Text>.Context context) throws IOException,
079                        InterruptedException
080        {
081                loadOptions(context);
082        }
083
084        @Override
085        protected void map(LongWritable key, Text value, Mapper<LongWritable, Text, LongWritable, Text>.Context context)
086                        throws java.io.IOException, InterruptedException
087        {
088                final USMFStatus status = new USMFStatus(GeneralJSONTwitter.class);
089                status.fillFromString(value.toString());
090                if (status.isInvalid())
091                        return;
092                for (final TwitterPreprocessingMode<?> mode : modes) {
093                        mode.process(status);
094                }
095                final StringWriter outTweetString = new StringWriter();
096                final PrintWriter outTweetWriter = new PrintWriter(outTweetString);
097                try {
098                        options.ouputMode().output(options.convertToOutputFormat(status), outTweetWriter);
099                        final DateTime date = status.createdAt();
100                        if (date == null)
101                                return;
102                        final DateTime dayDate = new DateTime(date.getYear(), date.getMonthOfYear(), date.getDayOfMonth(), 0, 0);
103
104                        context.write(new LongWritable(dayDate.getMillis()), new Text(outTweetString.getBuffer().toString()));
105                } catch (final Exception e) {
106                        System.err.println("Failed to write tweet: " + status.text);
107                        System.err.println("With error: ");
108                        e.printStackTrace();
109                }
110        }
111}
