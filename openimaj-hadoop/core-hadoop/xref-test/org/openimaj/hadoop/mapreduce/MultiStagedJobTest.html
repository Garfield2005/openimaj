57   * Test the MultiStagedJob

59   * @author Sina Samangooei (ss@ecs.soton.ac.uk)

62  public class MultiStagedJobTest {
63  	static class CountWords extends Mapper<LongWritable, Text, NullWritable, Text> {
64  
65  		@Override
66  		protected void map(LongWritable key, Text value, Mapper<LongWritable, Text, NullWritable, Text>.Context context)
67  				throws java.io.IOException, InterruptedException
68  		{
69  			context.write(NullWritable.get(), new Text(value.toString().split(" ").length + ""));
70  		}
71  	}
72  
73  	static class AddOne extends Mapper<LongWritable, Text, NullWritable, Text> {
74  		@Override
75  		protected void map(LongWritable key, Text value, Mapper<LongWritable, Text, NullWritable, Text>.Context context)
76  				throws java.io.IOException, InterruptedException
77  		{
78  			final Integer intVal = Integer.parseInt(value.toString());
79  			context.write(NullWritable.get(), new Text((intVal + 1) + ""));
80  		}
81  	}

83  	/**
84  	 * Working dir
85  	 */
86  	@Rule
87  	public TemporaryFolder folder = new TemporaryFolder();

89  	private File initialFile;
90  	private File outputFile;
91  
92  	/**
93  	 * Prepare the input file and output location
94  	 * 
95  	 * @throws IOException
96  	 */
97  	@Before
98  	public void setup() throws IOException {
99  		initialFile = folder.newFile("input");
100 		FileUtils.copyStreamToFile(MultiStagedJobTest.class.getResourceAsStream("/org/openimaj/hadoop/textfile"),
101 				initialFile);
102 
103 		outputFile = folder.newFile("out.dir");
104 		outputFile.delete();
105 	}

107 	/**
108 	 * Create a simple two stage process which counts the words per line,
109 	 * outputs the word counts per line and adds one to each word count per
110 	 * line.
111 	 * 
112 	 * @throws Exception
113 	 */
114 	@Test
115 	public void testMultipleStages() throws Exception {
116 		final MultiStagedJob mjob = new MultiStagedJob(initialFile.getAbsolutePath(), outputFile.getAbsolutePath(),
117 				new String[] {});
118 		mjob.queueStage(new Stage<
119 				TextInputFormat,
120 				TextOutputFormat<NullWritable, Text>,
121 				LongWritable, Text,
122 				NullWritable, Text,
123 				NullWritable, Text
124 				>()
125 				{
126 
127 					@Override
128 					public String outname() {
129 						return "countwords";
130 					}
131 
132 					@Override
133 					public Class<? extends Mapper<LongWritable, Text, NullWritable, Text>> mapper() {
134 						return CountWords.class;
135 					}
136 
137 				});
138 
139 		mjob.queueStage(new Stage<
140 				TextInputFormat,
141 				TextOutputFormat<NullWritable, Text>,
142 				LongWritable, Text,
143 				NullWritable, Text,
144 				NullWritable, Text
145 				>()
146 				{
147 
148 					@Override
149 					public String outname() {
150 						return "addone";
151 					}
152 
153 					@Override
154 					public Class<? extends Mapper<LongWritable, Text, NullWritable, Text>> mapper() {
155 						return AddOne.class;
156 					}
157 
158 				});
159 		System.out.println(mjob.runAll());
160 
161 		final Path[] countwordsp = mjob.getStagePaths("countwords");
162 		final Path[] addonep = mjob.getStagePaths("addone");
163 		final FileSystem fs = getFileSystem(countwordsp[0].toUri());
164 		final InputStream countstream = fs.open(countwordsp[0]);
165 		final InputStream addonestream = fs.open(addonep[0]);
166 		final String[] clines = FileUtils.readlines(countstream);
167 		final String[] alines = FileUtils.readlines(addonestream);
168 
169 		for (int i = 0; i < alines.length; i++) {
170 			final int cint = Integer.parseInt(clines[i]);
171 			final int aint = Integer.parseInt(alines[i]);
172 			assertTrue(aint == cint + 1);
173 		}
174 	}

176 	private static FileSystem getFileSystem(URI uri) throws IOException {
177 		final Configuration config = new Configuration();
178 		FileSystem fs = FileSystem.get(uri, config);
179 		if (fs instanceof LocalFileSystem)
180 			fs = ((LocalFileSystem) fs).getRaw();
181 		return fs;
182 	}
