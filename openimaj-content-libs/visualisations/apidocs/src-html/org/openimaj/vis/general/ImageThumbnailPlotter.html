033package org.openimaj.vis.general;
034
035import org.openimaj.image.MBFImage;
036import org.openimaj.image.processing.resize.ResizeProcessor;
037import org.openimaj.math.geometry.point.Point2d;
038import org.openimaj.vis.general.XYPlotVisualisation.LocatedObject;
039
040/**
041 *      An item plotter that is able to plot images into a visualisation at a given
042 *      position and size. The thumbnail size is initially 100 pixels, but if you need
043 *      to alter this based on the size of the visualisation, you can use the
044 *      {@link #setThumbnailSize(int)} method to set the maximum dimension.
045 *
046 *      @author David Dupplaw (dpd@ecs.soton.ac.uk)
047 *  @created 3 Jun 2013
048 */
049public class ImageThumbnailPlotter implements ItemPlotter<MBFImage,Float[],MBFImage>
050{
051        /** The maximum size of the thumbnail */
052        private int thumbnailSize = 100;
053
054        /**
055         *      @return the thumbnailSize
056         */
057        public int getThumbnailSize()
058        {
059                return this.thumbnailSize;
060        }
061
062        /**
063         *      @param thumbnailSize the thumbnailSize to set
064         */
065        public void setThumbnailSize( final int thumbnailSize )
066        {
067                this.thumbnailSize = thumbnailSize;
068        }
069
070        @Override
071        public void plotObject(
072                        final MBFImage visImage,
073                        final LocatedObject<MBFImage> object,
074                        final AxesRenderer2D<Float[],MBFImage> renderer )
075        {
076                final MBFImage thumbnail = object.object.process( new ResizeProcessor( this.thumbnailSize ) );
077                final Point2d p = renderer.calculatePosition( object.x, object.y );
078                visImage.createRenderer().drawImage( thumbnail, (int)p.getX(), (int)p.getY() );
079        }
080
081        /**
082         *      {@inheritDoc}
083         *      @see org.openimaj.vis.general.ItemPlotter#renderRestarting()
084         */
085        @Override
086        public void renderRestarting()
087        {
088        }
089}
