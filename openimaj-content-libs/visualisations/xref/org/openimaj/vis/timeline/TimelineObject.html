33  package org.openimaj.vis.timeline;
34  
35  import java.awt.Dimension;
36  
37  import org.openimaj.vis.DataUnitsTransformer;
38  import org.openimaj.vis.VisualisationImageProvider;
39  
40  /**
41   *	An object that can be drawn into a timeline needs to be able to provide
42   *	the start and end time of its data, and also be able to accept a size
43   *	at which it should draw itself. The object should draw itself in the
44   *	given image size with the start and end times fitting in the image exactly.
45   *	This allows the {@link Timeline} visualisation to scale the object as necessary
46   *	and allow it to align with the time ruler.
47   *
48   *	@author David Dupplaw (dpd@ecs.soton.ac.uk)
49   *  @created 19 Jun 2013
50   *	@version $Author$, $Revision$, $Date$
51   */
52  public interface TimelineObject extends VisualisationImageProvider
53  {
54  	/**
55  	 * 	The timeline object will need to know how to convert between
56  	 * 	time and pixels, and the {@link DataUnitsTransformer} given here
57  	 * 	should be able to do that. Use it to paint the image.
58  	 *	@param dpt
59  	 */
60  	public void setDataPixelTransformer( DataUnitsTransformer<Float[],double[],int[]> dpt );
61  
62  	/**
63  	 * 	Returns the start time of the timeline object in the timeline.
64  	 *	@return The start time
65  	 */
66  	public long getStartTimeMilliseconds();
67  
68  	/**
69  	 * 	Set the start time of this object
70  	 * 	@param l The start time in milliseconds
71  	 */
72  	public void setStartTimeMilliseconds( long l );
73  
74  	/**
75  	 * 	Returns the end time of the timeline object in the timeline.
76  	 *	@return The end time.
77  	 */
78  	public long getEndTimeMilliseconds();
79  
80  	/**
81  	 * 	The timeline object needs to be able to suggest a size, if it
82  	 * 	needs a particular amount. It's not guaranteed it will get this size,
83  	 * 	but it may ask for it by implementing this method.
84  	 *	@return A {@link Dimension}
85  	 */
86  	public Dimension getPreferredSize();
87  
88  	/**
89  	 * 	Returns the size that this object wishes to be.
90  	 *	@return The size the object wishes to be.
91  	 */
92  	public Dimension getRequiredSize();
93  }
