33  package org.openimaj.audio.samples;
34  
35  import org.openimaj.audio.AudioFormat;
36  import org.openimaj.audio.SampleChunk;
37  import org.openimaj.audio.timecode.AudioTimecode;
38  
39  /**
40   * 	This class provides a consistent API for access samples of different
41   * 	sizes (e.g. 8-bit and 16-bit). Subclasses implement get and set methods
42   * 	for samples in a sample byte buffer. The getters and setters return and
43   * 	take floats in all cases, so it is up to the implementing class to decide
44   * 	how best to convert the incoming value to an appropriate value for the
45   * 	sample buffer. The values given and expected should be normalised between
46   * 	{@link Integer#MAX_VALUE} and {@link Integer#MIN_VALUE}. Despite using
47   * 	floats, this allows us to detect clipping. Return values are signed such
48   * 	that an empty signal has sample value 0. If the audio has multiple channels
49   * 	they will occur in the same order as the original file (likely to be
50   * 	multiplexed - so <c>get(0)</c> will be channel 1 and <c>get(1)</c>
51   * 	will be channel 2, etc.).
52   *
53   * 	@author David Dupplaw (dpd@ecs.soton.ac.uk)
54   *	@created 23rd November 2011
55   */
56  public interface SampleBuffer extends Iterable<Float>
57  {
58  	/**
59  	 * 	Get the sample at the given index. The sample will be a float and have
60  	 * 	a value between {@link Integer#MAX_VALUE} and {@link Integer#MIN_VALUE}.
61  	 * 	That sample will be signed.
62  	 *
63  	 * 	@param index The index of the sample to retrieve
64  	 * 	@return The sample value as an float
65  	 */
66  	public float get( int index );
67  
68  	/**
69  	 * 	Set the sample value at the given index. The sample value should be
70  	 * 	scaled between {@link Integer#MAX_VALUE} and {@link Integer#MIN_VALUE}
71  	 * 	(i.e. it should be signed).
72  	 *
73  	 * 	@param index The index of the sample to set.
74  	 * 	@param sample The sample value to set.
75  	 */
76  	public void set( int index, float sample );
77  
78  	/**
79  	 * 	Returns the size of this buffer. Divide by the number of channels
80  	 * 	to get the number of samples per channel.
81  	 * 	@return the size of this buffer.
82  	 */
83  	public int size();
84  
85  	/**
86  	 * 	Returns the audio format for this set of samples.
87  	 *	@return The {@link AudioFormat} for this set of samples.
88  	 */
89  	public AudioFormat getFormat();
90  
91  	/**
92  	 * 	Reset the audio format for the samples in this buffer.
93  	 * 	@param af The {@link AudioFormat}
94  	 */
95  	public void setFormat( AudioFormat af );
96  
97  	/**
98  	 * 	Return a sample chunk that contains the data from
99  	 * 	this sample buffer. Note that any timestamps will be
100 	 * 	unset in the new sample chunk.
101 	 *
102 	 * 	@return A {@link SampleChunk} containing data in this buffer.
103 	 */
104 	public SampleChunk getSampleChunk();
105 
106 	/**
107 	 * 	Return a sample chunk that contains the data from
108 	 * 	a specific channel in this sample buffer. The channel is
109 	 * 	numbered from 0 (so mono audio streams will only have 1 channel
110 	 * 	accessed with getSampleChunk(0)).
111 	 *
112 	 * 	Note that any timestamps will be unset in the new sample chunk.
113 	 *
114 	 *  @param channel The channel
115 	 *
116 	 * 	@return A {@link SampleChunk} containing data in this buffer.
117 	 */
118 	public SampleChunk getSampleChunk( int channel );
119 
120 	/**
121 	 * 	Returns the normalised (0..1) sample buffer data as a double array.
122 	 *	@return A double array containing the normalised samples
123 	 */
124 	public double[] asDoubleArray();
125 
126 	/**
127 	 * 	Returns the samples in their channels as normalise (0..1) sample buffer
128 	 * 	data.
129 	 *	@return The array where the first dimension is the channels.
130 	 */
131 	public double[][] asDoubleChannelArray();
132 
133 	/**
134 	 * 	Returns the sample value at the given index without scaling - that is,
135 	 * 	at it's original sample value. This can be useful for debugging, or
136 	 * 	if the sample buffer is being used for non-audio applications.
137 	 *
138 	 *  @param index The index to retrieve
139 	 *  @return The value unscaled
140 	 */
141 	public float getUnscaled( int index );
142 
143 	/**
144 	 * 	Returns the timecode of the start of this sample buffer. May return null if
145 	 * 	the timecode is unknown or has no meaning in this context.
146 	 *	@return The timecode a the start of this buffer.
147 	 */
148 	public AudioTimecode getStartTimecode();
149 }
