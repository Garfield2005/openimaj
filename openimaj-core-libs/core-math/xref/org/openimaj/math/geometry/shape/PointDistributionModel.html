49  /**
50   * A 2d point distribution model learnt from a set of {@link PointList}s with
51   * corresponding points (the ith point in each {@link PointList} is the same
52   * landmark).
53   * 
54   * The pdm models the mean shape and the variance from the mean of the top N
55   * principal components. The model is generative and can generate new shapes
56   * from a scaling vector. To ensure that newly generated shapes are plausible,
57   * scaling vectors have {@link Constraint}s applied to them.
58   * 
59   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
60   * 
61   */
62  @References(references = {
63  		@Reference(
64  				author = { "Cootes, T. F.", "Taylor, C. J." },
65  				title = "Statistical Models of Appearance for Computer Vision",
66  				type = ReferenceType.Unpublished,
67  				month = "October",
68  				year = "2001",
69  				url = "http://isbe.man.ac.uk/~bim/Models/app_model.ps.gz"
70  		),
71  		@Reference(
72  				type = ReferenceType.Inproceedings,
73  				author = { "C. J. Taylor", "D. H. Cooper", "J. Graham" },
74  				title = "Training models of shape from sets of examples",
75  				year = "1992",
76  				booktitle = "Proc. BMVC92, Springer-Verlag",
77  				pages = { "9", "", "18" }
78  		)
79  })
80  public class PointDistributionModel {
81  	/**
82  	 * Interface for modelling constraints applied to the scaling vector of
83  	 * {@link PointDistributionModel}s so that generated models are plausible.
84  	 * 
85  	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
86  	 */
87  	public interface Constraint {
88  		/**
89  		 * Apply constraints to a scaling vector so that it will generated a
90  		 * plausible model and return the new constrained vector.
91  		 * 
92  		 * @param scaling
93  		 *            the scaling vector to constrain
94  		 * @param lamda
95  		 *            the eigenvalues of the {@link PointDistributionModel}
96  		 * @return the constrained scaling vector
97  		 */
98  		public double[] apply(double[] scaling, double[] lamda);
99  	}

101 	/**
102 	 * A constraint that does nothing.
103 	 * 
104 	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
105 	 * 
106 	 */
107 	public static class NullConstraint implements Constraint {
108 		@Override
109 		public double[] apply(double[] in, double[] lamda) {
110 			return in;
111 		}
112 	}
113 
114 	/**
115 	 * A constraint that ensures that each individual element of the scaling
116 	 * vector is within +/- x standard deviations of the model.
117 	 * 
118 	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
119 	 * 
120 	 */
121 	public static class BoxConstraint implements Constraint {
122 		double multiplier;
123 
124 		/**
125 		 * Construct with the given multiplier of the standard deviation.
126 		 * 
127 		 * @param multiplier
128 		 */
129 		public BoxConstraint(double multiplier) {
130 			this.multiplier = multiplier;
131 		}
132 
133 		@Override
134 		public double[] apply(double[] in, double[] lamda) {
135 			final double[] out = new double[in.length];
136 
137 			for (int i = 0; i < in.length; i++) {
138 				final double w = multiplier * Math.sqrt(lamda[i]);
139 				out[i] = in[i] > w ? w : in[i] < -w ? -w : in[i];
140 			}
141 
142 			return out;
143 		}
144 	}
145 
146 	/**
147 	 * Constrain the scaling vector to a hyper-ellipsoid.
148 	 * 
149 	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
150 	 * 
151 	 */
152 	public static class EllipsoidConstraint implements Constraint {
153 		double dmax;
154 
155 		/**
156 		 * Construct with the given maximum normalised ellipsoid radius.
157 		 * 
158 		 * @param dmax
159 		 */
160 		public EllipsoidConstraint(double dmax) {
161 			this.dmax = dmax;
162 		}
163 
164 		@Override
165 		public double[] apply(double[] in, double[] lamda) {
166 			double dmsq = 0;
167 			for (int i = 0; i < in.length; i++) {
168 				dmsq += in[i] * in[i] / lamda[i];
169 			}
170 
171 			if (dmsq < dmax * dmax) {
172 				return in;
173 			}
174 
175 			final double sc = dmax / Math.sqrt(dmsq);
176 			final double[] out = new double[in.length];
177 			for (int i = 0; i < in.length; i++) {
178 				out[i] = in[i] * sc;
179 			}
180 
181 			return out;
182 		}
183 	}

191 	/**
192 	 * Construct a {@link PointDistributionModel} from the given data with a
193 	 * {@link NullConstraint}.
194 	 * 
195 	 * @param data
196 	 */
197 	public PointDistributionModel(List<PointList> data) {
198 		this(new NullConstraint(), data);
199 	}

201 	/**
202 	 * Construct a {@link PointDistributionModel} from the given data and
203 	 * {@link Constraint}.
204 	 * 
205 	 * @param constraint
206 	 * @param data
207 	 */
208 	public PointDistributionModel(Constraint constraint, List<PointList> data) {
209 		this.constraint = constraint;
210 
211 		// align
212 		mean = GeneralisedProcrustesAnalysis.alignPoints(data, 5, 10);
213 
214 		// build data matrix
215 		final Matrix m = buildDataMatrix(data);
216 
217 		// perform pca
218 		this.pc = new SvdPrincipalComponentAnalysis();
219 		pc.learnBasis(m);
220 
221 		numComponents = this.pc.getEigenValues().length;
222 	}
223 
224 	private Matrix buildDataMatrix(PointList data) {
225 		final List<PointList> pls = new ArrayList<PointList>(1);
226 		pls.add(data);
227 		return buildDataMatrix(pls);
228 	}
229 
230 	private Matrix buildDataMatrix(List<PointList> data) {
231 		final int nData = data.size();
232 		final int nPoints = data.get(0).size();
233 
234 		final Matrix m = new Matrix(nData, nPoints * 2);
235 		final double[][] mData = m.getArray();
236 
237 		for (int i = 0; i < nData; i++) {
238 			final PointList pts = data.get(i);
239 			for (int j = 0, k = 0; k < nPoints; j += 2, k++) {
240 				final Point2d pt = pts.points.get(k);
241 
242 				mData[i][j] = pt.getX();
243 				mData[i][j + 1] = pt.getY();
244 			}
245 		}
246 
247 		return m;
248 	}

280 	/**
281 	 * Generate a plausible new shape from the scaling vector. The scaling
282 	 * vector is constrained by the underlying {@link Constraint} before being
283 	 * used to generate the model.
284 	 * 
285 	 * @param scaling
286 	 *            scaling vector.
287 	 * @return a new shape
288 	 */
289 	public PointList generateNewShape(double[] scaling) {
290 		final PointList newShape = new PointList();
291 
292 		final double[] pts = pc.generate(constraint.apply(scaling, pc.getEigenValues()));
293 
294 		for (int i = 0; i < pts.length; i += 2) {
295 			final float x = (float) pts[i];
296 			final float y = (float) pts[i + 1];
297 
298 			newShape.points.add(new Point2dImpl(x, y));
299 		}
300 
301 		return newShape;
302 	}
303 
304 	/**
305 	 * Compute the standard deviations of the shape components, multiplied by
306 	 * the given value.
307 	 * 
308 	 * @param multiplier
309 	 *            the multiplier
310 	 * @return the multiplied standard deviations
311 	 */
312 	public double[] getStandardDeviations(double multiplier) {
313 		final double[] rngs = pc.getStandardDeviations();
314 
315 		for (int i = 0; i < rngs.length; i++) {
316 			rngs[i] = rngs[i] * multiplier;
317 		}
318 
319 		return rngs;
320 	}

322 	/**
323 	 * Determine the best parameters of the PDM for the given model.
324 	 * 
325 	 * @param observed
326 	 *            the observed model.
327 	 * @return the parameters that best fit the model.
328 	 */
329 	public IndependentPair<Matrix, double[]> fitModel(PointList observed) {
330 		double[] model = new double[numComponents];
331 		double delta = 1.0;
332 		Matrix pose = null;
333 
334 		final ProcrustesAnalysis pa = new ProcrustesAnalysis(observed);
335 		int count = 0;
336 		while (delta > 1e-6 && count++ < maxIter) {
337 			final PointList instance = this.generateNewShape(model);
338 
339 			pose = pa.align(instance);
340 
341 			final PointList projected = observed.transform(pose.inverse());
342 
343 			// TODO: tangent space projection???
344 
345 			final Matrix y = buildDataMatrix(projected);
346 			final Matrix xbar = new Matrix(new double[][] { pc.getMean() });
347 			double[] newModel = (y.minus(xbar)).times(pc.getBasis()).getArray()[0];
348 
349 			newModel = constraint.apply(newModel, pc.getEigenValues());
350 
351 			delta = 0;
352 			for (int i = 0; i < newModel.length; i++)
353 				delta += (newModel[i] - model[i]) * (newModel[i] - model[i]);
354 			delta = Math.sqrt(delta);
355 
356 			model = newModel;
357 		}
358 
359 		return new IndependentPair<Matrix, double[]>(pose, model);
360 	}
