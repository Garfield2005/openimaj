38  public class EdgeTable
39  {
40  	private List<EdgeNode> m_List = new ArrayList<EdgeNode>();
41  
42  	/**
43  	 * @param x
44  	 * @param y
45  	 */
46  	public void addNode( double x, double y )
47  	{
48  		EdgeNode node = new EdgeNode();
49  		node.vertex.x = x;
50  		node.vertex.y = y;
51  		m_List.add( node );
52  	}
53  
54  	/**
55  	 * @param index
56  	 * @return edge node
57  	 */
58  	public EdgeNode getNode( int index )
59  	{
60  		return (EdgeNode) m_List.get( index );
61  	}
62  
63  	/**
64  	 * @param i
65  	 * @return FWD_MIN
66  	 */
67  	public boolean FWD_MIN( int i )
68  	{
69  		EdgeNode prev = (EdgeNode) m_List.get( PolygonUtils.PREV_INDEX( i, m_List.size() ) );
70  		EdgeNode next = (EdgeNode) m_List.get( PolygonUtils.NEXT_INDEX( i, m_List.size() ) );
71  		EdgeNode ith = (EdgeNode) m_List.get( i );
72  		return ((prev.vertex.getY() >= ith.vertex.getY()) && (next.vertex.getY() > ith.vertex.getY()));
73  	}
74  
75  	/**
76  	 * @param i
77  	 * @return NOT_FMAX
78  	 */
79  	public boolean NOT_FMAX( int i )
80  	{
81  		EdgeNode next = (EdgeNode) m_List.get( PolygonUtils.NEXT_INDEX( i, m_List.size() ) );
82  		EdgeNode ith = (EdgeNode) m_List.get( i );
83  		return (next.vertex.getY() > ith.vertex.getY());
84  	}
85  
86  	/**
87  	 * @param i
88  	 * @return REV_MIN
89  	 */
90  	public boolean REV_MIN( int i )
91  	{
92  		EdgeNode prev = (EdgeNode) m_List.get( PolygonUtils.PREV_INDEX( i, m_List.size() ) );
93  		EdgeNode next = (EdgeNode) m_List.get( PolygonUtils.NEXT_INDEX( i, m_List.size() ) );
94  		EdgeNode ith = (EdgeNode) m_List.get( i );
95  		return ((prev.vertex.getY() > ith.vertex.getY()) && (next.vertex.getY() >= ith.vertex.getY()));
96  	}
97  
98  	/**
99  	 * @param i
100 	 * @return NOT_RMAX
101 	 */
102 	public boolean NOT_RMAX( int i )
103 	{
104 		EdgeNode prev = (EdgeNode) m_List.get( PolygonUtils.PREV_INDEX( i, m_List.size() ) );
105 		EdgeNode ith = (EdgeNode) m_List.get( i );
106 		return (prev.vertex.getY() > ith.vertex.getY());
107 	}
