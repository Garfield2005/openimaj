033package org.openimaj.image.typography.mathml;
034
035import java.awt.Color;
036import java.text.AttributedCharacterIterator.Attribute;
037import java.util.Map;
038
039import net.sourceforge.jeuclid.LayoutContext;
040import net.sourceforge.jeuclid.MutableLayoutContext;
041import net.sourceforge.jeuclid.context.LayoutContextImpl;
042import net.sourceforge.jeuclid.context.Parameter;
043
044import org.openimaj.image.typography.FontStyle;
045
046/**
047 * @param <T> 
048 *      
049 */
050public class MathMLFontStyle<T> extends FontStyle<T>
051{
052        enum MathInput{
053                MATHML,LATEX;
054        }
055
056        /**
057         * 
058         */
059        public static final Attribute TEXT_MODE = null;
060
061        private MathInput mathInput = MathInput.LATEX;
062
063        private boolean textMode;
064        protected MathMLFontStyle(MathMLFont font, T col) {
065                super(font, col);
066        }
067
068        /**
069         * @return the {@link LayoutContext} encoding of this {@link MathMLFontStyle}
070         */
071        public LayoutContext getLayoutContext() {
072                MutableLayoutContext mlc = new LayoutContextImpl(
073                LayoutContextImpl.getDefaultLayoutContext());
074        T col = this.colour;
075        if(col instanceof Float[]){
076                Float[] mcol = (Float[]) col;
077                mlc.setParameter(Parameter.MATHCOLOR, new Color(mcol[0], mcol[1], mcol[2]));
078        }
079        else if(col instanceof Float){
080                mlc.setParameter(Parameter.MATHCOLOR, new Color((Float)col,(Float)col,(Float)col));
081        }
082        mlc.setParameter(Parameter.MATHSIZE, this.getFontSize());
083                return mlc;
084        }
085
086        /**
087         * @return how to interruprate math text
088         */
089        public MathInput getMathInput() {
090                return this.mathInput ;
091        }
092        
093        /**
094         * @param input
095         */
096        public void setMathInput(MathInput input){
097                this.mathInput = input;
098        }
099
100        /**
101         * @return whether text
102         */
103        public boolean isTextMode() {
104                return this.textMode;
105        }
106        
107        @Override
108        public void parseAttributes(Map<? extends Attribute, Object> attrs) {
109                super.parseAttributes(attrs);
110                if (attrs.containsKey(MathMLFontStyle.TEXT_MODE)) this.textMode = ((Boolean) attrs.get(TEXT_MODE)).booleanValue();
111                
112        }
113        
114}
