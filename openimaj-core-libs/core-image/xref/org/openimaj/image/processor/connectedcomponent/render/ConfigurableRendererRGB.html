30  package org.openimaj.image.processor.connectedcomponent.render;
31  
32  import java.util.EnumMap;
33  import java.util.EnumSet;
34  
35  import org.openimaj.image.MBFImage;
36  import org.openimaj.image.pixel.ConnectedComponent;
37  import org.openimaj.image.pixel.ConnectedComponent.ConnectMode;
38  import org.openimaj.image.processor.connectedcomponent.ConnectedComponentProcessor;
39  import org.openimaj.math.geometry.shape.Polygon;

41  /**
42   *	This renderer encapsulates various other renderers in an easy to
43   *	configure class, allowing multiple renderers to be called in one go.
44   *	This class is designed for multiband RGB {@link MBFImage}s. 
45   * 	
46   *  @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
47   *	
48   */
49  public class ConfigurableRendererRGB implements ConnectedComponentProcessor 
50  {
51  	/** The image to draw into */
52  	protected MBFImage image;
53  
54  	/** The options to draw with */
55  	protected EnumSet<ConfigurableRenderOptions> options;
56  
57  	/** The colour to draw the various options in */
58  	EnumMap<ConfigurableRenderOptions, Float[]> colours = new EnumMap<ConfigurableRenderOptions, Float[]>(ConfigurableRenderOptions.class);
59  	
60  	/**
61  	 * 	Protected constructor that takes a set of options to draw.
62  	 * 
63  	 *  @param options The options to draw
64  	 */
65  	protected ConfigurableRendererRGB(EnumSet<ConfigurableRenderOptions> options) {
66  		this.options = options;
67  		
68  		colours.put(ConfigurableRenderOptions.AXIS, new Float[] {0.8F, 0.0F, 0.0F});
69  		colours.put(ConfigurableRenderOptions.BLOB, new Float[] {0.5F, 0.5F, 0.5F});
70  		colours.put(ConfigurableRenderOptions.BORDER, new Float[] {0.7F, 0.0F, 0.0F});
71  		colours.put(ConfigurableRenderOptions.CENTROID, new Float[] {0.7F, 0.0F, 0.0F});
72  		colours.put(ConfigurableRenderOptions.CH_AXIS, new Float[] {0.9F, 0.0F, 0.0F});
73  		colours.put(ConfigurableRenderOptions.CH_BLOB, new Float[] {0.2F, 0.2F, 0.2F});
74  		colours.put(ConfigurableRenderOptions.CH_BORDER, new Float[] {1.0F, 0.0F, 0.0F});
75  		colours.put(ConfigurableRenderOptions.CH_CENTROID, new Float[] {1.0F, 0.0F, 0.0F});
76  	}
77  	
78  	/**
79  	 * 	Constructor that takes an image to draw into and a set of options to draw.
80  	 * 
81  	 *  @param image The image to draw into
82  	 *  @param options The options to draw
83  	 */
84  	public ConfigurableRendererRGB(MBFImage image, EnumSet<ConfigurableRenderOptions> options) {
85  		this(options);
86  		
87  		this.image = image;
88  	}
89  	
90  	/**
91  	 * 	Constructor that creates an image to draw into and takes a set of options
92  	 * 	to draw.
93  	 * 
94  	 *  @param rows The height of the image to create
95  	 *  @param cols The width of the image to create
96  	 *  @param options The set of options to draw
97  	 */
98  	public ConfigurableRendererRGB(int rows, int cols, EnumSet<ConfigurableRenderOptions> options) {
99  		this(options);
100 		
101 		image = new MBFImage(cols, rows, 3);
102 	}
103 	
104 	/**
105 	 * 	Get the rendered image.
106 	 * 
107 	 *  @return The rendered image.
108 	 */
109 	public MBFImage getImage() {
110 		return image;
111 	}
112 
113 	/**
114 	 *  {@inheritDoc}
115 	 *  @see org.openimaj.image.processor.connectedcomponent.ConnectedComponentProcessor#process(org.openimaj.image.pixel.ConnectedComponent)
116 	 */
117 	@Override
118 	public void process(ConnectedComponent cc) {
119 		Polygon ch = cc.calculateConvexHull();
120 		ConnectedComponent chcc = new ConnectedComponent(ch);
121 		
122 		if (options.contains(ConfigurableRenderOptions.CH_BLOB))
123 			chcc.process(new BlobRenderer<Float[]>(image, colours.get(ConfigurableRenderOptions.CH_BLOB)));
124 		
125 		if (options.contains(ConfigurableRenderOptions.BLOB))
126 			cc.process(new BlobRenderer<Float[]>(image, colours.get(ConfigurableRenderOptions.BLOB)));
127 		
128 		if (options.contains(ConfigurableRenderOptions.CH_BORDER)) 
129 			image.createRenderer().drawPolygon(ch, colours.get(ConfigurableRenderOptions.CH_BORDER));
130 		
131 		if (options.contains(ConfigurableRenderOptions.BORDER)) 
132 			cc.process(new BorderRenderer<Float[]>(image, colours.get(ConfigurableRenderOptions.BLOB), ConnectMode.CONNECT_8));
133 		
134 		if (options.contains(ConfigurableRenderOptions.CH_CENTROID)) 
135 			chcc.process(new CentroidRenderer<Float[]>(image, colours.get(ConfigurableRenderOptions.CH_CENTROID)));
136 		
137 		if (options.contains(ConfigurableRenderOptions.CENTROID))
138 			cc.process(new CentroidRenderer<Float[]>(image, colours.get(ConfigurableRenderOptions.CENTROID)));
139 
140 		if (options.contains(ConfigurableRenderOptions.CH_AXIS))
141 			chcc.process(new AxisRenderer<Float[]>(image, colours.get(ConfigurableRenderOptions.CH_AXIS)));
142 		
143 		if (options.contains(ConfigurableRenderOptions.AXIS))
144 			cc.process(new AxisRenderer<Float[]>(image, colours.get(ConfigurableRenderOptions.AXIS)));
145 	}
146 	
147 	/**
148 	 * 	Set the colour of a specific option to override the default.
149 	 * 
150 	 *  @param r The option to set the colour for
151 	 *  @param colour The colour to draw that option in.
152 	 */
153 	public void setColour(ConfigurableRenderOptions r, Float[] colour) {
154 		colours.put(r, colour);
155 	}
156 	
157 	/**
158 	 * 	Get the colour that a specific option will be drawn in. 
159 	 * 
160 	 *  @param r The option to get the colour of.
161 	 *  @return The colour the option will be drawn in.
162 	 */
163 	public Float[] getColour(ConfigurableRenderOptions r) {
164 		return colours.get(r);
165 	}
