30  package org.openimaj.util.queue;
31  
32  import java.util.Comparator;
33  import java.util.PriorityQueue;
34  
35  /**
36   * This class provides an inverted {@link PriorityQueue} implementation, where
37   * objects that are higher (according to the provided {@link Comparator} or the
38   * natural order) come first.
39   * <p>
40   * The Iterator provided in method {@link #iterator()} is <em>not</em>
41   * guaranteed to traverse the elements of the priority queue in any particular
42   * order.
43   * 
44   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
45   * 
46   * @param <T>
47   *            Type of objects stored in the queue
48   */
49  public class InvertedPriorityQueue<T> extends PriorityQueue<T> {
50  	private static final long serialVersionUID = 1L;
51  
52  	private static final int DEFAULT_INITIAL_CAPACITY = 11;
53  
54  	/**
55  	 * Creates a {@code InvertedPriorityQueue} with the default initial capacity
56  	 * (11) that orders its elements according to their inverted
57  	 * {@linkplain Comparable natural ordering}.
58  	 */
59  	public InvertedPriorityQueue() {
60  		super(DEFAULT_INITIAL_CAPACITY, InvertedComparableComparator.INSTANCE);
61  	}
62  
63  	/**
64  	 * Creates a {@code InvertedPriorityQueue} with the specified initial
65  	 * capacity that orders its elements according to the inverse of the
66  	 * specified comparator.
67  	 * 
68  	 * @param initialCapacity
69  	 *            the initial capacity for this priority queue
70  	 * @param comparator
71  	 *            the comparator that will be used to order this priority queue.
72  	 *            If {@code null}, the {@linkplain Comparable natural ordering}
73  	 *            of the elements will be used. Internally, the comparator is
74  	 *            inverted to reverse its meaning.
75  	 * @throws IllegalArgumentException
76  	 *             if {@code initialCapacity} is less than 1
77  	 */
78  	@SuppressWarnings("unchecked")
79  	public InvertedPriorityQueue(int initialCapacity, Comparator<? super T> comparator) {
80  		super(initialCapacity, comparator == null ? (Comparator<T>) InvertedComparableComparator.INSTANCE
81  				: new InvertedComparator<T>(comparator));
82  	}
83  
84  	/**
85  	 * Creates a {@code InvertedPriorityQueue} with the specified initial
86  	 * capacity that orders its elements according to their inverse
87  	 * {@linkplain Comparable natural ordering}.
88  	 * 
89  	 * @param initialCapacity
90  	 *            the initial capacity for this priority queue
91  	 * @throws IllegalArgumentException
92  	 *             if {@code initialCapacity} is less than 1
93  	 */
94  	public InvertedPriorityQueue(int initialCapacity) {
95  		super(initialCapacity, InvertedComparableComparator.INSTANCE);
96  	}
97  
98  	protected Comparator<? super T> originalComparator() {
99  		if (comparator() instanceof InvertedComparator)
100 			return ((InvertedComparator<? super T>) comparator()).innerComparator;
101 		else
102 			return ComparableComparator.INSTANCE;
103 	}
104 
105 	/**
106 	 * Inverted natural order comparator.
107 	 * 
108 	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
109 	 */
110 	private static class InvertedComparableComparator implements Comparator<Object> {
111 		public static final InvertedComparableComparator INSTANCE = new InvertedComparableComparator();
112 
113 		@Override
114 		@SuppressWarnings("unchecked")
115 		public int compare(Object o1, Object o2) {
116 			return ((Comparable<Object>) o2).compareTo(o1);
117 		}
118 	}
119 
120 	/**
121 	 * Natural order comparator.
122 	 * 
123 	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
124 	 */
125 	private static class ComparableComparator implements Comparator<Object> {
126 		public static final ComparableComparator INSTANCE = new ComparableComparator();
127 
128 		@Override
129 		@SuppressWarnings("unchecked")
130 		public int compare(Object o1, Object o2) {
131 			return ((Comparable<Object>) o1).compareTo(o2);
132 		}
133 	}
134 
135 	/**
136 	 * Inverted natural order comparator.
137 	 * 
138 	 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
139 	 */
140 	private static class InvertedComparator<T> implements Comparator<T> {
141 		protected Comparator<? super T> innerComparator;
142 
143 		public InvertedComparator(Comparator<? super T> innerComparator) {
144 			this.innerComparator = innerComparator;
145 		}
146 
147 		@Override
148 		public int compare(T o1, T o2) {
149 			return innerComparator.compare(o2, o1);
150 		}
151 	}
152 }
