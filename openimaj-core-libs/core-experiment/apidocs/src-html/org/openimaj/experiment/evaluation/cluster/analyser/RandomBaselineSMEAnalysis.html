030package org.openimaj.experiment.evaluation.cluster.analyser;
031
032import net.sf.jasperreports.engine.JRException;
033import net.sf.jasperreports.engine.JasperPrint;
034
035import org.openimaj.experiment.evaluation.AnalysisResult;
036
037/**
038 * Results of applying a RandomBaselineSMEAnalysis to evaluate clustering. These
039 * metrics are taken from the <a href=
040 * "http://nlp.stanford.edu/IR-book/html/htmledition/evaluation-of-clustering-1.html"
041 * >Introduction to Information Retrieval</a>.
042 * 
043 * Sina Samangooei (ss@ecs.soton.ac.uk)
044 * 
045 */
046public class RandomBaselineSMEAnalysis implements AnalysisResult {
047
048        /**
049         * A measure of how pure each cluster is.
050         * <code>P = 1/N Sigma_k max_j | w_k AND c_j |</code>
051         * <p>
052         * Count the true classes of all the elements in a class, make a count of
053         * the largest group from each cluster, divide by number of elements in all
054         * clusters.
055         * <p>
056         * High means: most of the clusters had a high number of a single class Low
057         * means: most of the clusters had a roughly equal spread of all the classes
058         */
059        public RandomBaselineClusterAnalysis<PurityAnalysis> purity;
060
061        /**
062         * F-Score
063         */
064        public RandomBaselineClusterAnalysis<FScoreAnalysis> fscore;
065
066        /**
067         * 
068         */
069        public RandomBaselineClusterAnalysis<RandomIndexAnalysis> randIndex;
070
071        /**
072         * General statistics
073         */
074        public ClusterStatsAnalysis stats;
075
076        /**
077         * The correct clustering
078         */
079        public int[][] correct;
080
081        /**
082         * The estimation
083         */
084        public int[][] estimated;
085
086        /**
087         * Construct with the correct assignments and estimated assignments.
088         * 
089         * @param correct
090         * @param estimated
091         */
092        public RandomBaselineSMEAnalysis(int[][] correct, int[][] estimated) {
093                this.correct = correct;
094                this.estimated = estimated;
095        }
096
097        @Override
098        public String getSummaryReport() {
099                return String.format("s=[%s],p=[%s],f1=[%s],r=[%s])", stats, purity, fscore, randIndex);
100        }
101
102        @Override
103        public String getDetailReport() {
104                return String.format("stats: [%s]\npurity: [%s]\nf1: [%s]\nrandIndex: [%s])", stats, purity, fscore, randIndex);
105        }
106
107        @Override
108        public String toString() {
109                return getDetailReport();
110        }
111
112        @Override
113        public JasperPrint getSummaryReport(String title, String info) throws JRException {
114                throw new UnsupportedOperationException();
115        }
116
117        @Override
118        public JasperPrint getDetailReport(String title, String info) throws JRException {
119                throw new UnsupportedOperationException();
120        }
121
122}
