030package org.openimaj.video;
031
032import org.openimaj.image.Image;
033
034/**
035 * A basic abstract implementation of a video that displays an image and
036 * provides double-buffering
037 * 
038 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
039 * 
040 * @param <I>
041 */
042public abstract class AnimatedVideo<I extends Image<?, I>> extends Video<I>
043{
044        private I currentFrame;
045        private I nextFrame;
046        private double fps;
047
048        /**
049         * Default video constructor with a rate of 30 fps using the given image as
050         * a basis.
051         * 
052         * @param blankFrame
053         *            blank video frame to pass to the update method
054         */
055        public AnimatedVideo(I blankFrame) {
056                this(blankFrame, 30);
057        }
058
059        /**
060         * Default video constructor with the given rate using the given image as a
061         * basis.
062         * 
063         * @param blankFrame
064         *            blank video frame to pass to the update method
065         * @param fps
066         *            the frame rate
067         */
068        public AnimatedVideo(I blankFrame, double fps) {
069                currentFrame = blankFrame;
070                nextFrame = blankFrame.clone();
071                this.fps = fps;
072
073                init();
074        }
075
076        protected abstract void updateNextFrame(I frame);
077
078        @Override
079        public I getNextFrame() {
080                updateNextFrame(nextFrame);
081                final I tmp = currentFrame;
082                currentFrame = nextFrame;
083                nextFrame = tmp;
084
085                super.currentFrame++;
086
087                return currentFrame;
088        }
089
090        @Override
091        public I getCurrentFrame() {
092                return currentFrame;
093        }
094
095        @Override
096        public int getWidth() {
097                return currentFrame.getWidth();
098        }
099
100        @Override
101        public int getHeight() {
102                return currentFrame.getHeight();
103        }
104
105        @Override
106        public long getTimeStamp() {
107                return (long) (super.currentFrame * 1000 / this.fps);
108        }
109
110        @Override
111        public double getFPS() {
112                return fps;
113        }
114
115        @Override
116        public boolean hasNextFrame() {
117                return true;
118        }
119
120        @Override
121        public long countFrames() {
122                return -1;
123        }
124
125        @Override
126        public void reset() {
127                super.currentFrame = 0;
128        }
129
130        /**
131         * Called by the constructor. Does nothing by default.
132         */
133        protected void init() {
134                // do nothing by default, but might be overridden to do things with
135                // animators
136        }
137}
