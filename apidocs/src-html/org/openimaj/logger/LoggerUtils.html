030package org.openimaj.logger;
031
032
033import org.apache.log4j.ConsoleAppender;
034import org.apache.log4j.Level;
035import org.apache.log4j.Logger;
036import org.apache.log4j.PatternLayout;
037
038/**
039 * Make using log4j slightly less awful
040 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
041 *
042 */
043public class LoggerUtils {
044        /**
045         * Prepare a console logger with a default layout.
046         * This allows {@link Logger} instances to be used while
047         * still being useful as the line to prepare logger can be removed
048         * and the default log4j.properties will be correctly listened to.
049         */
050        public static void prepareConsoleLogger(){
051                  ConsoleAppender console = new ConsoleAppender(); //create appender
052                  //configure the appender
053                  String PATTERN = "%m\n";
054                  console.setLayout(new PatternLayout(PATTERN)); 
055                  console.setThreshold(Level.DEBUG);
056                  console.activateOptions();
057                  //add appender to any Logger (here is root)
058                  Logger.getRootLogger().addAppender(console);
059        }
060
061        /**
062         * Debug message occationally
063         * @param logger
064         * @param message
065         * @param b
066         */
067        public static void debug(Logger logger, String message, boolean b) {
068                if(b){
069                        logger.debug(message);
070                }
071        }
072
073        /**
074         * Calls {@link #format(Logger, String, Level, Object...)} with level {@link Level#DEBUG}
075         * @param logger
076         * @param string
077         * @param obj
078         */
079        public static void debugFormat(Logger logger, String string, Object ... obj) {
080                format(logger,string,Level.DEBUG,obj);
081        }
082
083        /**
084         * Checks the level, if acceptable calls {@link String#format(String, Object...)} at the appropriate level
085         * @param logger
086         * @param string
087         * @param debug
088         * @param obj
089         */
090        public static void format(Logger logger, String string, Level debug, Object ... obj) {
091                Level l = logger.getEffectiveLevel();
092                if(debug.isGreaterOrEqual(l)){
093                        logger.log(debug, String.format(string,obj));
094                }
095        }
096}
