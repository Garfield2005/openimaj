033package org.openimaj.audio.features;
034
035import org.openimaj.audio.AudioStream;
036import org.openimaj.audio.samples.SampleBuffer;
037
038/**
039 *      A wrapper around the MFCC implementation of jAudio (which itself
040 *      is a wrapper around the OrangeCow Volume implementation of FFT).
041 *
042 *      @author David Dupplaw (dpd@ecs.soton.ac.uk)
043 *  @created 5 Mar 2013
044 *      @version $Author$, $Revision$, $Date$
045 */
046public class MFCC extends JAudioFeatureExtractor
047{
048        /** Processor for magnitude spectrum */
049        private MagnitudeSpectrum magSpec = null;
050
051        /**
052         *      Default constructor
053         */
054        public MFCC()
055        {
056                this.featureExtractor = new jAudioFeatureExtractor.AudioFeatures.MFCC();
057                this.magSpec = new MagnitudeSpectrum();
058        }
059
060        /**
061         *      Constructor for chaining to a stream
062         *      @param as The audio stream
063         */
064        public MFCC( final AudioStream as )
065        {
066                super( as );
067                this.featureExtractor = new jAudioFeatureExtractor.AudioFeatures.MFCC();
068                this.magSpec = new MagnitudeSpectrum();
069        }
070
071        /**
072         *      Calculate the MFCCs for the given sample buffer.
073         *      @param sb The sample buffer
074         *      @return The MFCCs
075         */
076        public double[][] calculateMFCC( final SampleBuffer sb )
077        {
078                this.process( sb );
079                return this.getLastCalculatedFeature();
080        }
081
082        /**
083         *      {@inheritDoc}
084         *      @see org.openimaj.audio.features.JAudioFeatureExtractor#getExtraInputs(double[], double)
085         */
086        @Override
087        public double[][] getExtraInputs( final double[] samples, final double sampleRate )
088        {
089                return new double[][] { this.magSpec.process( samples, sampleRate ) };
090        }
091
092        /**
093         *      Returns the MFCCs with the first coefficient set to zero.
094         *      @return The MFCCs with the first set to zero.
095         */
096        public double[][] getLastCalculatedFeatureWithoutFirst()
097        {
098                final double[][] d = this.getLastCalculatedFeature();
099                for( final double[] dd : d )
100                        dd[0] = 0;
101                return d;
102        }
103}
