058/**
059 * Output the word/time values for each word
060 *
061 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
062 *
063 */
064public class Values extends StageProvider {
065        private String outputPath;
066        private int valueReduceSplit;
067        private boolean sortValueByTime;
068        private boolean matlabOutput;
069
070        /**
071         * Assign the output path for the stage
072         *
073         * @param outputPath
074         * @param sortValueByTime
075         * @param matlabOutput
076         */
077        public Values(String outputPath, int valueReduceSplit, boolean sortValueByTime, boolean matlabOutput) {
078                this.outputPath = outputPath;
079                this.valueReduceSplit = valueReduceSplit;
080                this.sortValueByTime = sortValueByTime;
081                this.matlabOutput = matlabOutput;
082        }
083
084        /**
085         * The index location config option
086         */
087        public static final String ARGS_KEY = "INDEX_ARGS";
088        public static final String MATLAB_OUT = "org.openimaj.hadoop.tools.twitter.token.outputmode.sparsecsv.matlab_out";
089
090        @Override
091        public SequenceFileTextStage<?, ?, ?, ?, ?, ?> stage() {
092                if (this.sortValueByTime) {
093                        return new SequenceFileTextStage<Text, BytesWritable, LongWritable, BytesWritable, NullWritable, Text>() {
094                                @Override
095                                public void setup(Job job) {
096                                        job.setNumReduceTasks(valueReduceSplit);
097                                        job.getConfiguration().setStrings(Values.ARGS_KEY, new String[] { outputPath.toString() });
098                                        job.getConfiguration().setBoolean(MATLAB_OUT, matlabOutput);
099                                }
100
101                                @Override
102                                public Class<? extends Mapper<Text, BytesWritable, LongWritable, BytesWritable>> mapper() {
103                                        return MapValuesByTime.class;
104                                }
105
106                                @Override
107                                public Class<? extends Reducer<LongWritable, BytesWritable, NullWritable, Text>> reducer() {
108                                        return ReduceValuesByTime.class;
109                                }
110
111                                @Override
112                                public String outname() {
113                                        return "values";
114                                }
115
116                                @Override
117                                public void finished(Job job) {
118                                        if (matlabOutput) {
119                                                try {
120                                                        WordIndex.writeToMatlab(outputPath.toString());
121                                                        TimeIndex.writeToMatlab(outputPath.toString());
122                                                        System.out.println("Done writing the word and time index files to matlab");
123                                                } catch (final IOException e) {
124                                                        System.out.println("Failed to write the word and time index files");
125                                                }
126                                        }
127                                }
128                        };
129                }
130                else {
131                        return new SimpleSequenceFileTextStage<Text, BytesWritable, NullWritable, Text>() {
132                                @Override
133                                public void setup(Job job) {
134                                        job.setNumReduceTasks(valueReduceSplit);
135                                        job.getConfiguration().setStrings(Values.ARGS_KEY, new String[] { outputPath.toString() });
136                                }
137
138                                @Override
139                                public Class<? extends Mapper<Text, BytesWritable, NullWritable, Text>> mapper() {
140                                        return MapValuesByWord.class;
141                                }
142
143                                @Override
144                                public Class<? extends Reducer<NullWritable, Text, NullWritable, Text>> reducer() {
145                                        return ReduceValuesByWord.class;
146                                }
147
148                                @Override
149                                public String outname() {
150                                        return "values";
151                                }
152                        };
153                }
154        }
155
156        /**
157         * Construct a time series per word
158         *
159         * @param path
160         * @param timeIndex
161         * @param wordIndex
162         * @return hashmap containing a {@link WordDFIDFTimeSeries} instance per
163         *         word
164         * @throws IOException
165         */
166        public static LinkedHashMap<String, WordDFIDFTimeSeries> readWordDFIDF(String path,
167                        LinkedHashMap<Long, IndependentPair<Long, Long>> timeIndex,
168                        LinkedHashMap<String, IndependentPair<Long, Long>> wordIndex) throws IOException
169                        {
170                final LinkedHashMap<String, WordDFIDFTimeSeries> tsMap = new LinkedHashMap<String, WordDFIDFTimeSeries>();
171
172                final long[] timeReverseIndex = new long[timeIndex.size()];
173                for (final Entry<Long, IndependentPair<Long, Long>> l : timeIndex.entrySet()) {
174                        final long lineNum = l.getValue().secondObject();
175                        timeReverseIndex[(int) lineNum] = l.getKey();
176                }
177
178                final String[] wordReverseIndex = new String[wordIndex.size()];
179                for (final Entry<String, IndependentPair<Long, Long>> w : wordIndex.entrySet()) {
180                        final long lineNum = w.getValue().secondObject();
181                        wordReverseIndex[(int) lineNum] = w.getKey();
182                }
183                final String wordPath = path + "/values";
184                final Path p = HadoopToolsUtil.getInputPaths(wordPath)[0];
185                final FileSystem fs = HadoopToolsUtil.getFileSystem(p);
186                final FSDataInputStream toRead = fs.open(p);
187                final BufferedReader reader = new BufferedReader(new InputStreamReader(toRead));
188                final CSVParser csvreader = new CSVParser(reader);
189                String[] next = null;
190
191                while ((next = csvreader.getLine()) != null && next.length > 0) {
192                        // writer.writeln(new String[]{wordI + "",timeI + "",idf.wf +
193                        // "",idf.tf + "",idf.Twf + "", idf.Ttf + ""});
194                        final int wordI = Integer.parseInt(next[0]);
195                        final int timeI = Integer.parseInt(next[1]);
196                        final long wf = Long.parseLong(next[2]);
197                        final long tf = Long.parseLong(next[3]);
198                        final long Twf = Long.parseLong(next[4]);
199                        final long Ttf = Long.parseLong(next[5]);
200                        final long time = timeReverseIndex[timeI];
201                        final WordDFIDF wordDFIDF = new WordDFIDF(time, wf, tf, Twf, Ttf);
202                        final String word = wordReverseIndex[wordI];
203                        WordDFIDFTimeSeries current = tsMap.get(word);
204                        if (current == null) {
205                                tsMap.put(word, current = new WordDFIDFTimeSeries());
206                        }
207                        current.add(time, wordDFIDF);
208                }
209
210                return tsMap;
211                        }
