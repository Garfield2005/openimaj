048/**
049 * A {@link Dataset} instance of the standard wine clustering experiment found
050 * here:
051 *
052 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
053 *
054 */
055@DatasetDescription(
056                name = "Wine Data Set",
057                description = "" +
058                                "These data are the results of a chemical analysis of wines grown in the same region in " +
059                                "Italy but derived from three different cultivars. The analysis determined the quantities " +
060                                "of 13 constituents found in each of the three types of wines." +
061                                "" +
062                                "I think that the initial data set had around 30 variables, but for some reason I only have " +
063                                "the 13 dimensional version. I had a list of what the 30 or so variables were, but a.) I lost" +
064                                " it, and b.), I would not know which 13 variables are included in the set." +
065                                "" +
066                                "The attributes are (dontated by Riccardo Leardi, riclea '@' anchem.unige.it )" +
067                                "1) Alcohol" +
068                                "2) Malic acid" +
069                                "3) Ash" +
070                                "4) Alcalinity of ash" +
071                                "5) Magnesium" +
072                                "6) Total phenols" +
073                                "7) Flavanoids" +
074                                "8) Nonflavanoid phenols" +
075                                "9) Proanthocyanins" +
076                                "10)Color intensity" +
077                                "11)Hue" +
078                                "12)OD280/OD315 of diluted wines" +
079                                "13)Proline" +
080                                "" +
081                                "In a classification context, this is a well posed problem with \"well behaved\" class structures." +
082                                " A good data set for first testing of a new classifier, but not very challenging. ",
083                                creator = "Forina, M. et al, PARVUS - ",
084                                url = "http://archive.ics.uci.edu/ml/datasets/Wine",
085                                downloadUrls = {
086                                "http://archive.ics.uci.edu/ml/machine-learning-databases/wine/wine.data"
087                })
088public class WineDataset extends MapBackedDataset<Integer, ListDataset<double[]>, double[]> {
089        final static Logger logger = Logger.getLogger(WineDataset.class);
090
091        /**
092         * Loads the wine dataset, mean centres the dataset
093         * 
094         * @param clusters
095         *            valid clusters, if empty all clusters are chosen
096         */
097        public WineDataset(Integer... clusters) {
098                this(true, clusters);
099        }
100
101        /**
102         * Loads the wine dataset from wine.data
103         * 
104         * @param normalise
105         *            whether to mean center the dataset
106         * @param clusters
107         *            valid clusters, if empty all clusters are chosen
108         */
109        public WineDataset(boolean normalise, Integer... clusters) {
110                final BufferedReader br = new BufferedReader(
111                                new InputStreamReader(WineDataset.class.getResourceAsStream("wine.data")));
112                String line = null;
113                Vector mean = null;
114                Set<Integer> clusterSet = null;
115                if (clusters.length != 0) {
116                        clusterSet = new HashSet<Integer>();
117                        clusterSet.addAll(Arrays.asList(clusters));
118                }
119
120                try {
121                        while ((line = br.readLine()) != null) {
122                                final String[] parts = line.split(",");
123                                final int cluster = Integer.parseInt(parts[0].trim());
124                                if (clusterSet != null && !clusterSet.contains(cluster))
125                                        continue;
126                                final double[] data = new double[parts.length - 1];
127                                for (int i = 0; i < data.length; i++) {
128                                        data[i] = Double.parseDouble(parts[i + 1]);
129                                }
130
131                                ListDataset<double[]> ds = this.get(cluster);
132                                if (ds == null)
133                                        this.put(cluster, ds = new ListBackedDataset<double[]>());
134                                ds.add(data);
135                                final Vector copyArray = VectorFactory.getDefault().copyArray(data);
136                                if (mean == null) {
137                                        mean = copyArray.clone();
138                                }
139                                else {
140                                        mean.plusEquals(copyArray);
141                                }
142                        }
143                        mean.scaleEquals(1. / this.numInstances());
144                        if (normalise) {
145                                normalise(mean);
146                        }
147                } catch (final Exception e) {
148                        logger.error("Wine dataset failed to load", e);
149                }
150        }
151
152        private void normalise(Vector mean) {
153                for (final double[] data : this) {
154                        for (int i = 0; i < data.length; i++) {
155                                data[i] -= mean.getElement(i);
156                        }
157                }
158        }
159}
