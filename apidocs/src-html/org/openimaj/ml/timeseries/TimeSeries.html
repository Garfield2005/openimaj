030package org.openimaj.ml.timeseries;
031
032import org.openimaj.ml.timeseries.converter.TimeSeriesConverter;
033import org.openimaj.ml.timeseries.processor.TimeSeriesProcessor;
034import org.openimaj.util.pair.IndependentPair;
035
036/**
037 * A time series defines data at discrete points in time. The time series has
038 * the ability to return data at a specific point in time, return neighbours
039 * within some window, closest neighbours or n neighbours before and after a
040 * time.
041 * <p>
042 * These values can be used by a {@link TimeSeriesInterpolation} to get specific
043 * moments in time
044 * 
045 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
046 * 
047 * @param <DATA>
048 *            the type of the data at each point in time
049 * @param <SINGLE_TYPE>
050 *            the type of the an element at a single point in time
051 * @param <RETURNTYPE>
052 *            the time series returned by the get
053 * 
054 */
055public abstract class TimeSeries<DATA, SINGLE_TYPE, RETURNTYPE extends TimeSeries<DATA, SINGLE_TYPE, RETURNTYPE>>
056                implements
057                        Iterable<IndependentPair<Long, SINGLE_TYPE>>
058{
059
060        /**
061         * Same as calling {@link #get(long, int, int)} with spans as 0
062         * 
063         * @param time
064         * @return the requested data or null.
065         */
066        public RETURNTYPE get(long time) {
067                return get(time, 0, 0);
068        }
069
070        /**
071         * returns the DATA at a specific point in time and those before and after
072         * to the number requested. This method may not return data for the specific
073         * requested time if it does not exists. If the time series is completely
074         * empty this function may return at an empty array however if at least 1
075         * data point exists and either nbefore and nafter are bigger than 1 then at
076         * least 1 datapoint will be returned.
077         * <p>
078         * Data should be returned in order
079         * 
080         * @param time
081         * @param nbefore
082         * @param nafter
083         * @return all data found with these parameters
084         */
085        public abstract RETURNTYPE get(long time, int nbefore, int nafter);
086
087        /**
088         * Same as {@link #get(long, int, int)} but instead of createing the output
089         * DATA instance, an existing data instance is handed which is filled. For
090         * convenience this output is also returned
091         * <p>
092         * Data should be returned in order
093         * 
094         * @param time
095         * @param nbefore
096         * @param nafter
097         * @param output
098         * @return all data found with these parameters
099         */
100        public abstract RETURNTYPE get(long time, int nbefore, int nafter, RETURNTYPE output);
101
102        /**
103         * returns the RETURNTYPE at a specific point in time and those before and
104         * after within the specified thresholds. This method may not return data
105         * for the specific requested time if it does not exists. Similarly this
106         * method may return an empty array if no time data is available within the
107         * window specified.
108         * <p>
109         * Data should be returned in order
110         * 
111         * @param time
112         * @param threshbefore
113         * @param threshafter
114         * @return all data found with these parameters
115         */
116        public abstract RETURNTYPE get(long time, long threshbefore, long threshafter);
117
118        /**
119         * returns the RETURNTYPE between the specified time periods. This method
120         * may not return data for the specific requested time if it does not
121         * exists. Similarly this method may return an empty array if no time data
122         * is available within the window specified.
123         * <p>
124         * Data should be returned in order
125         * 
126         * @param start
127         * @param end
128         * @return all data found with these parameters
129         */
130        public abstract RETURNTYPE get(long start, long end);
131
132        /**
133         * Set the data associated with each time. This function explicitly assumes
134         * that time.length == data.length and there exists a single data instance
135         * per time instance
136         * 
137         * @param time
138         *            instances of time
139         * @param data
140         *            instances of data
141         * @throws TimeSeriesSetException
142         */
143        public abstract void set(long[] time, DATA data) throws TimeSeriesSetException;
144
145        /**
146         * @return all times
147         */
148        public abstract long[] getTimes();
149
150        /**
151         * @return all data
152         */
153        public abstract DATA getData();
154
155        /**
156         * @return an empty new instance of this timeseries type
157         */
158        public abstract RETURNTYPE newInstance();
159
160        /**
161         * @return the number of valid time steps in this timeseries
162         */
163        public abstract int size();
164
165        /**
166         * @param interpolate
167         *            assign this timeseries to the internal one, efforts should be
168         *            made to copy the data, not simply assign it
169         */
170        public abstract void internalAssign(RETURNTYPE interpolate);
171
172        /**
173         * @param times
174         * @param data
175         */
176        public void internalAssign(long[] times, DATA data) {
177                try {
178                        this.set(times, data);
179                } catch (final TimeSeriesSetException e) {
180                }
181        }
182
183        /**
184         * @return clone this time series
185         */
186        @SuppressWarnings("unchecked")
187        public RETURNTYPE copy() {
188                final RETURNTYPE t = newInstance();
189                t.internalAssign((RETURNTYPE) this);
190                return t;
191        }
192
193        /**
194         * process using the provided processor, return
195         * 
196         * @param tsp
197         * @return a new instance processed
198         */
199        public RETURNTYPE process(TimeSeriesProcessor<DATA, SINGLE_TYPE, RETURNTYPE> tsp) {
200                final RETURNTYPE copy = copy();
201                tsp.process(copy);
202                return copy;
203        }
204
205        @SuppressWarnings("unchecked")
206        private RETURNTYPE self() {
207                return (RETURNTYPE) this;
208        }
209
210        /**
211         * Process using the provided processor
212         * 
213         * @param tsp
214         * @return this object processed inplace
215         */
216        public RETURNTYPE processInplace(TimeSeriesProcessor<DATA, SINGLE_TYPE, RETURNTYPE> tsp) {
217                tsp.process(self());
218                return self();
219        }
220
221        /**
222         * Convert a {@link TimeSeries}
223         * 
224         * @param <OUTDATA>
225         * @param <OUTSING>
226         * @param <OUTRET>
227         * @param converter
228         *            the converter
229         * @return the converted timeseries
230         */
231        public <OUTDATA, OUTSING, OUTRET extends TimeSeries<OUTDATA, OUTSING, OUTRET>> OUTRET convert(
232                        TimeSeriesConverter<DATA, SINGLE_TYPE, RETURNTYPE, OUTDATA, OUTSING, OUTRET> converter)
233        {
234                return converter.convert(self());
235        }
236
237        /**
238         * Convert a {@link TimeSeries}
239         * 
240         * @param <OUTDATA>
241         * @param <OUTSING>
242         * @param <OUTRET>
243         * @param converter
244         *            the converter
245         * @param tsp
246         *            the processor
247         * @return the converted timeseries
248         */
249        public <OUTDATA, OUTSING, OUTRET extends TimeSeries<OUTDATA, OUTSING, OUTRET>> OUTRET convert(
250                        TimeSeriesConverter<DATA, SINGLE_TYPE, RETURNTYPE, OUTDATA, OUTSING, OUTRET> converter,
251                        TimeSeriesProcessor<OUTDATA, OUTSING, OUTRET> tsp)
252        {
253                return converter.convert(self(), tsp);
254        }
255
256        @Override
257        public abstract String toString();
258}
