030package org.openimaj.content.slideshow;
031
032import java.awt.image.BufferedImage;
033import java.io.IOException;
034import java.util.ArrayList;
035import java.util.List;
036
037import javax.swing.JFrame;
038
039/**
040 * A slideshow that can be used in a Java Swing application. The slideshow is
041 * created in a new window (JFrame).
042 * 
043 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
044 * 
045 */
046public class SlideshowApplication extends Slideshow {
047        protected FullscreenUtility fsutil;
048
049        /**
050         * Default constructor.
051         * 
052         * @param slides
053         *            The slides to display.
054         * @param slideWidth
055         *            The slide width.
056         * @param slideHeight
057         *            The slide height.
058         * @param background
059         *            The background image.
060         * @throws IOException
061         */
062        public SlideshowApplication(List<Slide> slides, int slideWidth, int slideHeight, BufferedImage background)
063                        throws IOException
064        {
065                super(new JFrame(), slides, slideWidth, slideHeight, background);
066                fsutil = new FullscreenUtility((JFrame) container);
067        }
068
069        /**
070         * Default constructor.
071         * 
072         * @param slide
073         *            The slide to display.
074         * @param slideWidth
075         *            The slide width.
076         * @param slideHeight
077         *            The slide height.
078         * @param background
079         *            The background image.
080         * @throws IOException
081         */
082        public SlideshowApplication(Slide slide, int slideWidth, int slideHeight, BufferedImage background)
083                        throws IOException
084        {
085                super(new JFrame(), createList(slide), slideWidth, slideHeight, background);
086                fsutil = new FullscreenUtility((JFrame) container);
087        }
088
089        /**
090         * Default constructor.
091         * 
092         * @param slide
093         *            The slide to display.
094         * @param slideWidth
095         *            The slide width.
096         * @param slideHeight
097         *            The slide height.
098         * @throws IOException
099         */
100        public SlideshowApplication(Slide slide, int slideWidth, int slideHeight)
101                        throws IOException
102        {
103                super(new JFrame(), createList(slide), slideWidth, slideHeight, null);
104                fsutil = new FullscreenUtility((JFrame) container);
105        }
106
107        private static List<Slide> createList(Slide... slides) {
108                final List<Slide> slidesl = new ArrayList<Slide>();
109
110                for (final Slide s : slides)
111                        slidesl.add(s);
112
113                return slidesl;
114        }
115
116        @Override
117        protected void pack() {
118                ((JFrame) container).pack();
119        }
120
121        @Override
122        public void setFullscreen(boolean fullscreen) {
123                fsutil.setFullscreen(fullscreen);
124        }
125
126        @Override
127        protected boolean isFullscreen() {
128                return fsutil.fullscreen;
129        }
130}
