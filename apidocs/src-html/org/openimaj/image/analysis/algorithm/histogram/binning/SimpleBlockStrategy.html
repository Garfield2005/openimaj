030package org.openimaj.image.analysis.algorithm.histogram.binning;
031
032import org.openimaj.image.analysis.algorithm.histogram.WindowedHistogramExtractor;
033import org.openimaj.image.pixel.sampling.RectangleSampler;
034import org.openimaj.math.geometry.shape.Rectangle;
035import org.openimaj.math.statistics.distribution.Histogram;
036
037/**
038 * A {@link SpatialBinningStrategy} that extracts histograms from a number of
039 * equally-sized, non-overlapping within the sample region and concatenates them
040 * together. Each sub-histogram is L2 normalised.
041 * 
042 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
043 * 
044 */
045public class SimpleBlockStrategy implements SpatialBinningStrategy {
046        int numBlocksX;
047        int numBlocksY;
048
049        /**
050         * Construct with the given number of blocks in both the x and y directions.
051         * 
052         * @param numBlocks
053         *            number of blocks in each direction
054         */
055        public SimpleBlockStrategy(int numBlocks) {
056                this(numBlocks, numBlocks);
057        }
058
059        /**
060         * Construct with the given number of blocks in the x and y directions.
061         * 
062         * @param numBlocksX
063         *            number of blocks in the x directions
064         * @param numBlocksY
065         *            number of blocks in the y directions
066         */
067        public SimpleBlockStrategy(int numBlocksX, int numBlocksY) {
068                this.numBlocksX = numBlocksX;
069                this.numBlocksY = numBlocksY;
070        }
071
072        @Override
073        public Histogram extract(WindowedHistogramExtractor binnedData, Rectangle region, Histogram output) {
074                final float dx = region.width / numBlocksX;
075                final float dy = region.height / numBlocksY;
076                final int blockSize = binnedData.getNumBins();
077
078                if (output == null || output.values.length != blockSize * numBlocksX * numBlocksY)
079                        output = new Histogram(blockSize * numBlocksX * numBlocksY);
080
081                final RectangleSampler rs = new RectangleSampler(region, dx, dy, dx, dy);
082                int block = 0;
083                final Histogram tmp = new Histogram(blockSize);
084
085                for (final Rectangle r : rs) {
086                        binnedData.computeHistogram(r, tmp);
087                        tmp.normaliseL2();
088
089                        System.arraycopy(tmp.values, 0, output.values, blockSize * block, blockSize);
090                        block++;
091                }
092
093                return output;
094        }
095}
