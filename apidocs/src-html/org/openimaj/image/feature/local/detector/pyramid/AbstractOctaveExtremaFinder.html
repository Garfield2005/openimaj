030package org.openimaj.image.feature.local.detector.pyramid;
031
032import org.openimaj.image.FImage;
033import org.openimaj.image.analysis.pyramid.gaussian.GaussianOctave;
034
035/**
036 * <p>
037 * Class for finding extrema within {@link Octave}s using the
038 * approach in Section 4 of Lowe's IJCV paper (minus the bit on
039 * using Brown's interpolation approach to improve localisation).
040 * </p>
041 * Interest points are detected if:
042 * <ul>
043 *      <li>They are at a local extremum in scale-space</li>
044 *  <li>The ratio of the Eigenvalues of the edge response at the point is above
045 *      a threshold (i.e. the point is not on a straight line, and has a certain
046 *      amount of curvature).
047 *  </li>
048 * </ul>
049 * 
050 *  <p>
051 *  The AbstractOctaveExtremaFinder uses an event listener paradigm. Once
052 *  interest points are found, the internal listener will be informed. 
053 *  </p>
054 * 
055 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
056 *
057 * @param <OCTAVE>
058 */
059public abstract class AbstractOctaveExtremaFinder< 
060                OCTAVE extends GaussianOctave<FImage>> 
061        extends 
062                AbstractOctaveInterestPointFinder<OCTAVE, FImage> 
063{
064        /** The default threshold for the edge response Eigenvalue ratio */
065        public static final float DEFAULT_EIGENVALUE_RATIO = 10.0f;
066        
067        //Threshold on the ratio of the Eigenvalues of the Hessian matrix (Lowe IJCV, p.12)
068        protected float eigenvalueRatio = DEFAULT_EIGENVALUE_RATIO;
069        
070        /**
071         * Construct an AbstractOctaveExtremaFinder with the default
072         * Eigenvalue ratio threshold.
073         */
074        public AbstractOctaveExtremaFinder() {
075                this(DEFAULT_EIGENVALUE_RATIO);
076        }
077        
078        /**
079         * Construct an AbstractOctaveExtremaFinder with the given
080         * Eigenvalue ratio threshold.
081         * @param eigenvalueRatio
082         */
083        public AbstractOctaveExtremaFinder(float eigenvalueRatio) {
084                this.eigenvalueRatio = eigenvalueRatio;
085        }
086        
087        @Override
088        public OCTAVE getOctave() {
089                return octave;
090        }
091        
092        @Override
093        public int getCurrentScaleIndex() {
094                return currentScaleIndex;
095        }
096        
097        @Override
098        public void process(OCTAVE octave) {
099                beforeProcess(octave);
100                
101                this.octave = octave;
102                
103                FImage[] images = octave.images;
104                int height = images[0].height;
105                int width = images[0].width;
106                int borderDist = octave.options.getBorderPixels();
107
108                //search through the scale-space images, leaving a border 
109                for (currentScaleIndex = 1; currentScaleIndex < images.length - 1; currentScaleIndex++) {
110                        for (int y = borderDist; y < height - borderDist; y++) {
111                                for (int x = borderDist; x < width - borderDist; x++) {
112                                        float val = images[currentScaleIndex].pixels[y][x];
113
114                                        if (firstCheck(val, x, y, currentScaleIndex, images) &&
115                                                isLocalExtremum(val, images[currentScaleIndex-1], x, y) && 
116                                                isLocalExtremum(val, images[currentScaleIndex], x, y) && 
117                                                isLocalExtremum(val, images[currentScaleIndex+1], x, y) && 
118                                                isNotEdge(images[currentScaleIndex], x, y)) {
119                                                processExtrema(images, currentScaleIndex, x, y, octave.octaveSize);
120                                        }
121                                }
122                        }
123                }
124        }
125
126        /**
127         * Perform the first of the checks that determine whether
128         * a point is a valid interest point. This can be overridden 
129         * to allow for cheaper tests to occur before more expensive ones.
130         * 
131         * @param val the value at the point.
132         * @param x the x-coordinate of the point.
133         * @param y the y-coordinate of the point.
134         * @param scaleIndex the scale index at which the point was found
135         * @param images the scale images
136         * @return true if the point is potentially an interest point; false otherwise.
137         */
138        protected boolean firstCheck(float val, int x, int y, int scaleIndex, FImage[] images) {
139                return true;
140        }
141
142        /**
143         * Called at the start of {@link AbstractOctaveExtremaFinder#process(OCTAVE)}
144         * @param octave the octave being processed
145         */
146        protected void beforeProcess(OCTAVE octave) {
147                //do nothing
148        }
149
150        /**
151         * Test to see if a point is a local extremum by searching
152         * the +/- 1 pixel neighbourhood in x and y. 
153         * 
154         * @param val the value at x,y 
155         * @param image the image to test against
156         * @param x the x-coordinate
157         * @param y the y-coordinate
158         * @return true if extremum, false otherwise.
159         */
160        protected boolean isLocalExtremum(float val, FImage image, int x, int y) {
161                float pix[][] = image.pixels;
162
163                if (val > 0.0) {
164                        for (int yy = y - 1; yy <= y + 1; yy++)
165                                for (int xx = x - 1; xx <= x + 1; xx++)
166                                        if (pix[yy][xx] > val)
167                                                return false;
168                } else {
169                        for (int yy = y - 1; yy <= y + 1; yy++)
170                                for (int xx = x - 1; xx <= x + 1; xx++)
171                                        if (pix[yy][xx] < val)
172                                                return false;
173                }
174                return true;
175        }
176
177        
178        /**
179         * Test if the pixel at x,y in the image is NOT on an edge.
180         * @param image the image
181         * @param x the x-coordinate
182         * @param y the y-coordinate
183         * @return true if the pixel is not an edge, false otherwise
184         */
185        protected boolean isNotEdge(FImage image, int x, int y) {
186                float pix[][] = image.pixels;
187
188                //estimate Hessian from finite differences
189                float H00 = pix[y - 1][x] - 2.0f * pix[y][x] + pix[y + 1][x];
190                float H11 = pix[y][x - 1] - 2.0f * pix[y][x] + pix[y][x + 1];
191                float H01 = ((pix[y + 1][x + 1] - pix[y + 1][x - 1]) - (pix[y - 1][x + 1] - pix[y - 1][x - 1])) / 4.0f;
192
193                //determinant and trace of Hessian
194                float det = H00 * H11 - H01 * H01;
195                float trace = H00 + H11;
196
197                float eigenvalueRatio1 = eigenvalueRatio + 1.0f;
198                
199                return (det * eigenvalueRatio1 * eigenvalueRatio1 > eigenvalueRatio * trace * trace);
200        }
201
202        /**
203         * Perform any additional checks on the point, and then inform
204         * the listener that a point has been found.
205         * 
206         * @param images the stack of images in this octave
207         * @param s the interest-point scale
208         * @param x the x-coordinate of the interest-point
209         * @param y the y-coordinate of the interest-point
210         * @param octSize
211         */
212        protected abstract void processExtrema(FImage[] images, int s, int x, int y, float octSize);
213}
