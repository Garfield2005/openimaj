030package org.openimaj.image.feature.local.affine;
031
032import org.openimaj.image.MBFImage;
033import org.openimaj.image.feature.local.engine.DoGColourSIFTEngine;
034import org.openimaj.image.feature.local.engine.DoGSIFTEngine;
035import org.openimaj.image.feature.local.engine.DoGSIFTEngineOptions;
036import org.openimaj.image.feature.local.engine.Engine;
037import org.openimaj.image.feature.local.keypoints.Keypoint;
038
039/**
040 * Colour Affine-simulated SIFT (CASIFT). A {@link DoGColourSIFTEngine} is used
041 * internally to find the features. Specifically, the
042 * {@link DoGColourSIFTEngine} creates a luminance image from which to apply the
043 * difference-of-Gaussian interest point detection algorithm, but extracts the
044 * actual SIFT features from the bands of the input image directly. This means
045 * that the type of Colour-SIFT feature is controlled directly by the
046 * colour-space of the input image; for example if an RGB image is given as
047 * input, then the feature will be standard RGB-SIFT.
048 *
049 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
050 *
051 */
052public class ColourASIFT extends ASIFT<MBFImage, Float[]> {
053        /**
054         * Construct with the given options for the internal {@link DoGSIFTEngine}.
055         *
056         * @param opts
057         */
058        public ColourASIFT(DoGSIFTEngineOptions<MBFImage> opts) {
059                super(opts);
060        }
061
062        /**
063         * Construct the ASIFT extractor using the default parameters for the
064         * {@link DoGSIFTEngine}, with the exception of the option to double the
065         * size of the initial image which can be overridden.
066         *
067         * @see DoGSIFTEngineOptions#setDoubleInitialImage(boolean)
068         *
069         * @param hires
070         *            if true, then the input image is doubled in size before the
071         *            SIFT features are extracted.
072         */
073        public ColourASIFT(boolean hires) {
074                super(hires);
075        }
076
077        @Override
078        public Engine<Keypoint, MBFImage> constructEngine(DoGSIFTEngineOptions<MBFImage> opts) {
079                return new DoGColourSIFTEngine(opts);
080        }
081}
