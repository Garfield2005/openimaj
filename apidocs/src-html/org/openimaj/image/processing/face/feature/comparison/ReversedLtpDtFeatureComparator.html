030package org.openimaj.image.processing.face.feature.comparison;
031
032import java.io.DataInput;
033import java.io.DataOutput;
034import java.io.IOException;
035import java.util.List;
036
037import org.openimaj.citation.annotation.Reference;
038import org.openimaj.citation.annotation.ReferenceType;
039import org.openimaj.image.FImage;
040import org.openimaj.image.pixel.Pixel;
041import org.openimaj.image.processing.face.feature.ltp.LtpDtFeature;
042
043/**
044 * A comparator for Local Trinary Pattern Features using a
045 * Euclidean distance transform. 
046 * <p>
047 * This comparator differs from that described in the original
048 * paper as it computes the distance map only on the query, rather
049 * than on each database target. This increases query time, but
050 * requires less memory.
051 * 
052 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
053 */
054@Reference(
055                type = ReferenceType.Article,
056                author = { "Tan, Xiaoyang", "Triggs, Bill" },
057                title = "Enhanced local texture feature sets for face recognition under difficult lighting conditions",
058                year = "2010",
059                journal = "Trans. Img. Proc.",
060                pages = { "1635", "1650" },
061                url = "http://dx.doi.org/10.1109/TIP.2010.2042645",
062                month = "June",
063                number = "6",
064                publisher = "IEEE Press",
065                volume = "19"
066        )
067public class ReversedLtpDtFeatureComparator implements FacialFeatureComparator<LtpDtFeature> {
068
069        @Override
070        public double compare(LtpDtFeature query, LtpDtFeature target) {
071                List<List<Pixel>> slicePixels = target.ltpPixels;
072                float distance = 0;
073                
074                FImage[] distanceMaps = query.getDistanceMaps();
075                
076                for (int i=0; i<distanceMaps.length; i++) {
077                        List<Pixel> pixels = slicePixels.get(i);
078                        double sliceDistance = 0;
079                        
080                        if (distanceMaps[i] == null || pixels == null)
081                                continue;
082                        
083                        for (Pixel p : pixels) {
084                                sliceDistance += distanceMaps[i].pixels[p.y][p.x];
085                        }
086                        distance += sliceDistance;
087                }
088                
089                return distance;
090        }
091
092        @Override
093        public boolean isDistance() {
094                return true;
095        }
096
097        @Override
098        public void readBinary(DataInput in) throws IOException {
099                //do nothing
100        }
101
102        @Override
103        public byte[] binaryHeader() {
104                //do nothing
105                return null;
106        }
107
108        @Override
109        public void writeBinary(DataOutput out) throws IOException {
110                //do nothing
111        }
112        
113        @Override
114        public String toString() {
115                return "ReversedLtpDtFeatureComparator";
116        }
117}
