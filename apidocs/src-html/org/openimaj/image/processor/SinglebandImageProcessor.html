030package org.openimaj.image.processor;
031
032import org.openimaj.image.Image;
033import org.openimaj.image.SingleBandImage;
034
035/**
036 * An interface for objects that are able to process only
037 * {@link SingleBandImage}s.
038 *
039 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
040 * @param <T>
041 *            The type of pixel in the image that this processor can process.
042 * @param <S>
043 *            The concrete subclass of the single band image that this processor
044 *            can process.
045 */
046public interface SinglebandImageProcessor<T, S extends Image<T, S>>
047extends ImageProcessor<S>
048{
049        /**
050         * An interface for {@link Image}s that are processable by
051         * {@link SinglebandImageProcessor}s.
052         *
053         * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
054         * @param <T>
055         *            The type of pixel in the image
056         * @param <S>
057         *            The concrete subclass of the single band image
058         * @param <I>
059         *            The type of image that is returned after processing
060         */
061        public interface Processable<T, S extends Image<T, S>, I extends Image<?, I>>
062        {
063                /**
064                 * Process with the given {@link SinglebandImageProcessor} returning a
065                 * new image.
066                 *
067                 * @param p
068                 *            The processor to process the image with
069                 * @return A new image containing the result.
070                 */
071                public I process(SinglebandImageProcessor<T, S> p);
072
073                /**
074                 * Process with the given {@link SinglebandImageProcessor} storing the
075                 * result in this processable image. Side-affects this processable
076                 * image.
077                 *
078                 * @param p
079                 *            The processor to process the image with
080                 * @return A new image containing the result.
081                 */
082                public I processInplace(SinglebandImageProcessor<T, S> p);
083        }
084}
