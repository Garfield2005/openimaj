030package org.openimaj.util.function.predicates;
031
032import java.util.ArrayList;
033import java.util.List;
034
035import org.openimaj.util.function.Predicate;
036
037/**
038 * "Or" together 2 or more predicates
039 * 
040 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
041 * 
042 * @param <T>
043 *            the input type of the predicates
044 */
045public class Or<T> implements Predicate<T> {
046        List<Predicate<T>> predicates;
047
048        /**
049         * Construct an empty "or"
050         */
051        public Or() {
052                predicates = new ArrayList<Predicate<T>>();
053        }
054
055        /**
056         * Construct with the given predicates
057         * 
058         * @param p1
059         *            first predicate
060         * @param p2
061         *            second predicate
062         */
063        public Or(Predicate<T> p1, Predicate<T> p2) {
064                this();
065                predicates.add(p1);
066                predicates.add(p2);
067        }
068
069        /**
070         * Construct with the given predicates
071         * 
072         * @param p1
073         *            first predicate
074         * @param p2
075         *            second predicate
076         * @param p3
077         *            third predicate
078         */
079        public Or(Predicate<T> p1, Predicate<T> p2, Predicate<T> p3) {
080                this();
081                predicates.add(p1);
082                predicates.add(p2);
083                predicates.add(p3);
084        }
085
086        /**
087         * Construct with the given predicates
088         * 
089         * @param predicates
090         *            the predicates
091         */
092        public Or(List<Predicate<T>> predicates) {
093                this.predicates = predicates;
094        }
095
096        /**
097         * Add a new predicate to this "or"
098         * 
099         * @param p
100         *            the predicate to add
101         * @return this
102         */
103        public Or<T> add(Predicate<T> p) {
104                predicates.add(p);
105                return this;
106        }
107
108        @Override
109        public boolean test(T object) {
110                for (final Predicate<T> p : predicates)
111                        if (p.test(object))
112                                return true;
113                return false;
114        }
115}
