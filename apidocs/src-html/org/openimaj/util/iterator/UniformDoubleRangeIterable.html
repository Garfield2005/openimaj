038/**
039 * Numeric range iterator for double types. A range goes from 
040 * a number <code>start</code> (inclusive) to another number 
041 * <code>stop</code> (exclusive). There is an optional step
042 * size. 
043 * 
044 * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
045 */
046public class UniformDoubleRangeIterable implements NumericIterable<Double> {
047        private double stop;
048        private double step;
049        private double current;
050        
051        /**
052         * Construct a UniformDoubleRangeIterator with the given values
053         * @param start the starting value (inclusive)
054         * @param stop the stopping number (exclusive)
055         * @param step the step size
056         */
057        public UniformDoubleRangeIterable(double start, double stop, double step) {
058                this.stop = stop;
059                this.step = step;
060                this.current = start;
061        }
062        /**
063         * Construct a UniformDoubleRangeIterator with the given values
064         *
065         * @param start the starting value (inclusive)
066         * @param stop the stopping number (exclusive)
067         */
068        public UniformDoubleRangeIterable(double start, double stop) {
069                this(start, stop, (double)1);
070        }
071        
072        @Override
073        public Iterator<Double> iterator() {
074                return new Iterator<Double>() {
075
076                        @Override
077                        public boolean hasNext() {
078                                return (current + step <= stop);
079                        }
080
081                        @Override
082                        public Double next() {
083                                double ret = current;
084                                current += step;
085                                return ret;
086                        }
087
088                        @Override
089                        public void remove() {
090                                throw new UnsupportedOperationException( "Not supported" );
091                        }
092                };
093        }
