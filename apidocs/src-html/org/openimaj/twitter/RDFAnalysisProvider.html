030package org.openimaj.twitter;
031
032import com.hp.hpl.jena.rdf.model.Model;
033import com.hp.hpl.jena.rdf.model.Resource;
034
035/**
036 * The analysis held in {@link GeneralJSON} can be translated to JSON very
037 * easily The same analysis cannot be easily translated to RDF so this class
038 * must be registered in GeneralJSONRDF's map to do so.
039 * <p>
040 * What this means that if things add analysis to a {@link GeneralJSONRDF}
041 * instance or a {@link USMFStatus} that will eventually feed a
042 * {@link GeneralJSONRDF} instance, they should register with
043 * {@link GeneralJSONRDF#registerRDFAnalysisProvider(String, RDFAnalysisProvider)}
044 * want to
045 * 
046 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
047 * 
048 */
049public interface RDFAnalysisProvider {
050        /**
051         * When given this analysis, fill the model with triples representing its
052         * existence. The IRI of the status the analysis is against is also provided
053         * and must be added to the model with the IRI of the analysis also.
054         * 
055         * @param m
056         *            the model to add triples to
057         * @param analysis
058         *            the status to associate the analysis to
059         * @param analysisSource
060         *            the analysis to transform to triples
061         */
062        public void addAnalysis(Model m, Resource analysis, GeneralJSON analysisSource);
063
064        /**
065         * Prepare yourself. Called once at the beggining of a write of a batch.
066         */
067        public void init();
068}
