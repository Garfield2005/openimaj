033package org.openimaj.docs.tutorial.fund.audio;
034
035import java.net.MalformedURLException;
036import java.net.URL;
037
038import org.openimaj.audio.analysis.FourierTransform;
039import org.openimaj.video.xuggle.XuggleAudio;
040import org.openimaj.vis.general.BarVisualisation;
041
042/**
043 * Example that demonstrates the processing of audio to extract the FFT
044 * magnitudes and then visualise it.
045 * 
046 * @author David Dupplaw (dpd@ecs.soton.ac.uk)
047 * @created 18 Jun 2013
048 * @version $Author$, $Revision$, $Date$
049 */
050public class FFT
051{
052        /**
053         * Main method
054         * 
055         * @param args
056         *            Command-line args (unused)
057         * @throws MalformedURLException
058         */
059        public static void main(final String[] args) throws MalformedURLException
060        {
061                // Open a URL to the sine wave sweep. If you have downloaded
062                // this file you should use a new File(<filename>) here.
063                final XuggleAudio xa = new XuggleAudio(
064                                new URL("http://www.audiocheck.net/download.php?" +
065                                                "filename=Audio/audiocheck.net_sweep20-20klin.wav"));
066
067                // Create the Fourier transform processor chained to the audio decoder
068                final FourierTransform fft = new FourierTransform(xa);
069
070                // Create a visualisation to show our FFT and open the window now
071                final BarVisualisation bv = new BarVisualisation(400, 200);
072                bv.showWindow("FFTs");
073                bv.getAxesRenderer().setDrawYTicks(false);
074
075                // Loop through the sample chunks from the audio capture thread
076                // sending each one through the feature extractor and displaying
077                // the results in the visualisation.
078                while (fft.nextSampleChunk() != null)
079                {
080                        final float[][] ffts = fft.getMagnitudes();
081                        bv.setData(ffts[0]);
082                }
083        }
084}
