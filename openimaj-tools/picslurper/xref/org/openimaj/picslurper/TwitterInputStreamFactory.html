23  /**
24   * Factory provides input streams to twitter, managing the disconnection of old
25   * streams
26   *
27   * @author Sina Samangooei (ss@ecs.soton.ac.uk)

31  public class TwitterInputStreamFactory {
32  	private static final Logger logger = Logger.getLogger(TwitterInputStreamFactory.class);
33  	private static final String TWITTER_SAMPLE_URL = "https://stream.twitter.com/1/statuses/sample.json";
34  	private static TwitterInputStreamFactory staticFactory;
35  	private InputStream inputStream;
36  	private PoolingClientConnectionManager manager;
37  	private BasicHttpParams params;
38  	private SchemeRegistry registry;
39  
40  	/**
41  	 * @return a new source of twitter input streams
42  	 */
43  	public static TwitterInputStreamFactory streamFactory() {
44  		if (staticFactory == null) {
45  			try {
46  				staticFactory = new TwitterInputStreamFactory();
47  			} catch (Exception e) {
48  				e.printStackTrace();
49  				System.err.println("Error creating TwitterInputStreamFactory: " + e.getMessage());
50  				return null;
51  			}
52  		}
53  		return staticFactory;
54  	}
55  
56  	private TwitterInputStreamFactory() throws FileNotFoundException, IOException {
57  		PicSlurper.loadConfig();
58  		this.params = new BasicHttpParams();
59  		HttpProtocolParams.setVersion(params, HttpVersion.HTTP_1_1);
60  		HttpProtocolParams.setContentCharset(params, "utf-8");
61  		HttpConnectionParams.setConnectionTimeout(params, 1000);
62  		HttpConnectionParams.setSoTimeout(params, 1000);
63  		this.registry = new SchemeRegistry();
64  		registry.register(new Scheme("https", 443, SSLSocketFactory.getSocketFactory()));
65  	}
66  
67  	private void reconnect() throws ClientProtocolException, IOException {
68  		logger.debug("Reconnecting, closing expired clients");
69  
70  		logger.debug("Constructing new client");
71  		DefaultHttpClient client = new DefaultHttpClient(manager, params);
72  
73  		UsernamePasswordCredentials twitterUserPassword = new UsernamePasswordCredentials(
74  				System.getProperty("twitter.user"),
75  				System.getProperty("twitter.password")
76  				);
77  		client.getCredentialsProvider().setCredentials(new AuthScope("stream.twitter.com", 443), twitterUserPassword);
78  		HttpGet get = new HttpGet(TWITTER_SAMPLE_URL);
79  		HttpResponse resp = client.execute(get);
80  		logger.debug("Done!");
81  		this.inputStream = resp.getEntity().getContent();
82  	}
83  
84  	/**
85  	 * @return Attempt to disconnect any previous streams and reconnect
86  	 * @throws IOException
87  	 */
88  	public InputStream nextInputStream() throws IOException {
89  		logger.debug("Attempting to get next input twitter input stream!");
90  		if (this.inputStream != null) {
91  			logger.debug("Closing old stream...");
92  			manager.shutdown();
93  			this.inputStream.close();
94  		}
95  		this.manager = new PoolingClientConnectionManager(registry);
96  		reconnect();
97  		return this.inputStream;
98  	}
