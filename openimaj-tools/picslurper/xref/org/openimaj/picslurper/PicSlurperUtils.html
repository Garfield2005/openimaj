33  	/**
34  	 * Update a specific file with statistics of URLs being consumed
35  	 *
36  	 * @param statsFile
37  	 * @param statusConsumption
38  	 * @param forgetImageURLs whether the actual image urls collected should be saved. On the global stats file this should be true
39  	 * @throws IOException
40  	 */
41  	public static synchronized void updateStats(File statsFile, StatusConsumption statusConsumption, boolean forgetImageURLs) throws IOException {
42  		StatusConsumption current = new StatusConsumption();
43  		if (statsFile.exists())
44  			current = IOUtils.read(statsFile, current);
45  		current.incr(statusConsumption);
46  		if(forgetImageURLs){
47  			// emtpy the imageURLs before you save
48  			current.imageURLs.clear();
49  		}
50  		IOUtils.writeASCII(statsFile, current); // initialise the output file
51  	}
52  
53  	private static transient Gson gson = new Gson();
54  
55  	/**
56  	 * Updated a tweets.json file in the specified location with the given
57  	 * {@link ReadableWritableJSON} instance
58  	 *
59  	 * @param outRoot
60  	 * @param status
61  	 * @throws IOException
62  	 */
63  	public static synchronized void updateTweets(File outRoot, Status status) throws IOException {
64  		if(status==null)return;
65  		File outFile = new File(outRoot, TWEET_FILE_NAME);
66  		FileOutputStream fstream = new FileOutputStream(outFile, true);
67  		PrintWriter pwriter = new PrintWriter(new BufferedWriter(new OutputStreamWriter(fstream, "UTF-8")));
68  		pwriter.println(gson.toJson(status));
69  		pwriter.println();
70  		pwriter.flush();
71  		pwriter.close();
72  	}
