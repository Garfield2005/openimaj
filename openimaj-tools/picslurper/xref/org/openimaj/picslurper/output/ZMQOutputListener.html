12  /**
13   * Using ZeroMQ (because it is awesome and I wanted to learn it so fuck off) we set up a
14   * subscription based server which tells anyone listening about any new images.
15   *
16   * The name of the queue can be specified
17   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
18   *
19   */
20  public class ZMQOutputListener implements OutputListener {
21  
22  	private static final Logger logger = Logger.getLogger(ZMQOutputListener.class);
23  
24  	private Socket publisher;
25  
26  	/**
27  	 * Construct the publishing connector
28  	 */
29  	public ZMQOutputListener() {
30  
31  	}
32  
33  	@Override
34  	public void newImageDownloaded(WriteableImageOutput written) {
35  		try {
36  			StringWriter writer = new StringWriter();
37  			IOUtils.writeASCII(writer, written);
38  			publisher.send("IMAGE".getBytes("UTF-8"), ZMQ.SNDMORE);
39  			boolean sent = publisher.send(writer.toString().getBytes("UTF-8"), 0);
40  			if(!sent){
41  				throw new IOException("Send failed");
42  			}
43  		} catch (IOException e) {
44  			logger.error("Unable to send written image: " + written.url);
45  			logger.error(e.getMessage());
46  		}
47  	}
48  
49  	@Override
50  	public void failedURL(URL url, String reason) {
51  		try {
52  			StringWriter writer = new StringWriter();
53  			if(url==null)return;
54  			IOUtils.writeASCII(writer, new WriteableFailedURL(url, reason));
55  			publisher.send("FAIL".getBytes("UTF-8"), ZMQ.SNDMORE);
56  			boolean sent = publisher.send(writer.toString().getBytes("UTF-8"), 0);
57  			if(!sent){
58  				throw new IOException("Send failed");
59  			}
60  		} catch (IOException e) {
61  			logger.error("Unable to send failure!");
62  		}
63  	}
64  
65  	@Override
66  	public void finished() {
67  		publisher.close();
68  	}
69  
70  	@Override
71  	public void prepare() {
72  		ZMQ.Context context = ZMQ.context(1);
73  		publisher = context.socket(ZMQ.PUB);
74  
75  		publisher.bind("tcp://*:5563");
76  	}
