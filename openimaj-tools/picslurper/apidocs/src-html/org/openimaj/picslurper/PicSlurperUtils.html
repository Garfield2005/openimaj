001package org.openimaj.picslurper;
002
003import java.io.BufferedWriter;
004import java.io.File;
005import java.io.FileOutputStream;
006import java.io.IOException;
007import java.io.OutputStreamWriter;
008import java.io.PrintWriter;
009
010import org.openimaj.io.IOUtils;
011import twitter4j.Status;
012
013import com.google.gson.Gson;
014
015/**
016 * Functions for writing various parts of PicSlurper
017 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
018 *
019 */
020public class PicSlurperUtils {
021        private static String TWEET_FILE_NAME = "tweets.json";
022        /**
023         * Update a specific file with statistics of URLs being consumed
024         *
025         * @param statsFile
026         * @param statusConsumption
027         * @throws IOException
028         */
029        public static synchronized void updateStats(File statsFile, StatusConsumption statusConsumption) throws IOException {
030                updateStats(statsFile, statusConsumption, false);
031        }
032
033        /**
034         * Update a specific file with statistics of URLs being consumed
035         *
036         * @param statsFile
037         * @param statusConsumption
038         * @param forgetImageURLs whether the actual image urls collected should be saved. On the global stats file this should be true
039         * @throws IOException
040         */
041        public static synchronized void updateStats(File statsFile, StatusConsumption statusConsumption, boolean forgetImageURLs) throws IOException {
042                StatusConsumption current = new StatusConsumption();
043                if (statsFile.exists())
044                        current = IOUtils.read(statsFile, current);
045                current.incr(statusConsumption);
046                if(forgetImageURLs){
047                        // emtpy the imageURLs before you save
048                        current.imageURLs.clear();
049                }
050                IOUtils.writeASCII(statsFile, current); // initialise the output file
051        }
052
053        private static transient Gson gson = new Gson();
054
055        /**
056         * Updated a tweets.json file in the specified location with the given
057         * {@link ReadableWritableJSON} instance
058         *
059         * @param outRoot
060         * @param status
061         * @throws IOException
062         */
063        public static synchronized void updateTweets(File outRoot, Status status) throws IOException {
064                if(status==null)return;
065                File outFile = new File(outRoot, TWEET_FILE_NAME);
066                FileOutputStream fstream = new FileOutputStream(outFile, true);
067                PrintWriter pwriter = new PrintWriter(new BufferedWriter(new OutputStreamWriter(fstream, "UTF-8")));
068                pwriter.println(gson.toJson(status));
069                pwriter.println();
070                pwriter.flush();
071                pwriter.close();
072        }
073}
