51  /**
52   * A tool that computes features from the foreground
53   * object in an image. The flood-fill algorithm is
54   * used to segment the foreground/background based 
55   * on a seed pixel and distance threshold.
56   * 
57   * @author Jonathon Hare (jsh2@ecs.soton.ac.uk)
58   */
59  public class SegmentingGlobalFeaturesTool {
60  	@Option(name="--feature-type", aliases="-f", handler=ProxyOptionHandler.class, usage="Feature type", required=false)
61  	private ShapeFeatures feature;
62  	private ShapeFeaturesOp featureOp;
63  	
64  	@Option(name = "--input", aliases="-i", required=true, usage="Set the input image", handler=MBFImageOptionHandler.class)
65  	private MBFImage input;
66  	
67  	@Option(name = "--output", aliases="-o", required=false, usage="Set the output file; if not set then output is to stdout", handler=FileOutputStreamOptionHandler.class)
68  	private OutputStream output = System.out;
69  	
70  	@Option(name = "--binary", aliases="-b", required=false, usage="Set output mode to binary")
71  	private boolean binary = false;
72  	
73  	@Option(name = "--mask-output", aliases="-mo", required=false, usage="Save the generated segmentation mask")
74  	private File maskoutput;
75  	
76  	@Option(name = "--px", aliases="-px", required=false, usage="x-position of the starting pixel")
77  	private int px = 0;
78  	
79  	@Option(name = "--py", aliases="-py", required=false, usage="y-position of the starting pixel")
80  	private int py = 0;
81  	
82  	@Option(name = "--thresh", aliases="-thresh", required=false, usage="threshold for flood-fill algorithm")
83  	private float thresh = 25F/255F;
84  	
85  	void execute() throws IOException {
86  		FImage mask = FloodFill.floodFill(input, px, py, thresh);
87  		
88  		if (maskoutput != null)
89  			ImageUtilities.write(mask, maskoutput.getName().substring(maskoutput.getName().lastIndexOf(".") + 1), maskoutput);
90  		
91  		if (feature != null) { 
92  			FeatureVector fv = featureOp.execute(input, mask);
93  		
94  			if (binary)
95  				IOUtils.writeBinary(output, fv);
96  			else
97  				IOUtils.writeASCII(output, fv);
98  		}
99  	}
100 	
101 	/**
102 	 * The main method for the tool.
103 	 * @param args
104 	 * @throws IOException
105 	 */
106 	public static void main(String [] args) throws IOException {
107 		SegmentingGlobalFeaturesTool tool = new SegmentingGlobalFeaturesTool();
108 		
109 		CmdLineParser parser = new CmdLineParser(tool);
110 
111 		try {
112 			parser.parseArgument(args);
113 		} catch (CmdLineException e) {
114 			System.err.println(e.getMessage());
115 			System.err.println("Usage: java -cp GlobalFeaturesTool.jar uk.ac.soton.ecs.jsh2.SegmentingGlobalFeaturesTool [options...]");
116 			parser.printUsage(System.err);
117 			
118 			if (tool.feature == null) {
119 				for (GlobalFeatureType m : GlobalFeatureType.values()) {
120 					System.err.println();
121 					System.err.println(m + " options: ");
122 					new CmdLineParser(m.getOptions()).printUsage(System.err);
123 				}
124 			}
125 			return;
126 		}
127 		
128 		tool.execute();
129 	}
