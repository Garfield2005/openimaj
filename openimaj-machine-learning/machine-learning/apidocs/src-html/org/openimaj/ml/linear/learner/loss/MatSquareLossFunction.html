030package org.openimaj.ml.linear.learner.loss;
031
032import org.apache.log4j.Logger;
033import org.openimaj.math.matrix.CFMatrixUtils;
034
035import gov.sandia.cognition.math.matrix.Matrix;
036import gov.sandia.cognition.math.matrix.Vector;
037import gov.sandia.cognition.math.matrix.mtj.SparseMatrixFactoryMTJ;
038
039public class MatSquareLossFunction extends LossFunction{
040        Logger logger = Logger.getLogger(MatSquareLossFunction.class);
041        private SparseMatrixFactoryMTJ spf;
042        public MatSquareLossFunction() {
043                spf = SparseMatrixFactoryMTJ.INSTANCE;
044        }
045        @Override
046        public Matrix gradient(Matrix W) {
047                Matrix ret = W.clone();
048                if(CFMatrixUtils.containsInfinity(X)){
049                        throw new RuntimeException();
050                }
051                if(CFMatrixUtils.containsInfinity(W)){
052                        throw new RuntimeException();
053                }
054                Matrix resid = CFMatrixUtils.fastdot(X,W);
055                if(CFMatrixUtils.containsInfinity(resid)){
056                        CFMatrixUtils.fastdot(X,W);
057                        throw new RuntimeException();
058                }
059                if(this.bias!=null)
060                {
061                        resid.plusEquals(this.bias);
062                }
063                CFMatrixUtils.fastminusEquals(resid, Y);
064                if(CFMatrixUtils.containsInfinity(resid)){
065                        throw new RuntimeException();
066                }
067                for (int t = 0; t < resid.getNumColumns(); t++) {
068                        Vector xcol = this.X.getRow(t).scale(resid.getElement(t, t)).clone();
069                        CFMatrixUtils.fastsetcol(ret,t, xcol);
070                }
071                return ret;
072        }
073        @Override
074        public double eval(Matrix W) {
075                Matrix resid = null;
076                if(W == null){
077                        resid = X.clone();
078                } else {
079                        resid = CFMatrixUtils.fastdot(X,W);
080                }
081                Matrix vnobias = resid.clone();
082                if(this.bias!=null)
083                {
084                        resid.plusEquals(this.bias);
085                }
086                Matrix v = resid.clone();
087                resid.minusEquals(Y);
088                double retval = 0;
089                
090                for (int t = 0; t < resid.getNumColumns(); t++) {
091                        double loss = resid.getElement(t, t);
092                        retval += loss * loss;
093                        logger.debug(
094                                        String.format(
095                                                        "yr=%d,y=%3.2f,v=%3.2f,v(no bias)=%2.5f,error=%2.5f,serror=%2.5f",
096                                                        t, Y.getElement(t, t), v.getElement(t, t), vnobias.getElement(t,t), loss, loss*loss
097                                                        )
098                                        );
099                }
100                return retval;
101        }
102        
103        @Override
104        public boolean test_backtrack(Matrix W, Matrix grad, Matrix prox, double eta) {
105                Matrix tmp = prox.minus(W);
106        double evalW = eval(W);
107                double evalProx = eval(prox);
108                Matrix fastdotGradTmp = CFMatrixUtils.fastdot(grad.transpose(),tmp);
109                double normGradProx = CFMatrixUtils.sum(fastdotGradTmp);
110                double normTmp = 0.5*eta*tmp.normFrobenius();
111                return (evalProx <= evalW + normGradProx + normTmp);
112        }
113        
114        @Override
115        public boolean isMatrixLoss() {
116                return true;
117        }
118}
