042/**
043 * An implementation of a general linear regressive such that the values of a
044 * timeseries Y are predicted using the values of a set of time series X at some
045 * offset over some time window. X may potentially contain Y itself which turns
046 * this into an auto-regressive model augmented with extra information.
047 * Furthermore, varying window sizes and offsets may be used for each time
048 * series X.
049 *
050 * This is all achieved with {@link SynchronisedTimeSeriesCollection} which
051 * model a set of timeseries which are synchronised.
052 *
053 * When intitalised, the Y time series must be explicitly specified. By default
054 * the
055 * 
056 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
057 *
058 */
059public class WindowedLinearRegressionAggregator implements SynchronisedTimeSeriesCollectionAggregator<
060                DoubleTimeSeries,
061                DoubleSynchronisedTimeSeriesCollection,
062                DoubleTimeSeries>
063{
064
065        private static final int DEFAULT_WINDOW_SIZE = 3;
066        private static final int DEFAULT_OFFSET = 1;

068        private boolean autoregressive = true;

070        private String ydataName;

079         * @param ydataName

082                this();
083                windowOffsets.add(IndependentPair.pair(DEFAULT_WINDOW_SIZE, DEFAULT_OFFSET));
084                this.ydataName = ydataName;
085        }

167        /**
168         * Perform regression s.t. y = Sum(w_{0-i} * x_{0-i}) + c for i from 1 to
169         * windowsize with some offset. The same windowsize and offset is used for
170         * each time series
171         * 
172         * @param ydataName
173         * @param windowsize
174         * @param offset
175         * @param autoregressive
176         *            whether the ydata should be used in regression
177         */
178        public WindowedLinearRegressionAggregator(String ydataName, int windowsize, int offset, boolean autoregressive,
179                        DoubleSynchronisedTimeSeriesCollection other)
180        {
181                this();
182                final WindowedLinearRegressionAggregator regress = new WindowedLinearRegressionAggregator(ydataName, windowsize,
183                                offset, autoregressive);
184                regress.aggregate(other);
185                this.reg = regress.reg;
186                this.ydataName = regress.ydataName;
187                this.autoregressive = regress.autoregressive;
188                this.windowOffsets = regress.windowOffsets;
189        }

191        /**
192         * Perform regression s.t. y = Sum(w_{0-i} * x_{0-i}) + c for i from 1 to
193         * windowsize with some offset. The same windowsize and offset is used for
194         * each time series
195         * 
196         * @param ydataName
197         * @param autoregressive
198         *            whether the ydata should be used in regression
199         * @param windowOffsets
200         */
201        @SafeVarargs
202        public WindowedLinearRegressionAggregator(String ydataName, boolean autoregressive,
203                        IndependentPair<Integer, Integer>... windowOffsets)
204        {
205                this();
206                this.ydataName = ydataName;
207                this.autoregressive = autoregressive;
208                for (final IndependentPair<Integer, Integer> independentPair : windowOffsets) {
209                        this.windowOffsets.add(independentPair);
210                }
211        }

246        public LinearRegression getRegression() {
247                return this.reg;
248        }

250        @Override
251        public DoubleTimeSeries aggregate(DoubleSynchronisedTimeSeriesCollection series) {
252
253                final Set<String> names = series.getNames();
254                if (!autoregressive) {
255                        names.remove(ydataName);
256                }
257                final DoubleTimeSeries yseries = series.series(ydataName);
258                final double[] ydata = yseries.getData();
259
260                series = series.collectionByNames(names);
261                final double[] data = series.flatten();
262                if (this.windowOffsets.size() != series.nSeries() && this.windowOffsets.size() == 1) {
263                        final IndependentPair<Integer, Integer> offset = this.windowOffsets.get(0);
264                        return aggregteSingle(yseries.getTimes(), ydata, data, offset.firstObject(), offset.secondObject(),
265                                        series.nSeries());
266                }
267
268                return null;
269        }
270
271        private DoubleTimeSeries aggregteSingle(long[] times, double[] ydata, double[] data, int windowsize, int offset,
272                        int nseries)
273        {
274                final List<IndependentPair<double[], double[]>> instances = new ArrayList<IndependentPair<double[], double[]>>();
275                for (int i = windowsize + (offset - 1); i < ydata.length; i++) {
276                        final int start = (i - windowsize - (offset - 1)) * nseries;
277                        // System.out.format("Range %d->%d (inclusive) used to calculate: %d\n",start,start+this.windowsize-1,i);
278                        final double[] datawindow = new double[windowsize * nseries];
279                        System.arraycopy(data, start, datawindow, 0, windowsize * nseries);
280                        instances.add(IndependentPair.pair(datawindow, new double[] { ydata[i] }));
281                }
282                if (this.reg == null) {
283                        this.reg = new LinearRegression();
284                        this.reg.estimate(instances);
285                }
286
287                final DoubleTimeSeries ret = new DoubleTimeSeries(times, new double[ydata.length]);
288
289                final Iterator<IndependentPair<double[], double[]>> instanceIter = instances.iterator();
290                data = ret.getData();
291                for (int i = windowsize + (offset - 1); i < ydata.length; i++) {
292                        final double[] predicted = this.reg.predict(instanceIter.next().firstObject());
293                        data[i] = predicted[0];
294                }
295                return ret.get(times[windowsize + (offset - 1)], times[ydata.length - 1]);
296        }
297
298        public LinearRegression getReg() {
299                return this.reg;
300        }
