53  /**
54   * An implementation of a stochastic gradient decent with proximal perameter adjustment
55   * (for regularised parameters).
56   *
57   * Data is dealt with sequentially using a one pass implementation of the
58   * online proximal algorithm described in chapter 9 and 10 of:
59   * The Geometry of Constrained Structured Prediction: Applications to Inference and
60   * Learning of Natural Language Syntax, PhD, Andre T. Martins
61   *
62   * The implementation does the following:
63   * 	- When an X,Y is recieved:
64   * 		- Update currently held batch
65   * 		- If the batch is full:
66   * 			- While There is a great deal of change in U and W:
67   * 				- Calculate the gradient of W holding U fixed
68   * 				- Proximal update of W
69   * 				- Calculate the gradient of U holding W fixed
70   * 				- Proximal update of U
71   * 				- Calculate the gradient of Bias holding U and W fixed
72   * 			- flush the batch
73   * 		- return current U and W (same as last time is batch isn't filled yet)
74   *
75   *
76   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
77   *
78   */

139 	private void initParams(Matrix x, Matrix y, int xrows, int xcols, int ycols) {
140 		final InitStrategy wstrat = getInitStrat(BilinearLearnerParameters.WINITSTRAT,x,y);
141 		final InitStrategy ustrat = getInitStrat(BilinearLearnerParameters.UINITSTRAT,x,y);
142 		this.w = wstrat.init(xrows, ycols);
143 		this.u = ustrat.init(xcols, ycols);
144 		if(this.forceSparcity)
145 		{
146 			this.u = CFMatrixUtils.asSparseColumn(this.u);
147 			this.w = CFMatrixUtils.asSparseColumn(this.w);
148 		}
149 
150 		this.bias = smf.createMatrix(ycols,ycols);
151 		if(this.biasMode){
152 			final InitStrategy bstrat = getInitStrat(BilinearLearnerParameters.BIASINITSTRAT,x,y);
153 			this.bias = bstrat.init(ycols, ycols);
154 			this.diagX = smf.createIdentity(ycols, ycols);
155 		}
156 	}

161 		final InitStrategy strat = this.params.getTyped(initstrat);
162 		if(strat instanceof ContextAwareInitStrategy){
163 			final ContextAwareInitStrategy<Matrix, Matrix> cwStrat = this.params.getTyped(initstrat);
164 			cwStrat.setLearner(this);
165 			cwStrat.setContext(x, y);

170 	@Override
171 	public void process(Matrix X, Matrix Y){
172 		prepareNextRound(X, Y);
173 		int iter = 0;
174 		Matrix xt = X.transpose();
175 		Matrix xtrows = xt;
176 		if(xt instanceof AbstractSparseMatrix){
177 			xtrows = CFMatrixUtils.asSparseRow(xt);
178 		}
179 		while(true) {
180 			// We need to set the bias here because it is used in the loss calculation of U and W
181 			if(this.biasMode) loss.setBias(this.bias);
182 			iter += 1;
183 
184 			// Perform the bilinear operation
185 			final Matrix neww = updateW(xt,eta0_w, lambda_u);
186 			final Matrix newu = updateU(xtrows,neww,eta0_u, lambda_w);
187 			Matrix newbias = null;
188 			if(this.biasMode){
189 				newbias = updateBias(xt, newu, neww, biasEta0);
190 			}
191 			
192 			// This part of the code checks if we can stop the bilinear steps by checking how much everything has changed proportionally
193 			
194 			double ratioB = 0;
195 			double totalbias = 0;
196 			
197 			final double sumchangew = CFMatrixUtils.absSum(neww.minus(this.w));
198 			final double totalw = CFMatrixUtils.absSum(this.w);
199 
200 			final double sumchangeu = CFMatrixUtils.absSum(newu.minus(this.u));
201 			final double totalu = CFMatrixUtils.absSum(this.u);
202 
203 			double ratioU = 0;
204 			if(totalu!=0) ratioU = sumchangeu/totalu;
205 			final double ratioW = 0;
206 			if(totalw!=0) ratioU = sumchangew/totalw;
207 			double ratio = ratioU + ratioW;
208 			if(this.biasMode){
209 				final double sumchangebias = CFMatrixUtils.absSum(newbias.minus(this.bias));
210 				totalbias = CFMatrixUtils.absSum(this.bias);
211 				if(totalbias!=0) ratioB = (sumchangebias/totalbias) ;
212 				ratio += ratioB;
213 				ratio/=3;
214 			} else {
215 				ratio/=2;
216 			}
217 			
218 			/**
219 			 * This is not a matter of simply type
220 			 * The 0 values of the sparse matrix are also removed. very important.
221 			 */
222 			if(this.forceSparcity)
223 			{
224 				this.u = CFMatrixUtils.asSparseColumn(newu);
225 				this.w = CFMatrixUtils.asSparseColumn(neww);
226 			}
227 			else{
228 
229 				this.w = neww;
230 				this.u = newu;
231 			}
232 			
233 			if(this.biasMode){
234 				this.bias = newbias;
235 			}
236 
237 			final Double biconvextol = this.params.getTyped("biconvex_tol");
238 			final Integer maxiter = this.params.getTyped("biconvex_maxiter");
239 			if(iter%3 == 0){
240 				logger.debug(String.format("Iter: %d. Last Ratio: %2.3f",iter,ratio));
241 				logger.debug("W row sparcity: " + CFMatrixUtils.rowSparsity(w));
242 				logger.debug("U row sparcity: " + CFMatrixUtils.rowSparsity(u));
243 				logger.debug("Total U magnitude: " + totalu);
244 				logger.debug("Total W magnitude: " + totalw);
245 				logger.debug("Total Bias: " + totalbias);
246 			}
247 			if(biconvextol  < 0 || ratio < biconvextol || iter >= maxiter) {
248 				logger.debug("tolerance reached after iteration: " + iter);
249 				logger.debug("W row sparcity: " + CFMatrixUtils.rowSparsity(w));
250 				logger.debug("U row sparcity: " + CFMatrixUtils.rowSparsity(u));
251 				logger.debug("Total U magnitude: " + totalu);
252 				logger.debug("Total W magnitude: " + totalw);
253 				logger.debug("Total Bias: " + totalbias);
254 				break;
255 			}
256 		}
257 	}
258 	private void prepareNextRound(Matrix X, Matrix Y) {
259 		final int nfeatures = X.getNumRows();
260 		final int nusers = X.getNumColumns();
261 		final int ntasks = Y.getNumColumns();
262 //		int ninstances = Y.getNumRows(); // Assume 1 instance!
263 		
264 		// only inits when the current params is null
265 		if (this.w == null){
266 			initParams(X,Y,nfeatures, nusers, ntasks); // Number of words, users and tasks
267 		}
268 
269 		final Double dampening = this.params.getTyped(BilinearLearnerParameters.DAMPENING);
270 		final double weighting = 1.0 - dampening ;
271 
272 		logger.debug("... dampening w, u and bias by: " + weighting);
273 
274 		// Adjust for weighting
275 		this.w.scaleEquals(weighting);
276 		this.u.scaleEquals(weighting);
277 		if(this.biasMode){
278 			this.bias.scaleEquals(weighting);
279 		}
280 		// First expand Y s.t. blocks of rows contain the task values for each row of Y.
281 		// This means Yexp has (n * t x t)
282 		final SparseMatrix Yexp = expandY(Y);
283 		loss.setY(Yexp);
284 	}

286 	protected Matrix updateBias(Matrix xt, Matrix nu, Matrix nw, double biasLossWeight) {
287 		Matrix newut = nu.transpose();
288 		Matrix utxt = CFMatrixUtils.fastdot(newut,xt);
289 		Matrix utxtw = CFMatrixUtils.fastdot(utxt,nw);
290 		final Matrix mult = utxtw.plus(this.bias);
291 		// We must set bias to null!
292 		loss.setBias(null);
293 		loss.setX(diagX);
294 		// Calculate gradient of bias (don't regularise)
295 		final Matrix biasGrad = loss.gradient(mult);
296 		Matrix newbias = null;
297 		for (int i = 0; i < 1000; i++) {
298 			logger.debug("... Line searching etab = " + biasLossWeight);
299 			newbias = this.bias.clone();
300 			Matrix scaledGradW = biasGrad.scale(1./biasLossWeight);
301 			newbias = CFMatrixUtils.fastminus(newbias,scaledGradW);
302 			
303 			if(loss.test_backtrack(this.bias, biasGrad, newbias, biasLossWeight)) 
304 				break;
305 			biasLossWeight *= eta_gamma;
306 		}
307 //		final Matrix newbias = this.bias.minus(
308 //				CFMatrixUtils.timesInplace(
309 //						biasGrad,
310 //						biasLossWeight
311 //				)
312 //		);
313 		return newbias;
314 	}
315 	protected Matrix updateW(Matrix xt, double wLossWeighted, double weightedLambda) {
316 		// Dprime is tasks x nwords
317 		
318 		Matrix Dprime = null;
319 		Matrix ut = this.u.transpose();				
320 		if(this.nodataseen){
321 			this.nodataseen = false;
322 			Matrix fakeu = new SparseSingleValueInitStrat(1).init(this.u.getNumColumns(), this.u.getNumRows());
323 			Dprime = CFMatrixUtils.fastdot(fakeu,xt);
324 		} else {
325 			Dprime = CFMatrixUtils.fastdot(ut, xt);
326 		}
327 		
328 		// ... as is the cost function's X
329 		if(zStandardise){
330 			Vector rowMean = CFMatrixUtils.rowMean(Dprime);
331 			CFMatrixUtils.minusEqualsCol(Dprime,rowMean);
332 		}
333 		loss.setX(Dprime);
334 		final Matrix gradW = loss.gradient(this.w);
335 		logger.debug("Abs w_grad: " + CFMatrixUtils.absSum(gradW));
336 		Matrix neww = null;
337 		for (int i = 0; i < 1000; i++) {
338 			logger.debug("... Line searching etaw = " + wLossWeighted);
339 			neww = this.w.clone();
340 			Matrix scaledGradW = gradW.scale(1./wLossWeighted);
341 			neww = CFMatrixUtils.fastminus(neww,scaledGradW);
342 			neww = regul.prox(neww, weightedLambda/wLossWeighted);
343 			if(loss.test_backtrack(this.w, gradW, neww, wLossWeighted)) 
344 				break;
345 			wLossWeighted *= eta_gamma;
346 		}
347 		
348 		return neww;
349 	}
350 	protected Matrix updateU(Matrix xtrows, Matrix neww, double uLossWeight, double uWeightedLambda) {
351 		// Vprime is nusers x tasks
352 		final Matrix Vprime = CFMatrixUtils.fastdot(xtrows,neww);
353 		// ... so the loss function's X is (tasks x nusers)
354 		Matrix Vt = CFMatrixUtils.asSparseRow(Vprime.transpose());
355 		if(zStandardise){
356 			Vector rowMean = CFMatrixUtils.rowMean(Vt);
357 			CFMatrixUtils.minusEqualsCol(Vt,rowMean);
358 		}
359 		loss.setX(Vt);
360 		final Matrix gradU = loss.gradient(this.u);
361 		logger.debug("Abs u_grad: " + CFMatrixUtils.absSum(gradU));
362 //		CFMatrixUtils.timesInplace(gradU,uLossWeight);
363 //		newu = regul.prox(newu, uWeightedLambda);
364 		Matrix newu = null;
365 		for (int i = 0; i < 1000; i++) {
366 			logger.debug("... Line searching etau = " + uLossWeight);
367 			newu = this.u.clone();
368 			Matrix scaledGradW = gradU.scale(1./uLossWeight);
369 			newu = CFMatrixUtils.fastminus(newu,scaledGradW);
370 			newu = regul.prox(newu, uWeightedLambda/uLossWeight);
371 			if(loss.test_backtrack(this.u, gradU, newu, uLossWeight)) 
372 				break;
373 			uLossWeight *= eta_gamma;
374 		}
375 		
376 		return newu;
377 	}

386 	public static SparseMatrix expandY(Matrix Y) {
387 		final int ntasks = Y.getNumColumns();
388 		final SparseMatrix Yexp = SparseMatrixFactoryMTJ.INSTANCE.createMatrix(ntasks, ntasks);
389 		for (int touter = 0; touter < ntasks; touter++) {
390 			for (int tinner = 0; tinner < ntasks; tinner++) {
391 				if(tinner == touter){
392 					Yexp.setElement(touter, tinner, Y.getElement(0, tinner));
393 				}
394 				else{
395 					Yexp.setElement(touter, tinner, Double.NaN);
396 				}
397 			}
398 		}
399 		return Yexp;
400 	}

496 		final int nusers = in.readInt();
497 		final int ntasks = in.readInt();
498 
499 
500 		this.w = SparseMatrixFactoryMTJ.INSTANCE.createMatrix(nwords, ntasks);
501 		for (int t = 0; t < ntasks; t++) {
502 			for (int r = 0; r < nwords; r++) {
503 				final double readDouble = in.readDouble();
504 				if(readDouble != 0){

507 			}
508 		}

510 		this.u = SparseMatrixFactoryMTJ.INSTANCE.createMatrix(nusers, ntasks);
511 		for (int t = 0; t < ntasks; t++) {
512 			for (int r = 0; r < nusers; r++) {

554 	@Override
555 	public Matrix predict(Matrix x) {
556 		final Matrix mult = this.u.transpose().times(x.transpose()).times(this.w);
557 		if(this.biasMode)mult.plusEquals(this.bias);
558 		final Vector ydiag = CFMatrixUtils.diag(mult);
559 		final Matrix createIdentity = SparseMatrixFactoryMTJ.INSTANCE.createIdentity(1, ydiag.getDimensionality());
560 		createIdentity.setRow(0, ydiag);
561 		return createIdentity;
562 	}
