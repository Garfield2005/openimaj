72  	 */
73  	public static final String UINITSTRAT = "uinitstrat";
74  	/**
75  	 * The initialisation strategy for W when it is expanded. Defaults to the
76  	 * context aware {@link CurrentWMean}
77  	 */
78  	public static final String EXPANDEDWINITSTRAT = "expandedwinitstrat";
79  	/**
80  	 * The initialisation strategy for U when it is expanded. Defaults to the
81  	 * context aware {@link CurrentUMean}
82  	 */
83  	public static final String EXPANDEDUINITSTRAT = "expandeduinitstrat";
84  	/**
85  	 * The initialisation strategy for BIAS. Defaults to a
86  	 * {@link SparseZerosInitStrategy}
87  	 */
88  	public static final String BIASINITSTRAT = "biasinitstrat";
89  	/**
90  	 * The maximum number of iterations in the biconvex iterative stage.
91  	 * defaults to 3
92  	 */
93  	public static final String BICONVEX_MAXITER = "biconvex_maxiter";
94  	/**
95  	 * The threshold of the ratio between the (sum(new_w - old_w) + sum(new_u -
96  	 * old_u)) / (sum(old_u) + sum(old_w)) i.e. some notion of normalised
97  	 * changed of the paramters. Defaults to 0.01
98  	 */
99  	public static final String BICONVEX_TOL = "biconvex_tol";
100 	/**
101 	 * The parameter of the regulariser for W, defaults to LAMBDA
102 	 */
103 	public static final String LAMBDA_W = "lambda_w";
104 	/**
105 	 * The parameter of the regulariser for U, defaults to LAMBDA
106 	 */
107 	public static final String LAMBDA_U = "lambda_u";
108 	/**
109 	 * The parameter of the regulariser for both W and U
110 	 */
111 	public static final String LAMBDA = "lambda";
112 	/**
113 	 * The weighting of the subgradient of U, weighted down each ETASTEPS number
114 	 * of iterations of the biconvex scheme, defaults to 0.05
115 	 */
116 	public static final String ETA0_U = "eta0u";
117 	/**
118 	 * The weighting of the subgradient of W, weighted down each ETASTEPS number
119 	 * of iterations of the biconvex scheme, defaults to 0.05
120 	 */
121 	public static final String ETA0_W = "eta0w";
122 	/**
123 	 * The weighting of the subgradient of BIAS, weighted down each ETASTEPS
124 	 * number of iterations of the biconvex scheme, defaults to eta0 (0.05)

126 	public static final String ETA0_BIAS = "biaseta0";
127 	/**
128 	 * The loss function, defaults to {@link SquareMissingLossFunction}
129 	 */
130 	public static final String LOSS = "loss";
131 	/**

133 	 */
134 	public static final String REGUL = "regul";

168 	/**
169 	 * sets up the defaults
170 	 */
171 	public BilinearLearnerParameters() {
172 		this.defaults.put(REGUL, new L1L2Regulariser());
173 		this.defaults.put(LOSS, new SquareMissingLossFunction());
174 		this.defaults.put(ETA0_U, 0.05);
175 		this.defaults.put(ETA0_W, 0.05);
176 		this.defaults.put(LAMBDA, 0.001);
177 		this.defaults.put(LAMBDA_W, new Placeholder(LAMBDA));
178 		this.defaults.put(LAMBDA_U, new Placeholder(LAMBDA));
179 		this.defaults.put(BICONVEX_TOL, 0.01);
180 		this.defaults.put(BICONVEX_MAXITER, 3);
181 		this.defaults.put(SEED, -1);
182 		this.defaults.put(WINITSTRAT, new SparseRandomInitStrategy(0, 1, 0.5, new Random()));
183 		this.defaults.put(UINITSTRAT, new SparseRandomInitStrategy(0, 1, 0.5, new Random()));
184 		this.defaults.put(EXPANDEDWINITSTRAT, new CurrentWMean());
185 		this.defaults.put(EXPANDEDUINITSTRAT, new CurrentUMean());
186 		this.defaults.put(BIAS, false);
187 		this.defaults.put(BIASINITSTRAT, new SparseZerosInitStrategy());
188 		this.defaults.put(ETA0_BIAS, 0.05);
189 		this.defaults.put(ETASTEPS, 3);
190 		this.defaults.put(FORCE_SPARCITY, true);
191 		this.defaults.put(DAMPENING, 0d);
192 		this.defaults.put(Z_STANDARDISE, false);
193 		this.defaults.put(ETA_GAMMA,1.5);
194 	}

196 	@Override
197 	public void writeASCII(PrintWriter out) throws IOException {
198 		final Set<String> a = new HashSet<String>(this.keySet());
199 		a.addAll(this.defaults.keySet());
200 		for (final String key : a) {
201 			out.printf("%s: %s\n", key, this.getTyped(key));
202 		}
203 	}
204 	
205 	@Override
206 	public String toString() {
207 		StringWriter writer = new StringWriter();
208 		PrintWriter pw = new PrintWriter(writer, true);
209 		try {
210 			writeASCII(pw);
211 		} catch (IOException e) {
212 		}
213 		pw.flush();
214 		return writer.toString();
215 	}
