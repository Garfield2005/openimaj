030package org.openimaj.ml.clustering.spectral;
031
032import java.util.Iterator;
033
034import org.apache.log4j.Logger;
035import org.openimaj.ml.clustering.DataClusterer;
036import org.openimaj.ml.clustering.IndexClusters;
037import org.openimaj.ml.clustering.SpatialClusterer;
038import org.openimaj.ml.clustering.SpatialClusters;
039import org.openimaj.util.pair.DoubleObjectPair;
040import org.openimaj.util.pair.IndependentPair;
041
042import ch.akuhn.matrix.Vector;
043import ch.akuhn.matrix.Vector.Entry;
044import ch.akuhn.matrix.eigenvalues.Eigenvalues;
045
046/**
047 * For a given set of {@link Eigenvalues} perform the stages of spectral
048 * clustering which involve the selection of the best eigen values and the
049 * calling of an internal clustering algorithm
050 *
051 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
052 */
053public class PreparedSpectralClustering implements DataClusterer<Eigenvalues, SpectralIndexedClusters> {
054        final static Logger logger = Logger.getLogger(PreparedSpectralClustering.class);
055        private SpectralClusteringConf<double[]> conf;
056
057        /**
058         * @param conf
059         */
060        public PreparedSpectralClustering(SpectralClusteringConf<double[]> conf) {
061                this.conf = conf;
062        }
063
064        @Override
065        public int[][] performClustering(Eigenvalues data) {
066                return cluster(data).clusters();
067        }
068
069        @Override
070        public SpectralIndexedClusters cluster(Eigenvalues eig) {
071                // Also normalise each row
072                final IndependentPair<double[], double[][]> lowestCols = bestCols(eig);
073                // Using the eigenspace, cluster
074                return eigenspaceCluster(lowestCols);
075        }
076
077        protected SpectralIndexedClusters eigenspaceCluster(IndependentPair<double[], double[][]> lowestCols) {
078                final SpatialClusterer<? extends SpatialClusters<double[]>, double[]> clusterer = conf.internal.apply(lowestCols);
079                // Cluster the rows with the internal spatial clusterer
080                final SpatialClusters<double[]> cluster = clusterer.cluster(lowestCols.getSecondObject());
081                // if the clusters contain the cluster indexes of the training examples
082                // use those
083                if (cluster instanceof IndexClusters) {
084                        final IndexClusters clusters = new IndexClusters(((IndexClusters) cluster).clusters());
085                        // logger.debug(clusters);
086                        return new SpectralIndexedClusters(clusters, lowestCols);
087                }
088                // Otherwise attempt to assign values to clusters
089                final int[] clustered = cluster.defaultHardAssigner().assign(lowestCols.getSecondObject());
090                // done!
091                return new SpectralIndexedClusters(new IndexClusters(clustered), lowestCols);
092        }
093
094        protected IndependentPair<double[], double[][]> bestCols(final Eigenvalues eig) {
095
096                int eigenVectorSelect = conf.eigenChooser.nEigenVectors(this.conf.laplacian.eigenIterator(eig), eig.getN());
097                final int eigenVectorSkip = this.conf.skipEigenVectors;
098                logger.debug("Selected dimensions: " + eigenVectorSelect);
099                logger.debug("Skipping dimesions: " + eigenVectorSkip);
100                eigenVectorSelect -= eigenVectorSkip;
101
102                final int nrows = eig.vector[0].size();
103                final double[][] ret = new double[nrows][eigenVectorSelect];
104                final double[] retSum = new double[nrows];
105                final double[] eigvals = new double[eigenVectorSelect];
106                final Iterator<DoubleObjectPair<Vector>> iterator = this.conf.laplacian.eigenIterator(eig);
107                // Skip a few at the beggining
108                for (int i = 0; i < eigenVectorSkip; i++)
109                        iterator.next();
110                int col = 0;
111                // Calculate U matrix (containing n smallests eigen valued columns)
112                for (; iterator.hasNext();) {
113                        final DoubleObjectPair<Vector> v = iterator.next();
114                        eigvals[col] = v.first;
115
116                        for (final Entry d : v.second.entries()) {
117                                double elColI = d.value;
118                                if (conf.eigenValueScale) {
119                                        elColI *= Math.sqrt(v.first);
120                                }
121                                ret[d.index][col] = elColI;
122                                retSum[d.index] += elColI * elColI;
123                        }
124                        col++;
125                        if (col == eigenVectorSelect)
126                                break;
127                }
128
129                if (!conf.eigenValueScale) {
130                        // normalise rows
131                        for (int i = 0; i < ret.length; i++) {
132                                final double[] row = ret[i];
133                                for (int j = 0; j < row.length; j++) {
134                                        row[j] /= Math.sqrt(retSum[i]);
135                                }
136                        }
137                }
138
139                return IndependentPair.pair(eigvals, ret);
140        }
141
142}
