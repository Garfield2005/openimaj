32  /**
33   * An annotation representing a phrase as per the Penn Treebank.
34   * 
35   * @author Laurence Willmore (lgw1e10@ecs.soton.ac.uk)
36   * 
37   */
38  public class PhraseAnnotation extends TextPipeAnnotation {
39  
40  	/**
41  	 * Penn Treebank phrase abbreviations.
42  	 * 
43  	 * @author Laurence Willmore (lgw1e10@ecs.soton.ac.uk)
44  	 * 
45  	 */
46  	public enum Phrase {
47  		ADJP("Adjective Phrase."),
48  		ADVP("Adverb Phrase."),
49  		CONJP("Conjunction Phrase."),
50  		FRAG("Fragment."),
51  		INTJ("Interjection. Corresponds approximately to the part-of-speech tag UH."),
52  		LST("List marker. Includes surrounding punctuation."),
53  		NAC("Not a Constituent; used to show the scope of certain prenominal modifiers within an NP."),
54  		NP("Noun Phrase. "),
55  		NX(
56  				"Used within certain complex NPs to mark the head of the NP. Corresponds very roughly to N-bar level but used quite differently."),
57  		PP("Prepositional Phrase."),
58  		PRN("Parenthetical. "),
59  		PRT("Particle. Category for words that should be tagged RP. "),
60  		QP("Quantifier Phrase (i.e. complex measure/amount phrase); used within NP."),
61  		RRC("Reduced Relative Clause. "),
62  		UCP("Unlike Coordinated Phrase. "),
63  		VP("Vereb Phrase. "),
64  		WHADJP("Wh-adjective Phrase. Adjectival phrase containing a wh-adverb, as in how hot."),
65  		WHAVP(
66  				"Wh-adverb Phrase. Introduces a clause with an NP gap. May be null (containing the 0 complementizer) or lexical, containing a wh-adverb such as how or why."),
67  		WHNP(
68  				"Wh-noun Phrase. Introduces a clause with an NP gap. May be null (containing the 0 complementizer) or lexical, containing some wh-word, e.g. who, which book, whose daughter, none of which, or how many leopards."),
69  		WHPP(
70  				"Wh-prepositional Phrase. Prepositional phrase containing a wh-noun phrase (such as of which or by whose authority) that either introduces a PP gap or is contained by a WHNP."),
71  		X(
72  				"Unknown, uncertain, or unbracketable. X is often used for bracketing typos and in bracketing the...the-constructions."),
73  		/**
74  		 * This is added for phrases with no mapping to penn bank.
75  		 */
76  		UK("Unknown");
77  
78  		/**
79  		 * Penn Tree Bank short description
80  		 */
81  		public final String DESCRIPTION;
82  
83  		Phrase(String description) {
84  			this.DESCRIPTION = description;
85  		}
86  
87  		/**
88  		 * Returns a {@link Phrase} based on the string.
89  		 * 
90  		 * @param pennAbreviation
91  		 * @return {@link Phrase}
92  		 */
93  		public static Phrase getPhrasefromString(String pennAbreviation) {
94  			for (final Phrase pos : Phrase.values()) {
95  				if (pos.toString().equals(pennAbreviation))
96  					return pos;
97  			}
98  			return Phrase.UK;
99  		}
100 	};
101 
102 	/**
103 	 * The {@link Phrase} label.
104 	 */
105 	public Phrase phrase;
106 	/**
107 	 * true if this is the start token of a phrase segment. false if it is a
108 	 * continuation.
109 	 */
110 	public boolean start;
111 
112 	public PhraseAnnotation(Phrase phrase, boolean start) {
113 		super();
114 		this.phrase = phrase;
115 		this.start = start;
116 	}
117 
118 	/**
119 	 * Returns a string representation of the Phrase order of this phrase.
120 	 * 
121 	 * @return String order.
122 	 */
123 	public String getOrder() {
124 		if (start)
125 			return "start";
126 		else
127 			return "continue";
128 	}
