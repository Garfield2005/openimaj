38  /**
39   * Match edge punctuations and correct them such that they can be matched by a
40   * simple space split
41   * 
42   * @author Sina Samangooei (ss@ecs.soton.ac.uk)
43   * 
44   */
45  public abstract class EdgePunctuationPatternProvider extends PatternProvider {
46  
47  	protected String EdgePunct;
48  	protected String NotEdgePunct;
49  	protected String StartPunc;
50  	protected String EndPunc;
51  
52  	/**
53  	 * @param punctuation
54  	 */
55  	public EdgePunctuationPatternProvider(PunctuationPatternProvider punctuation) {
56  		final String[] EdgePunctArr = new String[] { "'", "\"", "\\*", "\u201c", "\u201d", "\u2018", "\u2019", "\\<",
57  				"\\>", "\u00AB", "\u00BB", "{", "}", "\\(", "\\)", "\\[", "\\]", "\\\\", "\\|" };
58  		final HashSet<String> edgeSet = new HashSet<String>();
59  		for (final String string : EdgePunctArr) {
60  			edgeSet.add(string);
61  		}
62  		EdgePunct = "[" + StringUtils.join(EdgePunctArr, "") + "]";
63  		final List<String> puncArr = new ArrayList<String>();
64  		for (final String punc : punctuation.PunctCharsList) {
65  			if (edgeSet.contains(punc))
66  				continue;
67  			puncArr.add(punc);
68  		}
69  
70  		NotEdgePunct = "(?:[a-zA-Z0-9]|" + "[" + StringUtils.join(puncArr, "") + "\\-]" + ")";
71  		// NotEdgePunct = "(?:[a-zA-Z0-9])";
72  		StartPunc = "\\s|^|[.,]|" + "[a-zA-Z0-9]";
73  		EndPunc = "\\s|$|[.,]|" + "[a-zA-Z0-9]";
74  	}
75  
76  	/**
77  	 * @return the edge punctuation of the default
78  	 *         {@link EdgePunctuationPatternProvider}
79  	 */
80  	public static String edgePuncPattern() {
81  		return new EdgePunctuationPatternProvider(new PunctuationPatternProvider()) {
82  			@Override
83  			public String correctEdges(String s) {
84  				return null;
85  			}
86  
87  			@Override
88  			public String patternString() {
89  				return null;
90  			}
91  
92  		}.EdgePunct;
93  	}
