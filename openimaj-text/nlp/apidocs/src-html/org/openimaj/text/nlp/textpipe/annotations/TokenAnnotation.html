030package org.openimaj.text.nlp.textpipe.annotations;
031
032import java.util.ArrayList;
033import java.util.List;
034
035import org.apache.commons.lang.StringUtils;
036import org.openimaj.text.nlp.tokenisation.ReversableToken;
037
038/**
039 * A {@link TokenAnnotation} represents a single token generated by a tokeniser.
040 * It can be further annotated. It is aware of its off-sets in the
041 * RawStringAnnotation that generated it.
042 * 
043 * @author Laurence Willmore (lgw1e10@ecs.soton.ac.uk)
044 * 
045 */
046public class TokenAnnotation extends TextPipeAnnotation implements ReversableToken {
047        String stringToken;
048        String raw;
049        public int start;
050        public int stop;
051
052        public TokenAnnotation(String token,String raw, int start, int stop) {
053                super();
054                this.stringToken = token;
055                this.start = start;
056                this.stop = stop;
057                this.raw=raw;
058        }
059
060        public String getStringToken() {
061                return stringToken;
062        }       
063
064        @Override
065        public String getRawString() {
066                return raw;
067        }
068
069        @Override
070        public String reverse(List<? extends ReversableToken> tokens) {
071                List<String> raws = new ArrayList<String>();
072                for(ReversableToken tok:tokens){
073                        raws.add(tok.getRawString());
074                }
075                return StringUtils.join(raws,"");
076        }
077
078}
