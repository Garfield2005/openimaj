030package org.openimaj.text.util;
031
032import java.util.List;
033
034import org.apache.commons.lang.StringUtils;
035import org.openimaj.text.nlp.patterns.PatternProvider;
036
037/**
038 * A collection of utility functions for regex construction
039 * 
040 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
041 * 
042 */
043public class RegexUtil {
044
045        /**
046         * @param items
047         * @return match within brackets a set of "or"ed items
048         */
049        public static String regex_or_match(String... items) {
050                String r = StringUtils.join(items, "|");
051                r = '(' + r + ')';
052                return r;
053        }
054
055        /**
056         * @param patterns
057         * @return match wtihin brackets a set of "or"ed patterns (calling
058         *         {@link PatternProvider#patternString()})
059         */
060        public static String regex_or_match(PatternProvider... patterns) {
061                final String[] allpat = new String[patterns.length];
062                int i = 0;
063                for (final PatternProvider patternProvider : patterns) {
064                        allpat[i++] = patternProvider.patternString();
065                }
066                return regex_or_match(allpat);
067        }
068
069        /**
070         * @param items
071         * @return using {@link StringUtils#join(java.util.Collection, String)} or
072         *         together a list of strings
073         */
074        public static String regex_or_match(List<String> items) {
075                String r = StringUtils.join(items, "|");
076                r = '(' + r + ')';
077                return r;
078        }
079
080        /**
081         * @param puncs
082         * @return negate
083         */
084        public static String regex_char_neg(List<String> puncs) {
085                String r = StringUtils.join(puncs, "");
086                r = "[" + r + ']';
087                return r;
088        }
089
090        /**
091         * @param r
092         * @return construct a positive equality lookahead
093         */
094        public static String pos_lookahead(String r) {
095                return "(?=" + r + ')';
096        }
097
098        /**
099         * @param r
100         * @return construct a negative equality lookahead
101         */
102        public static String neg_lookahead(String r) {
103                return "(?!" + r + ')';
104        }
105
106        /**
107         * @param r
108         * @return construct an optional
109         */
110        public static String optional(String r) {
111                return String.format("(%s)?", r);
112        }
113}
