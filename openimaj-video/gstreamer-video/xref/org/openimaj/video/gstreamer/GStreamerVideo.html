62  	@Override
63  	public synchronized MBFImage getNextFrame() {
64  		if (isStopped)
65  			return frame;
66  
67  		final boolean err = gstreamer.nextFrame();
68  		if (!err)
69  			throw new RuntimeException(new GStreamerException("Error getting next frame"));
70  
71  		final Pointer<Byte> data = gstreamer.getImage();
72  		if (data == null) {
73  			return frame;
74  		}
75  
76  		if (gstreamer.getBands() == 3) {
77  			final byte[] d = data.getBytes(width * height * 3);
78  			final float[][] r = frame.bands.get(0).pixels;
79  			final float[][] g = frame.bands.get(1).pixels;
80  			final float[][] b = frame.bands.get(2).pixels;
81  
82  			for (int i = 0, y = 0; y < height; y++) {
83  				for (int x = 0; x < width; x++, i += 3) {
84  					final int blue = d[i + 0] & 0xFF;
85  					final int green = d[i + 1] & 0xFF;
86  					final int red = d[i + 2] & 0xFF;
87  					r[y][x] = ImageUtilities.BYTE_TO_FLOAT_LUT[red];
88  					g[y][x] = ImageUtilities.BYTE_TO_FLOAT_LUT[green];
89  					b[y][x] = ImageUtilities.BYTE_TO_FLOAT_LUT[blue];
90  				}
91  			}
92  		} else {
93  			final byte[] d = data.getBytes(width * height * 3);
94  			final float[][] g = frame.bands.get(0).pixels;
95  
96  			for (int i = 0, y = 0; y < height; y++) {
97  				for (int x = 0; x < width; x++, i++) {
98  					g[y][x] = ImageUtilities.BYTE_TO_FLOAT_LUT[d[i] & 0xFF];
99  				}
100 			}
101 		}
102 
103 		super.currentFrame++;
104 
105 		if (captureStartedTimestamp == 0)
106 			captureStartedTimestamp = System.currentTimeMillis();
107 		currentTimestamp = System.currentTimeMillis() - captureStartedTimestamp;
108 
109 		return frame;
110 	}
