058        /**
059         * Within a search window around the subimages detect most likely match and
060         * thus motion.
061         *
062         * @author Sina Samangooei (ss@ecs.soton.ac.uk)
063         *
064         */
065        public static class TEMPLATE_MATCH extends MotionEstimatorAlgorithm {
066                private float searchProp;
067                private Mode mode;
068
069                /**
070                 * Defaults to allowing a maximum of templatesize/2 movement using the
071                 * {@link Mode#CORRELATION}
072                 */
073                public TEMPLATE_MATCH() {
074                        this.searchProp = .5f;
075                        this.mode = TemplateMatcher.Mode.NORM_SUM_SQUARED_DIFFERENCE;
076                }
077
078                /**
079                 * Given the template's size, search around a border of size
080                 * template*searchWindowBorderProp
081                 *
082                 * @param searchWindowBorderProp
083                 * @param mode
084                 *            the matching mode
085                 */
086                public TEMPLATE_MATCH(float searchWindowBorderProp, Mode mode) {
087                        this.searchProp = searchWindowBorderProp;
088                        this.mode = mode;
089                }
090
091                @Override
092                Point2d estimateMotion(VideoSubFrame<FImage> img1sub,
093                                VideoSubFrame<FImage>... imagesSub)
094                {
095
096                        final VideoFrame<FImage> current = img1sub.extract();
097                        final VideoFrame<FImage> prev = imagesSub[0];
098                        final Rectangle prevSearchRect = imagesSub[0].roi;
099
100                        final int sw = (int) (prevSearchRect.width * this.searchProp);
101                        final int sh = (int) (prevSearchRect.height * this.searchProp);
102                        final int sx = (int) (prevSearchRect.x + ((prevSearchRect.width - sw) / 2.f));
103                        final int sy = (int) (prevSearchRect.y + ((prevSearchRect.height - sh) / 2.f));
104
105                        final Rectangle searchRect = new Rectangle(sx, sy, sw, sh);
106                        // System.out.println("Search window: " + searchRect);
107                        // MBFImage searchRectDraw = new
108                        // MBFImage(img1sub.frame.clone(),img1sub.frame.clone(),img1sub.frame.clone());
109                        // searchRectDraw.drawShape(searchRect, RGBColour.RED);
110                        // searchRectDraw.drawPoint(img1sub.roi.getCOG(), RGBColour.GREEN,
111                        // 3);
112                        final TemplateMatcher matcher = new TemplateMatcher(current.frame, mode, searchRect);
113                        matcher.analyseImage(prev.frame);
114                        final FValuePixel[] responses = matcher.getBestResponses(1);
115                        final FValuePixel firstBest = responses[0];
116                        // for (FValuePixel bestRespose : responses) {
117                        // if(bestRespose == null) continue;
118                        // if(firstBest == null) firstBest = bestRespose;
119                        // bestRespose.translate(current.frame.width/2,
120                        // current.frame.height/2);
121                        // // searchRectDraw.drawPoint(bestRespose, RGBColour.BLUE, 3);
122                        // }
123                        final Point2d centerOfGrid = img1sub.roi.calculateCentroid();
124                        // System.out.println("First reponse: " + firstBest );
125                        // System.out.println("Center of template: " + centerOfGrid);
126
127                        // DisplayUtilities.displayName(searchRectDraw, "searchWindow");
128                        if (firstBest == null)
129                                return new Point2dImpl(0, 0);
130                        // firstBest.translate(current.frame.width/2,
131                        // current.frame.height/2);
132                        // System.out.println("First reponse (corrected): " + firstBest );
133                        // System.out.println("Diff: " + centerOfGrid.minus(firstBest));
134                        return centerOfGrid.minus(firstBest);
135                }

137        }

139        /**
140         * Basic phase correlation algorithm that finds peaks in the cross-power
141         * spectrum between two images. This is the basic implementation without
142         * sub-pixel accuracy.
143         */
144        public static class PHASE_CORRELATION extends MotionEstimatorAlgorithm
145        {
146                /**
147                 * Calculate the estimated motion vector between <code>images</code>
148                 * which [0] is first in the sequence and <code>img2</code> which is
149                 * second in the sequence. This method uses phase correlation - the fact
150                 * that translations in space can be seen as shifts in phase in the
151                 * frequency domain. The returned vector will have a maximum horizontal
152                 * displacement of <code>img2.getWidth()/2</code> and a minimum
153                 * displacement of <code>-img2.getWidth()/2</code> and similarly for the
154                 * vertical displacement and height.
155                 *
156                 * @param img2sub
157                 *            The second image in the sequence
158                 * @param imagesSub
159                 *            The previous image in the sequence
160                 * @return the estimated motion vector as a {@link Point2d} in absolute
161                 *         x and y coordinates.
162                 */
163                @Override
164                public Point2d estimateMotion(VideoSubFrame<FImage> img2sub,
165                                VideoSubFrame<FImage>... imagesSub)
166                {
167                        // The previous image will be the first in the images array
168                        final FImage img1 = imagesSub[0].extract().frame;
169                        final VideoFrame<FImage> img2 = img2sub.extract();
170
171                        // No previous frame?
172                        if (img1 == null)
173                                return new Point2dImpl(0, 0);
174
175                        // The images must have comparable shapes and must be square
176                        if (img1.getRows() != img2.frame.getRows() ||
177                                        img1.getCols() != img2.frame.getCols() ||
178                                        img1.getCols() != img2.frame.getRows())
179                                return new Point2dImpl(0, 0);
180
181                        // Prepare and perform an FFT for each of the incoming images.
182                        final int h = img1.getRows();
183                        final int w = img1.getCols();
184
185                        try
186                        {
187                                final FloatFFT_2D fft1 = new FloatFFT_2D(h, w);
188                                final FloatFFT_2D fft2 = new FloatFFT_2D(h, w);
189                                final float[][] data1 = FourierTransform.prepareData(img1, h, w, false);
190                                final float[][] data2 = FourierTransform.prepareData(img2.frame, h, w, false);
191                                fft1.complexForward(data1);
192                                fft2.complexForward(data2);
193
194                                // Multiply (element-wise) the fft and the conjugate of the fft.
195                                Complex[][] cfft = new Complex[h][w];
196                                for (int y = 0; y < h; y++)
197                                {
198                                        for (int x = 0; x < w; x++)
199                                        {
200                                                final float re1 = data1[y][x * 2];
201                                                final float im1 = data1[y][1 + x * 2];
202                                                final float re2 = data2[y][x * 2];
203                                                final float im2 = data2[y][1 + x * 2];
204
205                                                final Complex c1 = new Complex(re1, im1);
206                                                final Complex c2 = new Complex(re2, -im2);
207                                                cfft[y][x] = c1.multiply(c2);
208                                        }
209                                }
210
211                                // ----------------------------------------
212                                // Normalise by the determinant
213                                // ----------------------------------------
214                                // First calculate the determinant
215                                final Array2DRowFieldMatrix<Complex> cmat =
216                                                new Array2DRowFieldMatrix<Complex>(cfft);
217                                final FieldLUDecompositionImpl<Complex> luDecomp =
218                                                new FieldLUDecompositionImpl<Complex>(cmat);
219                                final Complex det = luDecomp.getDeterminant();
220                                cmat.scalarMultiply(new Complex(1d / det.abs(), 0));
221
222                                // Convert back to an array for doing the inverse FFTing
223                                cfft = cmat.getData();
224                                for (int y = 0; y < h; y++)
225                                {
226                                        for (int x = 0; x < w; x++)
227                                        {
228                                                data1[y][x * 2] = (float) cfft[y][x].getReal();
229                                                data1[y][1 + x * 2] = (float) cfft[y][x].getImaginary();
230                                        }
231                                }
232
233                                // Perform the inverse FFT
234                                fft1.complexInverse(data1, false);
235
236                                // Get the data back out
237                                FourierTransform.unprepareData(data1, img1, false);
238
239                                // Get the estimated motion vector from the peak in the space
240                                final FValuePixel p = img1.maxPixel();
241                                return new Point2dImpl(
242                                                -(p.x > w / 2 ? p.x - w : p.x),
243                                                -(p.y > w / 2 ? p.y - w : p.y));
244                        } catch (final Exception e)
245                        {
246                                return new Point2dImpl(0, 0);
247                        }
248                }
249        };

251        /**
252         * Estimate the motion to the given subimage, <code>img1sub</code> from the
253         * previous frames. The previous frames will be given in reverse order so
254         * that imagesSub[0] will be the previous frame, imagesSub[1] the frame
255         * before that, etc. The number of frames given will be at most that given
256         * by {@link #requiredNumberOfFrames()}. It could be less if at the
257         * beginning of the video. If you require more frames, return an empty
258         * motion vector - that is (0,0).
259         *
260         * @param img1sub
261         *            The image to which we want to estimate the motion.
262         * @param imagesSub
263         *            The previous frames in reverse order
264         * @return The estimated motion vector.
265         */
266        abstract Point2d estimateMotion(VideoSubFrame<FImage> img1sub,
267                        VideoSubFrame<FImage>... imagesSub);
268
269        /**
270         * The required number of frames required for the given motion estimation
271         * algorithm to work. The default is 1 which means the algorithm will only
272         * receive the previous frame. If more are required, override this method
273         * and return the required number.
274         *
275         * @return The required number of frames to pass to the algorithm.
276         */
277        protected int requiredNumberOfFrames()
278        {
279                return 1;
280        }
